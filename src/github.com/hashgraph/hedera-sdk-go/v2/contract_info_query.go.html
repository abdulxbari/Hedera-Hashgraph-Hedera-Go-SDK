   164  func (query *ContractInfoQuery) Execute(client *Client) (ContractInfo, error) {
   165  	if client == nil || client.operator == nil {
   166  		return ContractInfo{}, errNoClientProvided
   167  	}
   168  
   169  	var err error
   170  
   171  	err = query._ValidateNetworkOnIDs(client)
   172  	if err != nil {
   173  		return ContractInfo{}, err
   174  	}
   175  
   176  	if !query.paymentTransactionIDs.locked {
   177  		query.paymentTransactionIDs._Clear()._Push(TransactionIDGenerate(client.operator.accountID))
   178  	}
   179  
   180  	var cost Hbar
   181  	if query.queryPayment.tinybar != 0 {
   182  		cost = query.queryPayment
   183  	} else {
   184  		if query.maxQueryPayment.tinybar == 0 {
   185  			cost = client.GetDefaultMaxQueryPayment()
   186  		} else {
   187  			cost = query.maxQueryPayment
   188  		}
   189  
   190  		actualCost, err := query.GetCost(client)
   191  		if err != nil {
   192  			return ContractInfo{}, err
   193  		}
   194  
   195  		if cost.tinybar < actualCost.tinybar {
   196  			return ContractInfo{}, ErrMaxQueryPaymentExceeded{
   197  				QueryCost:       actualCost,
   198  				MaxQueryPayment: cost,
   199  				query:           "ContractInfoQuery",
   200  			}
   201  		}
   202  
   203  		cost = actualCost
   204  	}
   205  
   206  	query.paymentTransactions = make([]*services.Transaction, 0)
   207  
   208  	if query.nodeAccountIDs.locked {
   209  		err = _QueryGeneratePayments(&query.Query, client, cost)
   210  		if err != nil {
   211  			if err != nil {
   212  				return ContractInfo{}, err
   213  			}
   214  		}
   215  	} else {
   216  		paymentTransaction, err := _QueryMakePaymentTransaction(query.paymentTransactionIDs._GetCurrent().(TransactionID), AccountID{}, client.operator, cost)
   217  		if err != nil {
   218  			if err != nil {
   219  				return ContractInfo{}, err
   220  			}
   221  		}
   222  		query.paymentTransactions = append(query.paymentTransactions, paymentTransaction)
   223  	}
   224  
   225  	pb := query._Build()
   226  	pb.ContractGetInfo.Header = query.pbHeader
   227  	query.pb = &services.Query{
   228  		Query: pb,
   229  	}
   230  
   231  	resp, err := _Execute(
   232  		client,
   233  		&query.Query,
   234  		_ContractInfoQueryShouldRetry,
   235  		_QueryMakeRequest,
   236  		_QueryAdvanceRequest,
   237  		_QueryGetNodeAccountID,
   238  		_ContractInfoQueryGetMethod,
   239  		_ContractInfoQueryMapStatusError,
   240  		_QueryMapResponse,
   241  		query._GetLogID(),
   242  		query.grpcDeadline,
   243  		query.maxBackoff,
   244  		query.minBackoff,
   245  		query.maxRetry,
   246  	)
   247  
   248  	if err != nil {
   249  		return ContractInfo{}, err
   250  	}
   251  
   252  	info, err := _ContractInfoFromProtobuf(resp.(*services.Response).GetContractGetInfo().ContractInfo)
   253  	if err != nil {
   254  		return ContractInfo{}, err
   255  	}
   256  
   257  	return info, nil
   258  }
