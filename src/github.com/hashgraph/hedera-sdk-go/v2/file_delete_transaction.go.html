   344  func (transaction *FileDeleteTransaction) AddSignature(publicKey PublicKey, signature []byte) *FileDeleteTransaction {
   345  	transaction._RequireOneNodeAccountID()
   346  
   347  	if transaction._KeyAlreadySigned(publicKey) {
   348  		return transaction
   349  	}
   350  
   351  	if transaction.signedTransactions._Length() == 0 {
   352  		return transaction
   353  	}
   354  
   355  	transaction.transactions = _NewLockableSlice()
   356  	transaction.publicKeys = append(transaction.publicKeys, publicKey)
   357  	transaction.transactionSigners = append(transaction.transactionSigners, nil)
   358  	transaction.transactionIDs.locked = true
   359  
   360  	for index := 0; index < transaction.signedTransactions._Length(); index++ {
   361  		var temp *services.SignedTransaction
   362  		switch t := transaction.signedTransactions._Get(index).(type) { //nolint
   363  		case *services.SignedTransaction:
   364  			temp = t
   365  		}
   366  		temp.SigMap.SigPair = append(
   367  			temp.SigMap.SigPair,
   368  			publicKey._ToSignaturePairProtobuf(signature),
   369  		)
   370  		transaction.signedTransactions._Set(index, temp)
   371  	}
   372  
   373  	return transaction
