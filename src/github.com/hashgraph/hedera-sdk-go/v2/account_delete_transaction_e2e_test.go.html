    35  func TestIntegrationAccountDeleteTransactionCanExecute(t *testing.T) {
    36  	env := NewIntegrationTestEnv(t)
    37  
    38  	newKey, err := PrivateKeyGenerateEd25519()
    39  	require.NoError(t, err)
    40  
    41  	newBalance := NewHbar(2)
    42  
    43  	assert.Equal(t, 2*HbarUnits.Hbar._NumberOfTinybar(), newBalance.tinybar)
    44  
    45  	resp, err := NewAccountCreateTransaction().
    46  		SetKey(newKey.PublicKey()).
    47  		SetNodeAccountIDs(env.NodeAccountIDs).
    48  		SetInitialBalance(newBalance).
    49  		Execute(env.Client)
    50  	require.NoError(t, err)
    51  
    52  	receipt, err := resp.SetValidateStatus(true).GetReceipt(env.Client)
    53  	require.NoError(t, err)
    54  
    55  	accountID := receipt.AccountID
    56  	require.NoError(t, err)
    57  
    58  	tx, err := NewAccountDeleteTransaction().
    59  		SetAccountID(*accountID).
    60  		SetTransferAccountID(env.Client.GetOperatorAccountID()).
    61  		SetNodeAccountIDs([]AccountID{resp.NodeID}).
    62  		SetTransactionID(TransactionIDGenerate(*accountID)).
    63  		FreezeWith(env.Client)
    64  	require.NoError(t, err)
    65  
    66  	tx = tx.Sign(newKey)
    67  
    68  	assert.True(t, newKey.PublicKey().VerifyTransaction(tx.Transaction))
    69  	assert.False(t, env.Client.GetOperatorPublicKey().VerifyTransaction(tx.Transaction))
    70  
    71  	resp, err = tx.Execute(env.Client)
    72  	require.NoError(t, err)
    73  
    74  	_, err = resp.SetValidateStatus(true).GetReceipt(env.Client)
    75  	require.NoError(t, err)
    76  
    77  	err = CloseIntegrationTestEnv(env, nil)
    78  	require.NoError(t, err)
    79  }
