    71  func NewIntegrationTestEnv(t *testing.T) IntegrationTestEnv {
    72  	var env IntegrationTestEnv
    73  	var err error
    74  
    75  	if os.Getenv("HEDERA_NETWORK") == "previewnet" { // nolint
    76  		env.Client = ClientForPreviewnet()
    77  	} else if os.Getenv("HEDERA_NETWORK") == "localhost" {
    78  		network := make(map[string]AccountID)
    79  		network["127.0.0.1:50213"] = AccountID{Account: 3}
    80  
    81  		env.Client = ClientForNetwork(network)
    82  		env.Client.cancelNetworkUpdate()
    83  	} else if os.Getenv("HEDERA_NETWORK") == "testnet" {
    84  		env.Client = ClientForTestnet()
    85  	} else if os.Getenv("CONFIG_FILE") != "" {
    86  		env.Client, err = ClientFromConfigFile(os.Getenv("CONFIG_FILE"))
    87  		if err != nil {
    88  			panic(err)
    89  		}
    90  	} else {
    91  		panic("Failed to construct client from environment variables")
    92  	}
    93  
    94  	configOperatorID := os.Getenv("OPERATOR_ID")
    95  	configOperatorKey := os.Getenv("OPERATOR_KEY")
    96  
    97  	if configOperatorID != "" && configOperatorKey != "" {
    98  		env.OperatorID, err = AccountIDFromString(configOperatorID)
    99  		require.NoError(t, err)
   100  
   101  		env.OperatorKey, err = PrivateKeyFromString(configOperatorKey)
   102  		require.NoError(t, err)
   103  
   104  		env.Client.SetOperator(env.OperatorID, env.OperatorKey)
   105  	}
   106  
   107  	assert.NotNil(t, env.Client.GetOperatorAccountID())
   108  	assert.NotNil(t, env.Client.GetOperatorPublicKey())
   109  
   110  	newKey, err := PrivateKeyGenerateEd25519()
   111  	require.NoError(t, err)
   112  
   113  	env.Client.SetMaxNodeAttempts(1)
   114  	env.Client.SetMinBackoff(250 * time.Millisecond)
   115  	env.Client.SetMaxBackoff(8 * time.Second)
   116  	env.Client.SetNodeMinReadmitPeriod(5 * time.Second)
   117  	env.Client.SetNodeMaxReadmitPeriod(1 * time.Hour)
   118  	env.Client.SetMaxAttempts(100)
   119  	env.Client.PingAll()
   120  
   121  	network := make(map[string]AccountID)
   122  
   123  	for key, value := range env.Client.GetNetwork() {
   124  		_, err = NewAccountBalanceQuery().
   125  			SetNodeAccountIDs([]AccountID{value}).
   126  			SetAccountID(value).
   127  			Execute(env.Client)
   128  
   129  		if err != nil {
   130  			println(err.Error())
   131  			continue
   132  		}
   133  
   134  		network[key] = value
   135  		break
   136  	}
   137  
   138  	_ = env.Client.SetNetwork(network)
   139  
   140  	if len(network) == 0 {
   141  		panic("failed to construct network; each node returned an error")
   142  	}
   143  
   144  	resp, err := NewAccountCreateTransaction().
   145  		SetKey(newKey.PublicKey()).
   146  		SetInitialBalance(NewHbar(50)).
   147  		SetAutoRenewPeriod(time.Hour*24*81 + time.Minute*26 + time.Second*39).
   148  		Execute(env.Client)
   149  	if err != nil {
   150  		panic(err)
   151  	}
   152  
   153  	receipt, err := resp.SetValidateStatus(true).GetReceipt(env.Client)
   154  	if err != nil {
   155  		panic(err)
   156  	}
   157  
   158  	env.OriginalOperatorID = env.Client.GetOperatorAccountID()
   159  	env.OriginalOperatorKey = env.Client.GetOperatorPublicKey()
   160  	env.OperatorID = *receipt.AccountID
   161  	env.OperatorKey = newKey
   162  	env.NodeAccountIDs = []AccountID{resp.NodeID}
   163  	env.Client.SetOperator(env.OperatorID, env.OperatorKey)
   164  
   165  	return env
   166  }

   168  func CloseIntegrationTestEnv(env IntegrationTestEnv, token *TokenID) error {
   169  	var resp TransactionResponse
   170  	var err error
   171  	if token != nil {
   172  		deleteTokenTx, err := NewTokenDeleteTransaction().
   173  			SetNodeAccountIDs(env.NodeAccountIDs).
   174  			SetTokenID(*token).
   175  			FreezeWith(env.Client)
   176  		if err != nil {
   177  			return err
   178  		}
   179  
   180  		resp, err = deleteTokenTx.
   181  			Sign(env.OperatorKey).
   182  			Execute(env.Client)
   183  		if err != nil {
   184  			return err
   185  		}
   186  
   187  		_, err = resp.SetValidateStatus(true).GetReceipt(env.Client)
   188  		if err != nil {
   189  			return err
   190  		}
   191  
   192  		// This is needed, because we can't delete the account while still having tokens.
   193  		// This works only, because the token is deleted, otherwise the acount would need to have 0 balance of it before dissociating.
   194  		dissociateTx, err := NewTokenDissociateTransaction().
   195  			SetAccountID(env.Client.operator.accountID).
   196  			SetNodeAccountIDs(env.NodeAccountIDs).
   197  			AddTokenID(*token).
   198  			Execute(env.Client)
   199  		if err != nil {
   200  			return err
   201  		}
   202  
   203  		_, err = dissociateTx.SetValidateStatus(true).GetReceipt(env.Client)
   204  		if err != nil {
   205  			return err
   206  		}
   207  	}
   208  	resp, err = NewAccountDeleteTransaction().
   209  		SetNodeAccountIDs(env.NodeAccountIDs).
   210  		SetAccountID(env.OperatorID).
   211  		SetTransferAccountID(env.OriginalOperatorID).
   212  		Execute(env.Client)
   213  	if err != nil {
   214  		return err
   215  	}
   216  
   217  	_, err = resp.SetValidateStatus(true).GetReceipt(env.Client)
   218  	if err != nil {
   219  		return err
   220  	}
   221  
   222  	return env.Client.Close()
   223  }
