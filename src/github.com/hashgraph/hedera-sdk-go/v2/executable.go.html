    92  func _Execute( // nolint
    93  	client *Client,
    94  	request interface{},
    95  	shouldRetry func(string, interface{}, interface{}) _ExecutionState,
    96  	makeRequest func(interface{}) interface{},
    97  	advanceRequest func(interface{}),
    98  	getNodeAccountID func(interface{}) AccountID,
    99  	getMethod func(interface{}, *_Channel) _Method,
   100  	mapStatusError func(interface{}, interface{}) error,
   101  	mapResponse func(interface{}, interface{}, AccountID, interface{}) (interface{}, error),
   102  	logID string,
   103  	deadline *time.Duration,
   104  	maxBackoff *time.Duration,
   105  	minBackoff *time.Duration,
   106  	maxRetry int,
   107  ) (interface{}, error) {
   108  	var maxAttempts int
   109  	backOff := backoff.NewExponentialBackOff()
   110  	backOff.InitialInterval = *minBackoff
   111  	backOff.MaxInterval = *maxBackoff
   112  	backOff.Multiplier = 2
   113  
   114  	if client.maxAttempts != nil {
   115  		maxAttempts = *client.maxAttempts
   116  	} else {
   117  		maxAttempts = maxRetry
   118  	}
   119  
   120  	currentBackoff := minBackoff
   121  
   122  	var attempt int64
   123  	var errPersistent error
   124  	var marshaledRequest []byte
   125  
   126  	for attempt = int64(0); attempt < int64(maxAttempts); attempt, *currentBackoff = attempt+1, *currentBackoff*2 {
   127  		var protoRequest interface{}
   128  		var node *_Node
   129  
   130  		if transaction, ok := request.(*Transaction); ok {
   131  			if attempt > 0 && transaction.nodeAccountIDs._Length() > 1 {
   132  				advanceRequest(request)
   133  			}
   134  
   135  			protoRequest = makeRequest(request)
   136  			nodeAccountID := getNodeAccountID(request)
   137  			if node, ok = client.network._GetNodeForAccountID(nodeAccountID); !ok {
   138  				return TransactionResponse{}, ErrInvalidNodeAccountIDSet{nodeAccountID}
   139  			}
   140  
   141  			marshaledRequest, _ = protobuf.Marshal(protoRequest.(*services.Transaction))
   142  		} else if query, ok := request.(*Query); ok {
   143  			if query.nodeAccountIDs.locked && query.nodeAccountIDs._Length() > 0 {
   144  				protoRequest = makeRequest(request)
   145  				nodeAccountID := getNodeAccountID(request)
   146  				if node, ok = client.network._GetNodeForAccountID(nodeAccountID); !ok {
   147  					return &services.Response{}, ErrInvalidNodeAccountIDSet{nodeAccountID}
   148  				}
   149  			} else {
   150  				node = client.network._GetNode()
   151  				if len(query.paymentTransactions) > 0 {
   152  					var paymentTransaction services.TransactionBody
   153  					_ = protobuf.Unmarshal(query.paymentTransactions[0].BodyBytes, &paymentTransaction) // nolint
   154  					paymentTransaction.NodeAccountID = node.accountID._ToProtobuf()
   155  
   156  					transferTx := paymentTransaction.Data.(*services.TransactionBody_CryptoTransfer)
   157  					transferTx.CryptoTransfer.Transfers.AccountAmounts[0].AccountID = node.accountID._ToProtobuf()
   158  					query.paymentTransactions[0].BodyBytes, _ = protobuf.Marshal(&paymentTransaction) // nolint
   159  
   160  					signature := client.operator.signer(query.paymentTransactions[0].BodyBytes) // nolint
   161  					sigPairs := make([]*services.SignaturePair, 0)
   162  					sigPairs = append(sigPairs, client.operator.publicKey._ToSignaturePairProtobuf(signature))
   163  
   164  					query.paymentTransactions[0].SigMap = &services.SignatureMap{ // nolint
   165  						SigPair: sigPairs,
   166  					}
   167  				}
   168  				query.nodeAccountIDs._Set(0, node.accountID)
   169  				protoRequest = makeRequest(request)
   170  			}
   171  			marshaledRequest, _ = protobuf.Marshal(protoRequest.(*services.Query))
   172  		}
   173  
   174  		node._InUse()
   175  
   176  		logCtx.Trace().Str("requestId", logID).Str("nodeAccountID", node.accountID.String()).Str("nodeIPAddress", node.address._String()).
   177  			Str("Request Proto:", hex.EncodeToString(marshaledRequest)).Msg("executing")
   178  
   179  		if !node._IsHealthy() {
   180  			logCtx.Trace().Str("requestId", logID).Str("delay", node._Wait().String()).Msg("node is unhealthy, waiting before continuing")
   181  			_DelayForAttempt(logID, backOff.NextBackOff(), attempt)
   182  			continue
   183  		}
   184  
   185  		logCtx.Trace().Str("requestId", logID).Msg("updating node account ID index")
   186  
   187  		channel, err := node._GetChannel()
   188  		if err != nil {
   189  			client.network._IncreaseBackoff(node)
   190  			continue
   191  		}
   192  
   193  		advanceRequest(request)
   194  
   195  		method := getMethod(request, channel)
   196  
   197  		var resp interface{}
   198  
   199  		ctx := context.TODO()
   200  		var cancel context.CancelFunc
   201  		if deadline != nil {
   202  			grpcDeadline := time.Now().Add(*deadline)
   203  			ctx, cancel = context.WithDeadline(ctx, grpcDeadline)
   204  		}
   205  
   206  		logCtx.Trace().Str("requestId", logID).Msg("executing gRPC call")
   207  
   208  		var marshaledResponse []byte
   209  		if method.query != nil {
   210  			resp, err = method.query(ctx, protoRequest.(*services.Query))
   211  			if err == nil {
   212  				marshaledResponse, _ = protobuf.Marshal(resp.(*services.Response))
   213  			}
   214  		} else {
   215  			resp, err = method.transaction(ctx, protoRequest.(*services.Transaction))
   216  			if err == nil {
   217  				marshaledResponse, _ = protobuf.Marshal(resp.(*services.TransactionResponse))
   218  			}
   219  		}
   220  
   221  		if cancel != nil {
   222  			cancel()
   223  		}
   224  		if err != nil {
   225  			errPersistent = err
   226  			if _ExecutableDefaultRetryHandler(logID, err) {
   227  				client.network._IncreaseBackoff(node)
   228  				continue
   229  			}
   230  			if errPersistent == nil {
   231  				errPersistent = errors.New("error")
   232  			}
   233  
   234  			if _, ok := request.(*Transaction); ok {
   235  				return TransactionResponse{}, errors.Wrapf(errPersistent, "retry %d/%d", attempt, maxAttempts)
   236  			}
   237  
   238  			return &services.Response{}, errors.Wrapf(errPersistent, "retry %d/%d", attempt, maxAttempts)
   239  		}
   240  
   241  		node._DecreaseBackoff()
   242  		switch shouldRetry(logID, request, resp) {
   243  		case executionStateRetry:
   244  			errPersistent = mapStatusError(request, resp)
   245  			_DelayForAttempt(logID, backOff.NextBackOff(), attempt)
   246  			continue
   247  		case executionStateExpired:
   248  			if transaction, ok := request.(*Transaction); ok {
   249  				if !client.GetOperatorAccountID()._IsZero() && transaction.regenerateTransactionID && !transaction.transactionIDs.locked {
   250  					logCtx.Trace().Str("requestId", logID).Msg("received `TRANSACTION_EXPIRED` with transaction ID regeneration enabled; regenerating")
   251  					transaction.transactionIDs._Set(transaction.transactionIDs.index, TransactionIDGenerate(client.GetOperatorAccountID()))
   252  					if err != nil {
   253  						panic(err)
   254  					}
   255  					continue
   256  				} else {
   257  					return TransactionResponse{}, mapStatusError(request, resp)
   258  				}
   259  			} else {
   260  				return &services.Response{}, mapStatusError(request, resp)
   261  			}
   262  		case executionStateError:
   263  			if _, ok := request.(*Transaction); ok {
   264  				return TransactionResponse{}, mapStatusError(request, resp)
   265  			}
   266  
   267  			return &services.Response{}, mapStatusError(request, resp)
   268  		case executionStateFinished:
   269  			logCtx.Trace().Str("Response Proto:", hex.EncodeToString(marshaledResponse)).Msg("finished")
   270  
   271  			return mapResponse(request, resp, node.accountID, protoRequest)
   272  		}
   273  	}
   274  
   275  	if errPersistent == nil {
   276  		errPersistent = errors.New("error")
   277  	}
   278  
   279  	if _, ok := request.(*Transaction); ok {
   280  		return TransactionResponse{}, errors.Wrapf(errPersistent, "retry %d/%d", attempt, maxAttempts)
   281  	}
   282  
   283  	return &services.Response{}, errPersistent
   284  }

   286  func _DelayForAttempt(logID string, backoff time.Duration, attempt int64) {
   287  	logCtx.Trace().Str("requestId", logID).Dur("delay", backoff).Int64("attempt", attempt+1).Msg("retrying request attempt")
   288  	time.Sleep(backoff)
   289  }

   291  func _ExecutableDefaultRetryHandler(logID string, err error) bool {
   292  	code := status.Code(err)
   293  	logCtx.Trace().Str("requestId", logID).Str("status", code.String()).Msg("received gRPC error with status code")
   294  	switch code {
   295  	case codes.ResourceExhausted, codes.Unavailable:
   296  		return true
   297  	case codes.Internal:
   298  		grpcErr, ok := status.FromError(err)
   299  
   300  		if !ok {
   301  			return false
   302  		}
   303  
   304  		return rstStream.Match([]byte(grpcErr.Message()))
   305  	default:
   306  		return false
   307  	}
