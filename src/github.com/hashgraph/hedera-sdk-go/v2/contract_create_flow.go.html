   330  func (this *ContractCreateFlow) Execute(client *Client) (TransactionResponse, error) {
   331  	this._SplitBytecode()
   332  
   333  	if len(this.appendBytecode) > 0 {
   334  		fileCreateResponse, err := this._CreateFileCreateTransaction(client).Execute(client)
   335  		if err != nil {
   336  			return TransactionResponse{}, err
   337  		}
   338  		fileCreateReceipt, err := this._CreateTransactionReceiptQuery(fileCreateResponse).Execute(client)
   339  		if err != nil {
   340  			return TransactionResponse{}, err
   341  		}
   342  		if fileCreateReceipt.FileID == nil {
   343  			return TransactionResponse{}, errors.New("fileID is nil")
   344  		}
   345  		fileID := *fileCreateReceipt.FileID
   346  
   347  		fileAppendResponse, err := this._CreateFileAppendTransaction(fileID).Execute(client)
   348  		if err != nil {
   349  			return TransactionResponse{}, err
   350  		}
   351  
   352  		_, err = this._CreateTransactionReceiptQuery(fileAppendResponse).Execute(client)
   353  		if err != nil {
   354  			return TransactionResponse{}, err
   355  		}
   356  
   357  		contractCreateResponse, err := this._CreateContractCreateTransaction(fileID).Execute(client)
   358  		if err != nil {
   359  			return TransactionResponse{}, err
   360  		}
   361  		_, err = this._CreateTransactionReceiptQuery(contractCreateResponse).Execute(client)
   362  		if err != nil {
   363  			return TransactionResponse{}, err
   364  		}
   365  
   366  		return contractCreateResponse, nil
   367  	}
   368  
   369  	contractCreateResponse, err := this._CreateContractCreateTransactionWithBytecode().Execute(client)
   370  	if err != nil {
   371  		return TransactionResponse{}, err
   372  	}
   373  	_, err = this._CreateTransactionReceiptQuery(contractCreateResponse).
   374  		Execute(client)
   375  	if err != nil {
   376  		return TransactionResponse{}, err
   377  	}
   378  
   379  	return contractCreateResponse, nil
   380  }
