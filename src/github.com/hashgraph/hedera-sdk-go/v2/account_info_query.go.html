   196  func (query *AccountInfoQuery) Execute(client *Client) (AccountInfo, error) {
   197  	if client == nil || client.operator == nil {
   198  		return AccountInfo{}, errNoClientProvided
   199  	}
   200  
   201  	var err error
   202  
   203  	err = query._ValidateNetworkOnIDs(client)
   204  	if err != nil {
   205  		return AccountInfo{}, err
   206  	}
   207  
   208  	if !query.paymentTransactionIDs.locked {
   209  		query.paymentTransactionIDs._Clear()._Push(TransactionIDGenerate(client.operator.accountID))
   210  	}
   211  
   212  	var cost Hbar
   213  	if query.queryPayment.tinybar != 0 {
   214  		cost = query.queryPayment
   215  	} else {
   216  		if query.maxQueryPayment.tinybar == 0 {
   217  			cost = client.GetDefaultMaxQueryPayment()
   218  		} else {
   219  			cost = query.maxQueryPayment
   220  		}
   221  
   222  		actualCost, err := query.GetCost(client)
   223  		if err != nil {
   224  			return AccountInfo{}, err
   225  		}
   226  
   227  		if cost.tinybar < actualCost.tinybar {
   228  			return AccountInfo{}, ErrMaxQueryPaymentExceeded{
   229  				QueryCost:       actualCost,
   230  				MaxQueryPayment: cost,
   231  				query:           "AccountInfoQuery",
   232  			}
   233  		}
   234  
   235  		cost = actualCost
   236  	}
   237  
   238  	query.paymentTransactions = make([]*services.Transaction, 0)
   239  	if query.nodeAccountIDs.locked {
   240  		err = _QueryGeneratePayments(&query.Query, client, cost)
   241  		if err != nil {
   242  			return AccountInfo{}, err
   243  		}
   244  	} else {
   245  		paymentTransaction, err := _QueryMakePaymentTransaction(query.paymentTransactionIDs._GetCurrent().(TransactionID), AccountID{}, client.operator, cost)
   246  		if err != nil {
   247  			return AccountInfo{}, err
   248  		}
   249  		query.paymentTransactions = append(query.paymentTransactions, paymentTransaction)
   250  	}
   251  
   252  	pb := query._Build()
   253  	pb.CryptoGetInfo.Header = query.pbHeader
   254  	query.pb = &services.Query{
   255  		Query: pb,
   256  	}
   257  
   258  	resp, err := _Execute(
   259  		client,
   260  		&query.Query,
   261  		_AccountInfoQueryShouldRetry,
   262  		_QueryMakeRequest,
   263  		_QueryAdvanceRequest,
   264  		_QueryGetNodeAccountID,

   270  		query.maxBackoff,

   272  		query.maxRetry,
   273  	)

   277  	}
