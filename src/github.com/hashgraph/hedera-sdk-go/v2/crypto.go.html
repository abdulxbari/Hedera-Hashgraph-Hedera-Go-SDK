    49  func _KeyFromProtobuf(pbKey *services.Key) (Key, error) {
    50  	if pbKey == nil {
    51  		return PublicKey{}, errParameterNull
    52  	}
    53  	switch key := pbKey.GetKey().(type) {
    54  	case *services.Key_Ed25519:
    55  		return PublicKeyFromBytesEd25519(key.Ed25519)
    56  
    57  	case *services.Key_ThresholdKey:
    58  		threshold := int(key.ThresholdKey.GetThreshold())
    59  		keys, err := _KeyListFromProtobuf(key.ThresholdKey.GetKeys())
    60  		if err != nil {

    62  		}

    64  
    65  		return &keys, nil
    66  
    67  	case *services.Key_KeyList:
    68  		keys, err := _KeyListFromProtobuf(key.KeyList)
    69  		if err != nil {
    70  			return nil, err
    71  		}
    72  
    73  		return &keys, nil
    74  
    75  	case *services.Key_ContractID:
    76  		return _ContractIDFromProtobuf(key.ContractID), nil
    77  
    78  	case *services.Key_ECDSASecp256K1:
    79  		return PublicKeyFromBytesECDSA(key.ECDSASecp256K1)
    80

   425  func _DeriveEd25519ChildKey(parentKey []byte, chainCode []byte, index uint32) ([]byte, []byte, error) {
   426  	if IsHardenedIndex(index) {
   427  		return nil, nil, errors.New("the index should not be pre-hardened")
   428  	}
   429  
   430  	h := hmac.New(sha512.New, chainCode)
   431  
   432  	input := make([]byte, 37)
   433  
   434  	// 0x00 + parentKey + _Index(BE)
   435  	input[0] = 0
   436  
   437  	copy(input[1:37], parentKey)
   438  
   439  	binary.BigEndian.PutUint32(input[33:37], index)
   440  
   441  	// harden the input
   442  	input[33] |= 128
   443  
   444  	if _, err := h.Write(input); err != nil {
   445  		return nil, nil, err
   446  	}
   447  
   448  	digest := h.Sum(nil)
   449  
   450  	return digest[0:32], digest[32:], nil
   451  }

   453  func _DeriveECDSAChildKey(parentKey []byte, chainCode []byte, index uint32) ([]byte, []byte, error) {
   454  	h := hmac.New(sha512.New, chainCode)
   455  
   456  	isHardened := IsHardenedIndex(index)
   457  	input := make([]byte, 37)
   458  	key, err := crypto.ToECDSA(parentKey)
   459  	if err != nil {
   460  		return nil, nil, err
   461  	}
   462  
   463  	if isHardened {
   464  		offset := 33 - len(parentKey)
   465  		copy(input[offset:], parentKey)
   466  	} else {
   467  		pubKey := crypto.CompressPubkey(&key.PublicKey)
   468  		copy(input, pubKey)
   469  	}
   470  
   471  	binary.BigEndian.PutUint32(input[33:37], index)
   472  
   473  	if _, err := h.Write(input); err != nil {
   474  		return nil, nil, err
   475  	}
   476  
   477  	i := h.Sum(nil)
   478  
   479  	il := new(big.Int)
   480  	il.SetBytes(i[0:32])
   481  	ir := i[32:]
   482  
   483  	ki := new(big.Int)
   484  	ki.Add(key.D, il)
   485  	ki.Mod(ki, key.Curve.Params().N)
   486  
   487  	return ki.Bytes(), ir, nil
   488  }

   490  func _DeriveLegacyChildKey(parentKey []byte, index int64) ([]byte, error) {
   491  	in := make([]uint8, 8)
   492  
   493  	switch switchIndex := index; {
   494  	case switchIndex == int64(0xffffffffff):
   495  		in = []uint8{0, 0, 0, 0xff, 0xff, 0xff, 0xff, 0xff}
   496  	case switchIndex > 0xffffffff:
   497  		return nil, errors.New("derive index is out of range")
   498  	default:
   499  		if switchIndex < 0 {
   500  			for i := 0; i < 4; i++ {
   501  				in[i] = uint8(0xff)
   502  			}
   503  		}
   504  
   505  		for i := 4; i < len(in); i++ {
   506  			in[i] = uint8(switchIndex)
   507  		}
   508  	}
   509  
   510  	password := make([]uint8, len(parentKey))
   511  	copy(password, parentKey)
   512  	password = append(password, in...)
   513  
   514  	salt := []byte{0xFF}
   515
