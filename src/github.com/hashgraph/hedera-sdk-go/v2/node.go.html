   122  func (node *_Node) _GetChannel() (*_Channel, error) {
   123  	if node.channel != nil {
   124  		return node.channel, nil
   125  	}
   126  
   127  	var kacp = keepalive.ClientParameters{
   128  		Time:                10 * time.Second,
   129  		Timeout:             2 * time.Second,
   130  		PermitWithoutStream: true,
   131  	}
   132  
   133  	var conn *grpc.ClientConn
   134  	var err error
   135  	security := grpc.WithInsecure() //nolint
   136  	if !node.verifyCertificate {
   137  		println("skipping certificate check")
   138  	}
   139  	if node._ManagedNode.address._IsTransportSecurity() {
   140  		security = grpc.WithTransportCredentials(credentials.NewTLS(&tls.Config{
   141  			InsecureSkipVerify: true, // nolint
   142  			VerifyPeerCertificate: func(rawCerts [][]byte, verifiedChains [][]*x509.Certificate) error {
   143  				if node.addressBook == nil {
   144  					logCtx.Warn().Msg("skipping certificate check since no cert hash was found")
   145  					return nil
   146  				}
   147  
   148  				if !node.verifyCertificate {
   149  					return nil
   150  				}
   151  
   152  				for _, cert := range rawCerts {
   153  					var certHash []byte
   154  
   155  					block := &pem.Block{
   156  						Type:  "CERTIFICATE",
   157  						Bytes: cert,
   158  					}
   159  
   160  					var encodedBuf bytes.Buffer
   161  					_ = pem.Encode(&encodedBuf, block)
   162  					digest := sha512.New384()
   163  
   164  					if _, err = digest.Write(encodedBuf.Bytes()); err != nil {
   165  						return err
   166  					}
   167  
   168  					certHash = digest.Sum(nil)
   169  
   170  					if string(node.addressBook.CertHash) == hex.EncodeToString(certHash) {
   171  						return nil
   172  					}
   173  				}
   174  
   175  				return x509.CertificateInvalidError{
   176  					Cert:   nil,
   177  					Reason: x509.Expired,
   178  					Detail: "",
   179  				}
   180  			},
   181  		}))
   182  	}
   183  
   184  	cont, cancel := context.WithTimeout(context.Background(), 10*time.Second)
   185  	defer cancel()
   186  	conn, err = grpc.DialContext(cont, node._ManagedNode.address._String(), security, grpc.WithKeepaliveParams(kacp), grpc.WithBlock())
   187  	if err != nil {
   188  		return nil, status.Error(codes.ResourceExhausted, "dial timeout of 10sec exceeded")
   189  	}
   190  
   191  	ch := _NewChannel(conn)
   192  	node.channel = &ch
   193  
   194  	return node.channel, nil
   195  }
