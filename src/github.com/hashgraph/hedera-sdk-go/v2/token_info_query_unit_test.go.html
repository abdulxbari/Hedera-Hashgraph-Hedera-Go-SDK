    81  func TestUnitTokenInfoQueryGet(t *testing.T) {
    82  	tokenID := TokenID{Token: 7}
    83  	deadline := time.Duration(time.Minute)
    84  	accountId := AccountID{Account: 123}
    85  	validStart := time.Now().Add(10 * time.Minute)
    86  	balance := NewTokenInfoQuery().
    87  		SetTokenID(tokenID).
    88  		SetQueryPayment(NewHbar(2)).
    89  		SetMaxQueryPayment(NewHbar(1)).
    90  		SetQueryPayment(HbarFromTinybar(25)).
    91  		SetNodeAccountIDs([]AccountID{{Account: 10}, {Account: 11}, {Account: 12}}).
    92  		SetMaxRetry(3).
    93  		SetMinBackoff(300 * time.Millisecond).
    94  		SetMaxBackoff(10 * time.Second).
    95  		SetPaymentTransactionID(TransactionID{AccountID: &accountId, ValidStart: &validStart}).
    96  		SetMaxQueryPayment(NewHbar(500)).
    97  		SetGrpcDeadline(&deadline)
    98  
    99  	require.Equal(t, tokenID, balance.GetTokenID())
   100  	require.Equal(t, []AccountID{{Account: 10}, {Account: 11}, {Account: 12}}, balance.GetNodeAccountIDs())
   101  	require.Equal(t, 300*time.Millisecond, balance.GetMinBackoff())
   102  	require.Equal(t, 10*time.Second, balance.GetMaxBackoff())
   103  	require.Equal(t, 3, balance.GetMaxRetryCount())
   104  	require.Equal(t, TransactionID{AccountID: &accountId, ValidStart: &validStart}, balance.GetPaymentTransactionID())
   105  	require.Equal(t, HbarFromTinybar(25), balance.GetQueryPayment())
   106  	require.Equal(t, NewHbar(500), balance.GetMaxQueryPayment())
   107  }

   124  	deadline := time.Second * 3

   128  
   129  	client := ClientForTestnet()

   131  
   132  	query := NewTokenInfoQuery().
   133  		SetTokenID(token).
   134  		SetMaxRetry(3).
   135  		SetMaxBackoff(time.Second * 30).
   136  		SetMinBackoff(time.Second * 10).
   137  		SetNodeAccountIDs(nodeAccountID).
   138  		SetPaymentTransactionID(transactionID).
   139  		SetMaxQueryPayment(NewHbar(23)).
   140  		SetQueryPayment(NewHbar(3)).
   141  		SetGrpcDeadline(&deadline)
   142  
   143  	err := query._ValidateNetworkOnIDs(client)
   144  	require.NoError(t, err)
   145  
   146  	require.Equal(t, nodeAccountID, query.GetNodeAccountIDs())
   147  	require.Equal(t, time.Second*30, query.GetMaxBackoff())
   148  	require.Equal(t, time.Second*10, query.GetMinBackoff())
   149  	require.Equal(t, token, query.GetTokenID())
   150  	require.Equal(t, NewHbar(3), query.GetQueryPayment())
   151  	require.Equal(t, NewHbar(23), query.GetMaxQueryPayment())
   152  	require.Equal(t, &deadline, query.GetGrpcDeadline())
   153  	require.Equal(t, fmt.Sprintf("TokenInfoQuery:%v", transactionID.ValidStart.UnixNano()), query._GetLogID())
