    82  		require.NotEmpty(t, request.SignedTransactionBytes)

    84  		_ = protobuf.Unmarshal(request.SignedTransactionBytes, &signedTransaction)
    85  
    86  		require.NotEmpty(t, signedTransaction.BodyBytes)
    87  		transactionBody := services.TransactionBody{}
    88  		_ = protobuf.Unmarshal(signedTransaction.BodyBytes, &transactionBody)
    89  
    90  		require.NotNil(t, transactionBody.TransactionID)
    91  		transactionId := transactionBody.TransactionID.String()
    92  		require.NotEqual(t, "", transactionId)
    93

    95  		require.NotNil(t, sigMap)
    96  
    97  		for _, sigPair := range sigMap.SigPair {
    98  			verified := false
    99  
   100  			switch k := sigPair.Signature.(type) {
   101  			case *services.SignaturePair_Ed25519:
   102  				pbTemp, _ := PublicKeyFromBytesEd25519(sigPair.PubKeyPrefix)
   103  				verified = pbTemp.Verify(signedTransaction.BodyBytes, k.Ed25519)
   104  			case *services.SignaturePair_ECDSASecp256K1:
   105  				pbTemp, _ := PublicKeyFromBytesECDSA(sigPair.PubKeyPrefix)
   106  				verified = pbTemp.Verify(signedTransaction.BodyBytes, k.ECDSASecp256K1)
   107  			}

   111  		if bod, ok := transactionBody.Data.(*services.TransactionBody_ContractUpdateInstance); ok {

   117  		}
   118  
   119  		return &services.TransactionResponse{
   120  			NodeTransactionPrecheckCode: services.ResponseCodeEnum_OK,
   121  		}

   125  	}}

   127  	client, server := NewMockClientAndServer(responses)
