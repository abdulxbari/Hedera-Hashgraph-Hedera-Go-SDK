   163  func (query *AccountStakersQuery) Execute(client *Client) ([]Transfer, error) {
   164  	if client == nil || client.operator == nil {
   165  		return []Transfer{}, errNoClientProvided
   166  	}
   167  
   168  	var err error
   169  
   170  	err = query._ValidateNetworkOnIDs(client)
   171  	if err != nil {
   172  		return []Transfer{}, err
   173  	}
   174  
   175  	if !query.paymentTransactionIDs.locked {
   176  		query.paymentTransactionIDs._Clear()._Push(TransactionIDGenerate(client.operator.accountID))
   177  	}
   178  
   179  	var cost Hbar
   180  	if query.queryPayment.tinybar != 0 {
   181  		cost = query.queryPayment
   182  	} else {
   183  		if query.maxQueryPayment.tinybar == 0 {
   184  			cost = client.GetDefaultMaxQueryPayment()
   185  		} else {
   186  			cost = query.maxQueryPayment
   187  		}
   188  
   189  		actualCost, err := query.GetCost(client)
   190  		if err != nil {
   191  			return []Transfer{}, err
   192  		}
   193  
   194  		if cost.tinybar < actualCost.tinybar {
   195  			return []Transfer{}, ErrMaxQueryPaymentExceeded{
   196  				QueryCost:       actualCost,
   197  				MaxQueryPayment: cost,
   198  				query:           "AccountStakersQuery",
   199  			}
   200  		}
   201  
   202  		cost = actualCost
   203  	}
   204  
   205  	query.paymentTransactions = make([]*services.Transaction, 0)
   206  
   207  	if query.nodeAccountIDs.locked {
   208  		err = _QueryGeneratePayments(&query.Query, client, cost)
   209  		if err != nil {
   210  			return []Transfer{}, err
   211  		}
   212  	} else {
   213  		paymentTransaction, err := _QueryMakePaymentTransaction(query.paymentTransactionIDs._GetCurrent().(TransactionID), AccountID{}, client.operator, cost)
   214  		if err != nil {
   215  			return []Transfer{}, err
   216  		}
   217  		query.paymentTransactions = append(query.paymentTransactions, paymentTransaction)
   218  	}
   219  
   220  	pb := query._Build()
   221  	pb.CryptoGetProxyStakers.Header = query.pbHeader
   222  	query.pb = &services.Query{
   223  		Query: pb,
   224  	}
   225  
   226  	resp, err := _Execute(
   227  		client,
   228  		&query.Query,
   229  		_AccountStakersQueryShouldRetry,
   230  		_QueryMakeRequest,
   231  		_QueryAdvanceRequest,
   232  		_QueryGetNodeAccountID,
   233  		_AccountStakersQueryGetMethod,
   234  		_AccountStakersQueryMapStatusError,
   235  		_QueryMapResponse,
   236  		query._GetLogID(),
   237  		query.grpcDeadline,
   238  		query.maxBackoff,
   239  		query.minBackoff,
   240  		query.maxRetry,
   241  	)
   242  
   243  	if err != nil {
   244  		return []Transfer{}, err
   245  	}
   246  
   247  	var stakers = make([]Transfer, len(resp.(*services.Response).GetCryptoGetProxyStakers().Stakers.ProxyStaker))
   248  
   249  	// TODO: This is wrong, this _Method shold return `[]ProxyStaker` not `[]Transfer`
   250  	for i, element := range resp.(*services.Response).GetCryptoGetProxyStakers().Stakers.ProxyStaker {
   251  		id := _AccountIDFromProtobuf(element.AccountID)
   252  		accountID := AccountID{}
   253  
   254  		if id == nil {
   255  			accountID = *id
   256  		}
   257  
   258  		stakers[i] = Transfer{
   259  			AccountID: accountID,
   260  			Amount:    HbarFromTinybar(element.Amount),
   261  		}
   262  	}
   263  
   264  	return stakers, err
   265  }
