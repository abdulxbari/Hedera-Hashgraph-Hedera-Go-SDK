   197  	logCtx.Trace().Str("requestId", logID).Str("status", status.String()).Msg("receipt status received")

   297  func (query *TransactionReceiptQuery) Execute(client *Client) (TransactionReceipt, error) {
   298  	if client == nil {
   299  		return TransactionReceipt{}, errNoClientProvided
   300  	}
   301  
   302  	var err error
   303  
   304  	err = query._ValidateNetworkOnIDs(client)
   305  	if err != nil {
   306  		return TransactionReceipt{}, err
   307  	}
   308  
   309  	query.timestamp = time.Now()
   310  
   311  	query.paymentTransactions = make([]*services.Transaction, 0)
   312  
   313  	pb := query._Build()
   314  	pb.TransactionGetReceipt.Header = query.pbHeader
   315  	query.pb = &services.Query{
   316  		Query: pb,
   317  	}
   318  
   319  	resp, err := _Execute(
   320  		client,
   321  		&query.Query,
   322  		_TransactionReceiptQueryShouldRetry,
   323  		_QueryMakeRequest,
   324  		_QueryAdvanceRequest,
   325  		_QueryGetNodeAccountID,
   326  		_TransactionReceiptQueryGetMethod,
   327  		_TransactionReceiptQueryMapStatusError,
   328  		_QueryMapResponse,
   329  		query._GetLogID(),
   330  		query.grpcDeadline,
   331  		query.maxBackoff,
   332  		query.minBackoff,
   333  		query.maxRetry,
   334  	)
   335  
   336  	if err, ok := err.(ErrHederaPreCheckStatus); ok {
   337  		if resp.(*services.Response).GetTransactionGetReceipt() != nil {
   338  			return _TransactionReceiptFromProtobuf(resp.(*services.Response).GetTransactionGetReceipt(), query.transactionID), err
   339  		}
   340  		// Manually add the receipt status, because an empty receipt has no status and no status defaults to 0, which means success
   341  		return TransactionReceipt{Status: err.Status}, err
   342  	}
   343  
   344  	return _TransactionReceiptFromProtobuf(resp.(*services.Response).GetTransactionGetReceipt(), query.transactionID), nil
   345  }
