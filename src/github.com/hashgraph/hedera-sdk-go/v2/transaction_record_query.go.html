    30  // TransactionRecordQuery
    31  // Get the record for a transaction. If the transaction requested a record, then the record lasts
    32  // for one hour, and a state proof is available for it. If the transaction created an account, file,
    33  // or smart contract instance, then the record will contain the ID for what it created. If the
    34  // transaction called a smart contract function, then the record contains the result of that call.
    35  // If the transaction was a cryptocurrency transfer, then the record includes the TransferList which
    36  // gives the details of that transfer. If the transaction didn't return anything that should be in
    37  // the record, then the results field will be set to nothing.

   305  func (query *TransactionRecordQuery) Execute(client *Client) (TransactionRecord, error) {
   306  	if client == nil || client.operator == nil {
   307  		return TransactionRecord{}, errNoClientProvided
   308  	}
   309  
   310  	var err error
   311  
   312  	err = query._ValidateNetworkOnIDs(client)
   313  	if err != nil {
   314  		return TransactionRecord{}, err
   315  	}
   316  
   317  	if !query.paymentTransactionIDs.locked {
   318  		query.paymentTransactionIDs._Clear()._Push(TransactionIDGenerate(client.operator.accountID))
   319  	}
   320  
   321  	var cost Hbar
   322  	if query.queryPayment.tinybar != 0 {
   323  		cost = query.queryPayment
   324  	} else {
   325  		if query.maxQueryPayment.tinybar == 0 {
   326  			cost = client.GetDefaultMaxQueryPayment()
   327  		} else {
   328  			cost = query.maxQueryPayment
   329  		}
   330  
   331  		actualCost, err := query.GetCost(client)
   332  		if err != nil {
   333  			return TransactionRecord{}, err
   334  		}
   335  
   336  		if cost.tinybar < actualCost.tinybar {
   337  			return TransactionRecord{}, ErrMaxQueryPaymentExceeded{
   338  				QueryCost:       actualCost,
   339  				MaxQueryPayment: cost,
   340  				query:           "TransactionRecordQuery",
   341  			}
   342  		}
   343  
   344  		cost = actualCost
   345  	}
   346  
   347  	query.paymentTransactions = make([]*services.Transaction, 0)
   348  
   349  	if query.nodeAccountIDs.locked {
   350  		err = _QueryGeneratePayments(&query.Query, client, cost)
   351  		if err != nil {
   352  			return TransactionRecord{}, err
   353  		}
   354  	} else {
   355  		paymentTransaction, err := _QueryMakePaymentTransaction(query.paymentTransactionIDs._GetCurrent().(TransactionID), AccountID{}, client.operator, cost)
   356  		if err != nil {
   357  			return TransactionRecord{}, err
   358  		}
   359  		query.paymentTransactions = append(query.paymentTransactions, paymentTransaction)
   360  	}
   361  
   362  	pb := query._Build()
   363  	pb.TransactionGetRecord.Header = query.pbHeader
   364  	query.pb = &services.Query{
   365  		Query: pb,
   366  	}
   367  
   368  	resp, err := _Execute(
   369  		client,
   370  		&query.Query,
   371  		_TransactionRecordQueryShouldRetry,
   372  		_QueryMakeRequest,
   373  		_QueryAdvanceRequest,
   374  		_QueryGetNodeAccountID,
   375  		_TransactionRecordQueryGetMethod,
   376  		_TransactionRecordQueryMapStatusError,
   377  		_QueryMapResponse,
   378  		query._GetLogID(),
   379  		query.grpcDeadline,
   380  		query.maxBackoff,
   381  		query.minBackoff,
   382  		query.maxRetry,
   383  	)
   384  
   385  	if err != nil {
   386  		if precheckErr, ok := err.(ErrHederaPreCheckStatus); ok {
   387  			return TransactionRecord{}, _NewErrHederaReceiptStatus(precheckErr.TxID, precheckErr.Status)
   388  		}
   389  		return TransactionRecord{}, err
   390  	}
   391  
   392  	return _TransactionRecordFromProtobuf(resp.(*services.Response).GetTransactionGetRecord(), query.transactionID), nil
   393  }
