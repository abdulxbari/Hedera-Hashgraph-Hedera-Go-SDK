    31  // ContractFunctionResult is a struct which allows users to convert between solidity and Go types, and is typically
    32  // returned by `ContractCallQuery` and is present in the transaction records of `ContractExecuteTransaction`.
    33  // Use the methods `Get<Type>()` to get a parameter. Not all solidity types
    34  // are supported out of the box, but the most common types are. The larger variants
    35  // of number types return just the bytes for the integer instead of converting to a big int type.
    36  // To convert those bytes into a usable integer using "github.com/ethereum/go-ethereum/common/math" and "math/big" do the following:
    37  // ```
    38  // contractFunctionResult.GetUint256(<index>)
    39  // bInt := new(big.Int)
    40  // bInt.SetBytes(query.GetUint256(0))
    41  // ```
    42  type ContractFunctionResult struct {
    43  	// ContractID is the smart contract instance whose function was called
    44  	ContractID *ContractID

   361  func _ContractFunctionResultFromProtobuf(pb *services.ContractFunctionResult) ContractFunctionResult {
   362  	infos := make([]ContractLogInfo, len(pb.LogInfo))

   364  	for i, info := range pb.LogInfo {

   366  	}
   367  
   368  	createdContractIDs := make([]ContractID, 0)
   369  	for _, id := range pb.CreatedContractIDs { // nolint
   370  		temp := _ContractIDFromProtobuf(id)
   371  		if temp != nil {
   372  			createdContractIDs = append(createdContractIDs, *temp)
   373  		}
   374  	}
   375  
   376  	var evm ContractID
   377  	if len(pb.EvmAddress.GetValue()) > 0 {
   378  		evm = ContractID{
   379  			Shard:      0,
   380  			Realm:      0,
   381  			Contract:   0,
   382  			EvmAddress: pb.EvmAddress.GetValue(),
   383  			checksum:   nil,
   384  		}
   385  	}
   386  
   387  	result := ContractFunctionResult{
   388  		ContractCallResult: pb.ContractCallResult,
   389  		ErrorMessage:       pb.ErrorMessage,
   390  		Bloom:              pb.Bloom,
   391  		GasUsed:            pb.GasUsed,
   392  		LogInfo:            infos,
   393  		CreatedContractIDs: createdContractIDs,
   394  		EvmAddress:         evm,
   395  		GasAvailable:       pb.Gas,

   397  		FunctionParameters: pb.FunctionParameters,
   398  	}
