    32  // AccountCreateTransaction creates a new account. After the account is created, the AccountID for it is in the receipt,
    33  // or by asking for a Record of the transaction to be created, and retrieving that. The account can then automatically
    34  // generate records for large transfers into it or out of it, which each last for 25 hours. Records are generated for
    35  // any transfer that exceeds the thresholds given here. This account is charged hbar for each record generated, so the
    36  // thresholds are useful for limiting Record generation to happen only for large transactions.
    37  //
    38  // The current API ignores shardID, realmID, and newRealmAdminKey, and creates everything in shard 0 and realm 0,
    39  // with a null key. Future versions of the API will support multiple realms and multiple shards.

   432  // Execute executes the Transaction with the provided client
   433  func (transaction *AccountCreateTransaction) Execute(
   434  	client *Client,
   435  ) (TransactionResponse, error) {
   436  	if client == nil {
   437  		return TransactionResponse{}, errNoClientProvided
   438  	}
   439  
   440  	if transaction.freezeError != nil {
   441  		return TransactionResponse{}, transaction.freezeError
   442  	}
   443  
   444  	if !transaction.IsFrozen() {
   445  		_, err := transaction.FreezeWith(client)
   446  		if err != nil {
   447  			return TransactionResponse{}, err
   448  		}
   449  	}
   450  
   451  	transactionID := transaction.transactionIDs._GetCurrent().(TransactionID)
   452  
   453  	if !client.GetOperatorAccountID()._IsZero() && client.GetOperatorAccountID()._Equals(*transactionID.AccountID) {
   454  		transaction.SignWith(
   455  			client.GetOperatorPublicKey(),
   456  			client.operator.signer,
   457  		)
   458  	}
   459  
   460  	if transaction.grpcDeadline == nil {
   461  		transaction.grpcDeadline = client.requestTimeout
   462  	}
   463  
   464  	resp, err := _Execute(
   465  		client,
   466  		&transaction.Transaction,
   467  		_TransactionShouldRetry,
   468  		_TransactionMakeRequest,
   469  		_TransactionAdvanceRequest,
   470  		_TransactionGetNodeAccountID,
   471  		_AccountCreateTransactionGetMethod,
   472  		_TransactionMapStatusError,
   473  		_TransactionMapResponse,
   474  		transaction._GetLogID(),
   475  		transaction.grpcDeadline,
   476  		transaction.maxBackoff,
   477  		transaction.minBackoff,
   478  		transaction.maxRetry,
   479  	)
   480  
   481  	if err != nil {
   482  		return TransactionResponse{
   483  			TransactionID:  transaction.GetTransactionID(),
   484  			NodeID:         resp.(TransactionResponse).NodeID,
   485  			ValidateStatus: true,
   486  		}, err
   487  	}
   488  
   489  	return TransactionResponse{
   490  		TransactionID:  transaction.GetTransactionID(),
   491  		NodeID:         resp.(TransactionResponse).NodeID,
   492  		Hash:           resp.(TransactionResponse).Hash,
   493  		ValidateStatus: true,

   501  func (transaction *AccountCreateTransaction) FreezeWith(client *Client) (*AccountCreateTransaction, error) {
   502  	if transaction.IsFrozen() {
   503  		return transaction, nil
   504  	}
   505  	transaction._InitFee(client)
   506  	if err := transaction._InitTransactionID(client); err != nil {
   507  		return transaction, err
   508  	}
   509  	err := transaction._ValidateNetworkOnIDs(client)

   513  	}
   514  
   515  	return transaction, _TransactionFreezeWith(&transaction.Transaction, client, body)
   516  }
