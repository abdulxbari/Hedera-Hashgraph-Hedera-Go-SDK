    30  // TokenCreateTransaction
    31  // Create a new token. After the token is created, the Token ID for it is in the receipt.
    32  // The specified Treasury Account is receiving the initial supply of tokens as-well as the tokens
    33  // from the Token Mint operation once executed. The balance of the treasury account is decreased
    34  // when the Token Burn operation is executed.
    35  //
    36  // The initialSupply is the initial supply of the smallest parts of a token (like a
    37  // tinybar, not an hbar). These are the smallest units of the token which may be transferred.
    38  //
    39  // The supply can change over time. If the total supply at some moment is S parts of tokens,
    40  // and the token is using D decimals, then S must be less than or equal to
    41  // 2<sup>63</sup>-1, which is 9,223,372,036,854,775,807. The number of whole tokens (not parts) will
    42  // be S / 10<sup>D</sup>.
    43  //
    44  // If decimals is 8 or 11, then the number of whole tokens can be at most a few billions or
    45  // millions, respectively. For example, it could match Bitcoin (21 million whole tokens with 8
    46  // decimals) or hbars (50 billion whole tokens with 8 decimals). It could even match Bitcoin with
    47  // milli-satoshis (21 million whole tokens with 11 decimals).
    48  //
    49  // Note that a created token is immutable if the adminKey is omitted. No property of
    50  // an immutable token can ever change, with the sole exception of its expiry. Anyone can pay to
    51  // extend the expiry time of an immutable token.
    52  //
    53  // A token can be either FUNGIBLE_COMMON or NON_FUNGIBLE_UNIQUE, based on its
    54  // TokenType. If it has been omitted, FUNGIBLE_COMMON type is used.
    55  //
    56  // A token can have either INFINITE or FINITE supply type, based on its
    57  // TokenType. If it has been omitted, INFINITE type is used.
    58  //
    59  // If a FUNGIBLE TokenType is used, initialSupply should explicitly be set to a
    60  // non-negative. If not, the transaction will resolve to INVALID_TOKEN_INITIAL_SUPPLY.
    61  //
    62  // If a NON_FUNGIBLE_UNIQUE TokenType is used, initialSupply should explicitly be set
    63  // to 0. If not, the transaction will resolve to INVALID_TOKEN_INITIAL_SUPPLY.
    64  //
    65  // If an INFINITE TokenSupplyType is used, maxSupply should explicitly be set to 0. If
    66  // it is not 0, the transaction will resolve to INVALID_TOKEN_MAX_SUPPLY.
    67  //
    68  // If a FINITE TokenSupplyType is used, maxSupply should be explicitly set to a
    69  // non-negative value. If it is not, the transaction will resolve to INVALID_TOKEN_MAX_SUPPLY.

   371  func (transaction *TokenCreateTransaction) _ValidateNetworkOnIDs(client *Client) error {
   372  	if client == nil || !client.autoValidateChecksums {
   373  		return nil
   374  	}
   375  
   376  	if transaction.treasuryAccountID != nil {
   377  		if err := transaction.treasuryAccountID.ValidateChecksum(client); err != nil {
   378  			return err
   379  		}
   380  	}
   381  
   382  	if transaction.autoRenewAccountID != nil {
   383  		if err := transaction.autoRenewAccountID.ValidateChecksum(client); err != nil {
   384  			return err
   385  		}
   386  	}
   387  
   388  	for _, customFee := range transaction.customFees {
   389  		if err := customFee._ValidateNetworkOnIDs(client); err != nil {
   390  			return err
   391  		}
   392  	}
   393  
   394  	return nil
   395  }

   694  // Execute executes the Transaction with the provided client
   695  func (transaction *TokenCreateTransaction) Execute(
   696  	client *Client,
   697  ) (TransactionResponse, error) {
   698  	if client == nil {
   699  		return TransactionResponse{}, errNoClientProvided
   700  	}
   701  
   702  	if transaction.freezeError != nil {
   703  		return TransactionResponse{}, transaction.freezeError
   704  	}
   705  
   706  	if !transaction.IsFrozen() {
   707  		_, err := transaction.FreezeWith(client)
   708  		if err != nil {
   709  			return TransactionResponse{}, err
   710  		}
   711  	}
   712  
   713  	transactionID := transaction.transactionIDs._GetCurrent().(TransactionID)
   714  
   715  	if !client.GetOperatorAccountID()._IsZero() && client.GetOperatorAccountID()._Equals(*transactionID.AccountID) {
   716  		transaction.SignWith(
   717  			client.GetOperatorPublicKey(),
   718  			client.operator.signer,
   719  		)
   720  	}
   721  
   722  	resp, err := _Execute(
   723  		client,
   724  		&transaction.Transaction,
   725  		_TransactionShouldRetry,
   726  		_TransactionMakeRequest,
   727  		_TransactionAdvanceRequest,
   728  		_TransactionGetNodeAccountID,
   729  		_TokenCreateTransactionGetMethod,
   730  		_TransactionMapStatusError,
   731  		_TransactionMapResponse,
   732  		transaction._GetLogID(),
   733  		transaction.grpcDeadline,
   734  		transaction.maxBackoff,
   735  		transaction.minBackoff,
   736  		transaction.maxRetry,
   737  	)
   738  
   739  	if err != nil {
   740  		return TransactionResponse{
   741  			TransactionID:  transaction.GetTransactionID(),
   742  			NodeID:         resp.(TransactionResponse).NodeID,
   743  			ValidateStatus: true,
   744  		}, err
   745  	}
   746  
   747  	return TransactionResponse{
   748  		TransactionID:  transaction.GetTransactionID(),
   749  		NodeID:         resp.(TransactionResponse).NodeID,
   750  		Hash:           resp.(TransactionResponse).Hash,
   751  		ValidateStatus: true,
   752  	}, nil

   759  func (transaction *TokenCreateTransaction) FreezeWith(client *Client) (*TokenCreateTransaction, error) {
   760  	if transaction.autoRenewPeriod != nil && client != nil && !client.GetOperatorAccountID()._IsZero() {
   761  		transaction.SetAutoRenewAccount(client.GetOperatorAccountID())
   762  	}
   763  
   764  	if transaction.IsFrozen() {
   765  		return transaction, nil
   766  	}
   767  	transaction._InitFee(client)
   768  	err := transaction._ValidateNetworkOnIDs(client)
   769  	if err != nil {
   770  		return &TokenCreateTransaction{}, err
   771  	}
   772  	if err := transaction._InitTransactionID(client); err != nil {
   773  		return transaction, err
   774  	}
   775  	body := transaction._Build()
   776  
   777  	return transaction, _TransactionFreezeWith(&transaction.Transaction, client, body)
   778  }

   849  func (transaction *TokenCreateTransaction) AddSignature(publicKey PublicKey, signature []byte) *TokenCreateTransaction {
   850  	transaction._RequireOneNodeAccountID()
   851  
   852  	if transaction._KeyAlreadySigned(publicKey) {
   853  		return transaction
   854  	}
   855  
   856  	if transaction.signedTransactions._Length() == 0 {
   857  		return transaction
   858  	}
   859  
   860  	transaction.transactions = _NewLockableSlice()
   861  	transaction.publicKeys = append(transaction.publicKeys, publicKey)
   862  	transaction.transactionSigners = append(transaction.transactionSigners, nil)
   863  	transaction.transactionIDs.locked = true
   864  
   865  	for index := 0; index < transaction.signedTransactions._Length(); index++ {
   866  		var temp *services.SignedTransaction
   867  		switch t := transaction.signedTransactions._Get(index).(type) { //nolint
   868  		case *services.SignedTransaction:
   869  			temp = t
   870  		}
   871  		temp.SigMap.SigPair = append(
   872  			temp.SigMap.SigPair,
   873  			publicKey._ToSignaturePairProtobuf(signature),
   874  		)
   875  		transaction.signedTransactions._Set(index, temp)
   876  	}
   877  
   878  	return transaction
