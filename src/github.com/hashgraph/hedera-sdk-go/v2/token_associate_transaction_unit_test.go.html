    74  func TestUnitTokenAssociateTransactionMock(t *testing.T) {
    75  	newKey, err := PrivateKeyFromStringEd25519("302e020100300506032b657004220420a869f4c6191b9c8c99933e7f6b6611711737e4b1a1a5a4cb5370e719a1f6df98")
    76  	require.NoError(t, err)
    77  
    78  	call := func(request *services.Transaction) *services.TransactionResponse {
    79  		require.NotEmpty(t, request.SignedTransactionBytes)
    80  		signedTransaction := services.SignedTransaction{}
    81  		_ = protobuf.Unmarshal(request.SignedTransactionBytes, &signedTransaction)
    82  
    83  		require.NotEmpty(t, signedTransaction.BodyBytes)
    84  		transactionBody := services.TransactionBody{}
    85  		_ = protobuf.Unmarshal(signedTransaction.BodyBytes, &transactionBody)
    86  
    87  		require.NotNil(t, transactionBody.TransactionID)
    88  		transactionId := transactionBody.TransactionID.String()
    89  		require.NotEqual(t, "", transactionId)
    90  
    91  		sigMap := signedTransaction.GetSigMap()
    92  		require.NotNil(t, sigMap)
    93  
    94  		key, _ := PrivateKeyFromStringEd25519("302e020100300506032b657004220420d45e1557156908c967804615af59a000be88c7aa7058bfcbe0f46b16c28f887d")
    95  		for _, sigPair := range sigMap.SigPair {
    96  			verified := false
    97  
    98  			switch k := sigPair.Signature.(type) {
    99  			case *services.SignaturePair_Ed25519:
   100  				pbTemp, _ := PublicKeyFromBytesEd25519(sigPair.PubKeyPrefix)
   101  				verified = pbTemp.Verify(signedTransaction.BodyBytes, k.Ed25519)
   102  			case *services.SignaturePair_ECDSASecp256K1:
   103  				pbTemp, _ := PublicKeyFromBytesECDSA(sigPair.PubKeyPrefix)
   104  				verified = pbTemp.Verify(signedTransaction.BodyBytes, k.ECDSASecp256K1)
   105  			}
   106  			require.True(t, verified)
   107  		}
   108  
   109  		require.Equal(t, bytes.Compare(sigMap.SigPair[1].PubKeyPrefix, key.PublicKey().BytesRaw()), 0)
   110  		require.Equal(t, bytes.Compare(sigMap.SigPair[0].PubKeyPrefix, newKey.PublicKey().BytesRaw()), 0)
   111  
   112  		if bod, ok := transactionBody.Data.(*services.TransactionBody_TokenAssociate); ok {
   113  			require.Equal(t, bod.TokenAssociate.Account.GetAccountNum(), int64(123))
   114  			require.Equal(t, bod.TokenAssociate.Tokens[0].TokenNum, int64(123))
   115  		}
   116  
   117  		return &services.TransactionResponse{
   118  			NodeTransactionPrecheckCode: services.ResponseCodeEnum_OK,
   119  		}
   120  	}
   121  	responses := [][]interface{}{{
   122  		call,
   123  	}}
   124  
   125  	client, server := NewMockClientAndServer(responses)
   126  	defer server.Close()
   127  
   128  	freez, err := NewTokenAssociateTransaction().
   129  		SetNodeAccountIDs([]AccountID{{Account: 3}}).
   130  		SetAccountID(AccountID{Account: 123}).
   131  		SetTokenIDs(TokenID{Token: 123}).
   132  		FreezeWith(client)
   133  	require.NoError(t, err)
   134  
   135  	_, err = freez.
   136  		Sign(newKey).
   137  		Execute(client)
   138  	require.NoError(t, err)
   139  }
