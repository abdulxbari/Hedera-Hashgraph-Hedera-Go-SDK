    34  func TestIntegrationAccountInfoQueryCanExecute(t *testing.T) {
    35  	env := NewIntegrationTestEnv(t)
    36  
    37  	newKey, err := PrivateKeyGenerateEd25519()
    38  	require.NoError(t, err)
    39  
    40  	newBalance := NewHbar(2)
    41  	assert.Equal(t, 2*HbarUnits.Hbar._NumberOfTinybar(), newBalance.tinybar)
    42  
    43  	resp, err := NewAccountCreateTransaction().
    44  		SetKey(newKey.PublicKey()).
    45  		SetInitialBalance(newBalance).
    46  		Execute(env.Client)
    47  	require.NoError(t, err)
    48  
    49  	receipt, err := resp.SetValidateStatus(true).GetReceipt(env.Client)
    50  	require.NoError(t, err)
    51  
    52  	accountID := *receipt.AccountID
    53  	require.NoError(t, err)
    54  
    55  	info, err := NewAccountInfoQuery().
    56  		SetAccountID(accountID).
    57  		SetNodeAccountIDs([]AccountID{resp.NodeID}).
    58  		SetMaxQueryPayment(NewHbar(1)).
    59  		SetQueryPayment(HbarFromTinybar(25)).
    60  		Execute(env.Client)
    61  	require.NoError(t, err)
    62  
    63  	assert.Equal(t, accountID, info.AccountID)
    64  	assert.Equal(t, false, info.IsDeleted)
    65  	assert.Equal(t, newKey.PublicKey(), info.Key)
    66  	assert.Equal(t, newBalance.tinybar, info.Balance.tinybar)
    67  
    68  	tx, err := NewAccountDeleteTransaction().
    69  		SetAccountID(accountID).
    70  		SetNodeAccountIDs([]AccountID{resp.NodeID}).
    71  		SetTransferAccountID(env.Client.GetOperatorAccountID()).
    72  		SetTransactionID(TransactionIDGenerate(accountID)).
    73  		FreezeWith(env.Client)
    74  	require.NoError(t, err)
    75  
    76  	resp, err = tx.
    77  		Sign(newKey).
    78  		Execute(env.Client)
    79  	require.NoError(t, err)
    80  
    81  	_, err = resp.SetValidateStatus(true).GetReceipt(env.Client)
    82  	require.NoError(t, err)
    83  
    84  	//err = CloseIntegrationTestEnv(env, nil)
    85  	//require.NoError(t, err)
