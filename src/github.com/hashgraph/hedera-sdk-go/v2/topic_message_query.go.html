   180  func (query *TopicMessageQuery) Subscribe(client *Client, onNext func(TopicMessage)) (SubscriptionHandle, error) {
   181  	handle := SubscriptionHandle{}
   182  
   183  	err := query._ValidateNetworkOnIDs(client)
   184  	if err != nil {
   185  		return SubscriptionHandle{}, err
   186  	}
   187  
   188  	pb := query._Build()
   189  
   190  	messages := make(map[string][]*mirror.ConsensusTopicResponse)
   191  
   192  	channel, err := client.mirrorNetwork._GetNextMirrorNode()._GetConsensusServiceClient()
   193  	if err != nil {
   194  		return handle, err
   195  	}
   196  
   197  	go func() {
   198  		var subClient mirror.ConsensusService_SubscribeTopicClient
   199  		var err error
   200  
   201  		for {
   202  			if err != nil {
   203  				handle.Unsubscribe()
   204  
   205  				if grpcErr, ok := status.FromError(err); ok { // nolint
   206  					if query.attempt < query.maxAttempts && query.retryHandler(err) {
   207  						subClient = nil
   208  
   209  						delay := math.Min(250.0*math.Pow(2.0, float64(query.attempt)), 8000)
   210  						time.Sleep(time.Duration(delay) * time.Millisecond)
   211  						query.attempt++
   212  					} else {
   213  						query.errorHandler(*grpcErr)
   214  						break
   215  					}
   216  				} else if err == io.EOF {
   217  					query.completionHandler()
   218  					break
   219  				} else {
   220  					panic(err)
   221  				}
   222  			}
   223  
   224  			if subClient == nil {
   225  				ctx, cancel := context.WithCancel(context.TODO())
   226  				handle.onUnsubscribe = cancel
   227  
   228  				subClient, err = (*channel).SubscribeTopic(ctx, pb)
   229  
   230  				if err != nil {
   231  					continue
   232  				}
   233  			}
   234  
   235  			var resp *mirror.ConsensusTopicResponse
   236  			resp, err = subClient.Recv()
   237  
   238  			if err != nil {
   239  				continue
   240  			}
   241  
   242  			if resp.ConsensusTimestamp != nil {
   243  				pb.ConsensusStartTime = _TimeToProtobuf(_TimeFromProtobuf(resp.ConsensusTimestamp).Add(1 * time.Nanosecond))
   244  			}
   245  
   246  			if pb.Limit > 0 {
   247  				pb.Limit--
   248  			}
   249  
   250  			if resp.ChunkInfo == nil || resp.ChunkInfo.Total == 1 {
   251  				onNext(_TopicMessageOfSingle(resp))
   252  			} else {
   253  				txID := _TransactionIDFromProtobuf(resp.ChunkInfo.InitialTransactionID).String()
   254  				message, ok := messages[txID]
   255  				if !ok {
   256  					message = make([]*mirror.ConsensusTopicResponse, 0, resp.ChunkInfo.Total)
   257  				}
   258  
   259  				message = append(message, resp)
   260  				messages[txID] = message
   261  
   262  				if int32(len(message)) == resp.ChunkInfo.Total {
   263  					delete(messages, txID)
   264  
   265  					onNext(_TopicMessageOfMany(message))
   266  				}
   267  			}
   268  		}
   269  	}()
   270  
   271  	return handle, nil
   272  }

   282  func _DefaultRetryHandler(err error) bool {
   283  	code := status.Code(err)
   284  
   285  	switch code {
   286  	case codes.NotFound, codes.ResourceExhausted, codes.Unavailable:
   287  		return true
   288  	case codes.Internal:
   289  		grpcErr, ok := status.FromError(err)
   290  
   291  		if !ok {
   292  			return false
   293  		}
   294  
   295  		return rstStream.FindIndex([]byte(grpcErr.Message())) != nil
   296  	default:
   297  		return false
   298  	}
   299  }
