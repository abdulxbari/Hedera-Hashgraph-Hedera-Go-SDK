   283  // Execute executes the Transaction with the provided client
   284  func (transaction *ScheduleCreateTransaction) Execute(
   285  	client *Client,
   286  ) (TransactionResponse, error) {
   287  	if client == nil {
   288  		return TransactionResponse{}, errNoClientProvided
   289  	}
   290  
   291  	if transaction.freezeError != nil {
   292  		return TransactionResponse{}, transaction.freezeError
   293  	}
   294  
   295  	if !transaction.IsFrozen() {
   296  		_, err := transaction.FreezeWith(client)
   297  		if err != nil {
   298  			return TransactionResponse{}, err
   299  		}
   300  	}
   301  
   302  	transactionID := transaction.transactionIDs._GetCurrent().(TransactionID)
   303  
   304  	if !client.GetOperatorAccountID()._IsZero() && client.GetOperatorAccountID()._Equals(*transactionID.AccountID) {
   305  		transaction.SignWith(
   306  			client.GetOperatorPublicKey(),
   307  			client.operator.signer,
   308  		)
   309  	}
   310  
   311  	resp, err := _Execute(
   312  		client,
   313  		&transaction.Transaction,
   314  		_TransactionShouldRetry,
   315  		_TransactionMakeRequest,
   316  		_TransactionAdvanceRequest,
   317  		_TransactionGetNodeAccountID,
   318  		_ScheduleCreateTransactionGetMethod,
   319  		_TransactionMapStatusError,
   320  		_TransactionMapResponse,
   321  		transaction._GetLogID(),
   322  		transaction.grpcDeadline,
   323  		transaction.maxBackoff,
   324  		transaction.minBackoff,
   325  		transaction.maxRetry,
   326  	)
   327  
   328  	if err != nil {
   329  		return TransactionResponse{
   330  			TransactionID:  transaction.GetTransactionID(),
   331  			NodeID:         resp.(TransactionResponse).NodeID,
   332  			ValidateStatus: true,
   333  		}, err
   334  	}
   335  
   336  	return TransactionResponse{
   337  		TransactionID:          transaction.GetTransactionID(),
   338  		NodeID:                 resp.(TransactionResponse).NodeID,
   339  		Hash:                   resp.(TransactionResponse).Hash,
   340  		ScheduledTransactionId: transaction.GetTransactionID(),
   341  	}, nil
   342  }

   348  func (transaction *ScheduleCreateTransaction) FreezeWith(client *Client) (*ScheduleCreateTransaction, error) {
   349  	if transaction.IsFrozen() {
   350  		return transaction, nil
   351  	}
   352  	transaction._InitFee(client)
   353  	err := transaction._ValidateNetworkOnIDs(client)
   354  	if err != nil {
   355  		return &ScheduleCreateTransaction{}, err
   356  	}
   357  	if err := transaction._InitTransactionID(client); err != nil {
   358  		return transaction, err
   359  	}
   360  	body := transaction._Build()

   362  	// transaction.transactionIDs[0] = transaction.transactionIDs[0].SetScheduled(true)
   363  
   364  	return transaction, _TransactionFreezeWith(&transaction.Transaction, client, body)
