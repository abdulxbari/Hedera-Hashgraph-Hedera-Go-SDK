   269  // ExecuteAll executes the all the Transactions with the provided client
   270  func (transaction *FileAppendTransaction) ExecuteAll(
   271  	client *Client,
   272  ) ([]TransactionResponse, error) {
   273  	if client == nil || client.operator == nil {
   274  		return []TransactionResponse{}, errNoClientProvided
   275  	}
   276  
   277  	if !transaction.IsFrozen() {
   278  		_, err := transaction.FreezeWith(client)
   279  		if err != nil {
   280  			return []TransactionResponse{}, err
   281  		}
   282  	}
   283  
   284  	var transactionID TransactionID
   285  	if transaction.transactionIDs._Length() > 0 {
   286  		transactionID = transaction.GetTransactionID()
   287  	} else {
   288  		return []TransactionResponse{}, errors.New("transactionID list is empty")
   289  	}
   290  
   291  	if !client.GetOperatorAccountID()._IsZero() && client.GetOperatorAccountID()._Equals(*transactionID.AccountID) {
   292  		transaction.SignWith(
   293  			client.GetOperatorPublicKey(),
   294  			client.operator.signer,
   295  		)
   296  	}
   297  
   298  	size := transaction.signedTransactions._Length() / transaction.nodeAccountIDs._Length()
   299  	list := make([]TransactionResponse, size)
   300  
   301  	for i := 0; i < size; i++ {
   302  		resp, err := _Execute(
   303  			client,
   304  			&transaction.Transaction,
   305  			_TransactionShouldRetry,
   306  			_TransactionMakeRequest,
   307  			_TransactionAdvanceRequest,
   308  			_TransactionGetNodeAccountID,
   309  			_FileAppendTransactionGetMethod,
   310  			_TransactionMapStatusError,
   311  			_TransactionMapResponse,
   312  			transaction._GetLogID(),
   313  			transaction.grpcDeadline,
   314  			transaction.maxBackoff,
   315  			transaction.minBackoff,
   316  			transaction.maxRetry,
   317  		)
   318  
   319  		if err != nil {
   320  			return list, err
   321  		}
   322  
   323  		list[i] = resp.(TransactionResponse)
   324  
   325  		_, err = NewTransactionReceiptQuery().
   326  			SetNodeAccountIDs([]AccountID{resp.(TransactionResponse).NodeID}).
   327  			SetTransactionID(resp.(TransactionResponse).TransactionID).
   328  			Execute(client)
   329  		if err != nil {
   330  			return list, err
   331  		}
   332  	}
   333  
   334  	return list, nil
   335  }

   341  func (transaction *FileAppendTransaction) FreezeWith(client *Client) (*FileAppendTransaction, error) {
   342  	if transaction.IsFrozen() {
   343  		return transaction, nil
   344  	}
   345  
   346  	if transaction.nodeAccountIDs._Length() == 0 {
   347  		if client == nil {
   348  			return transaction, errNoClientOrTransactionIDOrNodeId
   349  		}
   350  
   351  		transaction.SetNodeAccountIDs(client.network._GetNodeAccountIDsForExecute())
   352  	}
   353  
   354  	transaction._InitFee(client)
   355  	err := transaction._ValidateNetworkOnIDs(client)
   356  	if err != nil {
   357  		return &FileAppendTransaction{}, err
   358  	}
   359  	if err := transaction._InitTransactionID(client); err != nil {
   360  		return transaction, err
   361  	}
   362  	body := transaction._Build()
   363  
   364  	chunks := uint64((len(transaction.contents) + (transaction.chunkSize - 1)) / transaction.chunkSize)
   365  	if chunks > transaction.maxChunks {
   366  		return transaction, ErrMaxChunksExceeded{
   367  			Chunks:    chunks,
   368  			MaxChunks: transaction.maxChunks,
   369  		}
   370  	}
   371  
   372  	nextTransactionID := transaction.transactionIDs._GetCurrent().(TransactionID)
   373  
   374  	transaction.transactionIDs = _NewLockableSlice()
   375  	transaction.transactions = _NewLockableSlice()
   376  	transaction.signedTransactions = _NewLockableSlice()
   377  
   378  	if b, ok := body.Data.(*services.TransactionBody_FileAppend); ok {
   379  		for i := 0; uint64(i) < chunks; i++ {
   380  			start := i * transaction.chunkSize
   381  			end := start + transaction.chunkSize
   382  
   383  			if end > len(transaction.contents) {
   384  				end = len(transaction.contents)
   385  			}
   386  
   387  			transaction.transactionIDs._Push(_TransactionIDFromProtobuf(nextTransactionID._ToProtobuf()))
   388  			if err != nil {
   389  				panic(err)
   390  			}
   391  			b.FileAppend.Contents = transaction.contents[start:end]
   392  
   393  			body.TransactionID = nextTransactionID._ToProtobuf()
   394  			body.Data = &services.TransactionBody_FileAppend{
   395  				FileAppend: b.FileAppend,
   396  			}
   397  
   398  			for _, nodeAccountID := range transaction.GetNodeAccountIDs() {
   399  				body.NodeAccountID = nodeAccountID._ToProtobuf()
   400  
   401  				bodyBytes, err := protobuf.Marshal(body)
   402  				if err != nil {
   403  					return transaction, errors.Wrap(err, "error serializing body for file append")
   404  				}
   405  
   406  				transaction.signedTransactions._Push(&services.SignedTransaction{
   407  					BodyBytes: bodyBytes,
   408  					SigMap:    &services.SignatureMap{},
   409  				})
   410  			}
   411  
   412  			validStart := *nextTransactionID.ValidStart
   413  
   414  			*nextTransactionID.ValidStart = validStart.Add(1 * time.Nanosecond)
   415  		}
   416  	}
   417  
   418  	return transaction, nil
   419  }

   490  func (transaction *FileAppendTransaction) AddSignature(publicKey PublicKey, signature []byte) *FileAppendTransaction {
   491  	transaction._RequireOneNodeAccountID()
   492  
   493  	if transaction._KeyAlreadySigned(publicKey) {
   494  		return transaction
   495  	}
   496  
   497  	if transaction.signedTransactions._Length() == 0 {
   498  		return transaction
   499  	}
   500  
   501  	transaction.transactions = _NewLockableSlice()
   502  	transaction.publicKeys = append(transaction.publicKeys, publicKey)
   503  	transaction.transactionSigners = append(transaction.transactionSigners, nil)
   504  	transaction.transactionIDs.locked = true
   505  
   506  	for index := 0; index < transaction.signedTransactions._Length(); index++ {
   507  		var temp *services.SignedTransaction
   508  		switch t := transaction.signedTransactions._Get(index).(type) { //nolint
   509  		case *services.SignedTransaction:
   510  			temp = t
   511  		}
   512  		temp.SigMap.SigPair = append(
   513  			temp.SigMap.SigPair,
   514  			publicKey._ToSignaturePairProtobuf(signature),
   515  		)
   516  		transaction.signedTransactions._Set(index, temp)
   517  	}
   518  
   519  	return transaction
   520  }
