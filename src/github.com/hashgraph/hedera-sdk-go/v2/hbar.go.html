    79  func HbarFromString(hbar string) (Hbar, error) {
    80  	var err error
    81  	match := regexp.MustCompile(`^((?:\+|\-)?\d+(?:\.\d+)?)(?: (tℏ|μℏ|mℏ|ℏ|kℏ|Mℏ|Gℏ))?$`)
    82  
    83  	matchArray := match.FindStringSubmatch(hbar)
    84  	if len(matchArray) == 0 {
    85  		return Hbar{}, errors.New("invalid number and/or symbol")
    86  	}
    87  
    88  	a, err := strconv.ParseFloat(matchArray[1], 64)
    89  	if err != nil {
    90  		return Hbar{}, err
    91  	}
    92  
    93  	return HbarFrom(a, _HbarUnitFromString(matchArray[2])), nil
    94  }
