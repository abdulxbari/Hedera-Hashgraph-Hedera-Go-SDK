    37  func TestUnitFileCreateTransactionMock(t *testing.T) {
    38  	call := func(request *services.Transaction) *services.TransactionResponse {
    39  		require.NotEmpty(t, request.SignedTransactionBytes)
    40  		signedTransaction := services.SignedTransaction{}
    41  		_ = protobuf.Unmarshal(request.SignedTransactionBytes, &signedTransaction)
    42  
    43  		require.NotEmpty(t, signedTransaction.BodyBytes)
    44  		transactionBody := services.TransactionBody{}
    45  		_ = protobuf.Unmarshal(signedTransaction.BodyBytes, &transactionBody)
    46  
    47  		require.NotNil(t, transactionBody.TransactionID)
    48  		transactionId := transactionBody.TransactionID.String()
    49  		require.NotEqual(t, "", transactionId)
    50  
    51  		sigMap := signedTransaction.GetSigMap()
    52  		require.NotNil(t, sigMap)
    53  
    54  		for _, sigPair := range sigMap.SigPair {
    55  			verified := false
    56  
    57  			switch k := sigPair.Signature.(type) {
    58  			case *services.SignaturePair_Ed25519:
    59  				pbTemp, _ := PublicKeyFromBytesEd25519(sigPair.PubKeyPrefix)
    60  				verified = pbTemp.Verify(signedTransaction.BodyBytes, k.Ed25519)
    61  			case *services.SignaturePair_ECDSASecp256K1:
    62  				pbTemp, _ := PublicKeyFromBytesECDSA(sigPair.PubKeyPrefix)
    63  				verified = pbTemp.Verify(signedTransaction.BodyBytes, k.ECDSASecp256K1)
    64  			}
    65  			require.True(t, verified)
    66  		}
    67  
    68  		if bod, ok := transactionBody.Data.(*services.TransactionBody_FileCreate); ok {
    69  			require.Equal(t, bod.FileCreate.Memo, "go sdk e2e tests")
    70  			require.Equal(t, hex.EncodeToString(bod.FileCreate.Keys.Keys[0].GetEd25519()), "87deee7c3e3bd41d94d710531d529a1ae0c2e0463b27250072e177879267f501")
    71  			require.Equal(t, bytes.Compare(bod.FileCreate.Contents, []byte("Hello, World")), 0)
    72  		}
    73  
    74  		return &services.TransactionResponse{
    75  			NodeTransactionPrecheckCode: services.ResponseCodeEnum_OK,
    76  		}
    77  	}
    78  	responses := [][]interface{}{{
    79  		call,
    80  	}}
    81  
    82  	client, server := NewMockClientAndServer(responses)
    83  	defer server.Close()
    84  	newKey, err := PrivateKeyFromStringEd25519("302e020100300506032b657004220420a869f4c6191b9c8c99933e7f6b6611711737e4b1a1a5a4cb5370e719a1f6df98")
    85  	require.NoError(t, err)
    86  
    87  	_, err = NewFileCreateTransaction().
    88  		SetKeys(newKey).
    89  		SetNodeAccountIDs([]AccountID{{Account: 3}}).
    90  		SetContents([]byte("Hello, World")).
    91  		SetMemo("go sdk e2e tests").
    92  		Execute(client)
    93  	require.NoError(t, err)
    94  }
