   159  func (query *ScheduleInfoQuery) Execute(client *Client) (ScheduleInfo, error) {
   160  	if client == nil || client.operator == nil {
   161  		return ScheduleInfo{}, errNoClientProvided
   162  	}
   163  
   164  	var err error
   165  
   166  	err = query._ValidateNetworkOnIDs(client)
   167  	if err != nil {
   168  		return ScheduleInfo{}, err
   169  	}
   170  
   171  	if !query.paymentTransactionIDs.locked {
   172  		query.paymentTransactionIDs._Clear()._Push(TransactionIDGenerate(client.operator.accountID))
   173  	}
   174  
   175  	var cost Hbar
   176  	if query.queryPayment.tinybar != 0 {
   177  		cost = query.queryPayment
   178  	} else {
   179  		if query.maxQueryPayment.tinybar == 0 {
   180  			cost = client.GetDefaultMaxQueryPayment()
   181  		} else {
   182  			cost = query.maxQueryPayment
   183  		}
   184  
   185  		actualCost, err := query.GetCost(client)
   186  		if err != nil {
   187  			return ScheduleInfo{}, err
   188  		}
   189  
   190  		if cost.tinybar < actualCost.tinybar {
   191  			return ScheduleInfo{}, ErrMaxQueryPaymentExceeded{
   192  				QueryCost:       actualCost,
   193  				MaxQueryPayment: cost,
   194  				query:           "ScheduleInfo",
   195  			}
   196  		}
   197  
   198  		cost = actualCost
   199  	}
   200  
   201  	query.paymentTransactions = make([]*services.Transaction, 0)
   202  
   203  	if query.nodeAccountIDs.locked {
   204  		err = _QueryGeneratePayments(&query.Query, client, cost)
   205  		if err != nil {
   206  			return ScheduleInfo{}, err
   207  		}
   208  	} else {
   209  		paymentTransaction, err := _QueryMakePaymentTransaction(query.paymentTransactionIDs._GetCurrent().(TransactionID), AccountID{}, client.operator, cost)
   210  		if err != nil {
   211  			return ScheduleInfo{}, err
   212  		}
   213  		query.paymentTransactions = append(query.paymentTransactions, paymentTransaction)
   214  	}
   215  
   216  	pb := query._Build()
   217  	pb.ScheduleGetInfo.Header = query.pbHeader
   218  	query.pb = &services.Query{
   219  		Query: pb,
   220  	}
   221  
   222  	resp, err := _Execute(
   223  		client,
   224  		&query.Query,
   225  		_ScheduleInfoQueryShouldRetry,
   226  		_QueryMakeRequest,
   227  		_QueryAdvanceRequest,

   229  		_ScheduleInfoQueryGetMethod,
