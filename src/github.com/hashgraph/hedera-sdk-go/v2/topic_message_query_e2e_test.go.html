    85  func TestIntegrationTopicMessageQueryCanExecute(t *testing.T) {
    86  	env := NewIntegrationTestEnv(t)
    87  
    88  	resp, err := NewTopicCreateTransaction().
    89  		SetAdminKey(env.Client.GetOperatorPublicKey()).
    90  		SetNodeAccountIDs(env.NodeAccountIDs).
    91  		Execute(env.Client)
    92  
    93  	require.NoError(t, err)
    94  
    95  	receipt, err := resp.SetValidateStatus(true).GetReceipt(env.Client)
    96  	require.NoError(t, err)
    97  
    98  	topicID := *receipt.TopicID
    99  	assert.NotNil(t, topicID)
   100  
   101  	finished := false
   102  	start := time.Now()
   103  
   104  	_, err = NewTopicMessageQuery().
   105  		SetTopicID(topicID).
   106  		SetStartTime(time.Unix(0, 0)).
   107  		SetLimit(14).
   108  		SetCompletionHandler(func() {
   109  			finished = true
   110  		}).
   111  		Subscribe(env.Client, func(message TopicMessage) {
   112  			// Do nothing
   113  			println(string(message.Contents))
   114  		})
   115  	require.NoError(t, err)
   116  	resp, err = NewTopicMessageSubmitTransaction().
   117  		SetNodeAccountIDs([]AccountID{resp.NodeID}).
   118  		SetMessage([]byte(bigContents)).
   119  		SetTopicID(topicID).
   120  		Execute(env.Client)
   121  	require.NoError(t, err)
   122  
   123  	_, err = resp.SetValidateStatus(true).GetReceipt(env.Client)
   124  	require.NoError(t, err)
   125  
   126  	for {
   127  		if finished || uint64(time.Since(start).Seconds()) > 60 {
   128  			break
   129  		}
   130  
   131  		time.Sleep(2500)
   132  	}
   133  
   134  	resp, err = NewTopicDeleteTransaction().
   135  		SetTopicID(topicID).
   136  		SetNodeAccountIDs([]AccountID{resp.NodeID}).
   137  		Execute(env.Client)
   138  	require.NoError(t, err)
   139  
   140  	_, err = resp.SetValidateStatus(true).GetReceipt(env.Client)
   141  	require.NoError(t, err)
   142  
   143  	if !finished {
   144  		err = errors.New("Message was not received within 60 seconds")
   145  	}
   146  	require.NoError(t, err)
   147  
   148  	err = CloseIntegrationTestEnv(env, nil)
   149  	require.NoError(t, err)
   150  }
