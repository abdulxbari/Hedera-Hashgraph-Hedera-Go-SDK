    94  	given := hex.EncodeToString(byt)

    97  	if err != nil {
    98  		return &_Ed25519PrivateKey{}, err

   100  
   101  	length := len(decoded)
   102  	if length != 32 && length != 64 {
   103  		return &_Ed25519PrivateKey{}, _NewErrBadKeyf("invalid private key length: %v byt", len(byt))
   104  	}

   107  		keyData: ed25519.NewKeyFromSeed(decoded[0:32]),

   111  func _Ed25519PrivateKeyFromSeed(seed []byte) (*_Ed25519PrivateKey, error) {
   112  	h := hmac.New(sha512.New, []byte("ed25519 seed"))
   113  
   114  	_, err := h.Write(seed)
   115  	if err != nil {
   116  		return nil, err
   117  	}
   118  
   119  	digest := h.Sum(nil)
   120  
   121  	keyBytes := digest[0:32]
   122  	chainCode := digest[32:]
   123  	privateKey, err := _Ed25519PrivateKeyFromBytes(keyBytes)
   124  
   125  	if err != nil {
   126  		return nil, err
   127  	}
   128  	privateKey.chainCode = chainCode
   129  
   130  	return privateKey, nil
   131  }

   201  func _Ed25519PrivateKeyFromPem(bytes []byte, passphrase string) (*_Ed25519PrivateKey, error) {
   202  	var blockType string
   203  
   204  	if len(passphrase) == 0 {
   205  		blockType = "PRIVATE KEY"
   206  	} else {
   207  		// the pem is encrypted
   208  		blockType = "ENCRYPTED PRIVATE KEY"
   209  	}
   210  
   211  	var pk *pem.Block
   212  	for block, rest := pem.Decode(bytes); block != nil; {
   213  		if block.Type == blockType {
   214  			pk = block
   215  			break
   216  		}
   217  
   218  		bytes = rest
   219  		if len(bytes) == 0 {
   220  			// no key was found
   221  			return &_Ed25519PrivateKey{}, _NewErrBadKeyf("pem file did not contain a private key")
   222  		}
   223  	}
   224  
   225  	if pk == nil {
   226  		// no key was found
   227  		return &_Ed25519PrivateKey{}, _NewErrBadKeyf("no PEM data is found")
   228  	}
   229  
   230  	if len(passphrase) == 0 {
   231  		// key does not need decrypted, end here
   232  		key, err := PrivateKeyFromString(hex.EncodeToString(pk.Bytes))
   233  		if err != nil {
   234  			return &_Ed25519PrivateKey{}, err
   235  		}
   236  		return key.ed25519PrivateKey, nil
   237  	}
   238  
   239  	keyI, err := pkcs8.ParsePKCS8PrivateKey(pk.Bytes, []byte(passphrase))
   240  	if err != nil {
   241  		return &_Ed25519PrivateKey{}, err
   242  	}
   243  
   244  	return _Ed25519PrivateKeyFromBytes(keyI.(ed25519.PrivateKey))
   245  }

   314  // Derive a child key compatible with the iOS and Android wallets using a provided wallet/account index. Use index 0 for

   318  func (sk _Ed25519PrivateKey) _Derive(index uint32) (*_Ed25519PrivateKey, error) {
   319  	if !sk._SupportsDerivation() {
   320  		return nil, _NewErrBadKeyf("child key cannot be derived from this key")

   323  	derivedKeyBytes, chainCode, err := _DeriveEd25519ChildKey(sk._BytesRaw(), sk.chainCode, index)

   352  func (sk _Ed25519PrivateKey) _SignTransaction(transaction *Transaction) ([]byte, error) {
   353  	transaction._RequireOneNodeAccountID()
   354  
   355  	if transaction.signedTransactions._Length() == 0 {
   356  		return make([]byte, 0), errTransactionRequiresSingleNodeAccountID
   357  	}
   358  
   359  	signature := sk._Sign(transaction.signedTransactions._Get(0).(*services.SignedTransaction).GetBodyBytes())
   360  
   361  	publicKey := sk._PublicKey()
   362  	if publicKey == nil {
   363  		return []byte{}, errors.New("public key is nil")
   364  	}
   365  
   366  	wrappedPublicKey := PublicKey{
   367  		ed25519PublicKey: publicKey,
   368  	}
   369  
   370  	if transaction._KeyAlreadySigned(wrappedPublicKey) {
   371  		return []byte{}, nil
   372  	}
   373  
   374  	transaction.transactions = _NewLockableSlice()
   375  	transaction.publicKeys = append(transaction.publicKeys, wrappedPublicKey)
   376  	transaction.transactionSigners = append(transaction.transactionSigners, nil)
   377  	transaction.transactionIDs.locked = true
   378  
   379  	for index := 0; index < transaction.signedTransactions._Length(); index++ {
   380  		temp := transaction.signedTransactions._Get(index).(*services.SignedTransaction)
   381  		temp.SigMap.SigPair = append(
   382  			temp.SigMap.SigPair,
   383  			publicKey._ToSignaturePairProtobuf(signature),
   384  		)
   385  		transaction.signedTransactions._Set(index, temp)
   386  	}
   387  
   388  	return signature, nil
   389  }
