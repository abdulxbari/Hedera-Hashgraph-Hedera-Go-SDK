   230  		SetKey(key).
   231  		SetInitialBalance(NewHbar(3)).
   232  		SetAccountMemo("ty").
   233  		SetReceiverSignatureRequired(true).
   234  		SetMaxAutomaticTokenAssociations(2).
   235  		SetStakedAccountID(stackedAccountID).
   236  		SetDeclineStakingReward(true).

   238  		SetTransactionMemo("").
   239  		SetTransactionValidDuration(60 * time.Second).

   254  	require.Equal(t, proto.DeclineReward, true)
   255  	require.Equal(t, proto.AutoRenewPeriod.String(), _DurationToProtobuf(60*time.Second).String())
