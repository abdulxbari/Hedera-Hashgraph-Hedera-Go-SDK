    34  func TestIntegrationTokenInfoQueryCanExecute(t *testing.T) {
    35  	env := NewIntegrationTestEnv(t)
    36  
    37  	resp, err := NewTokenCreateTransaction().
    38  		SetNodeAccountIDs(env.NodeAccountIDs).
    39  		SetTokenName("ffff").
    40  		SetTokenSymbol("F").
    41  		SetDecimals(3).
    42  		SetInitialSupply(1000000).
    43  		SetTreasuryAccountID(env.Client.GetOperatorAccountID()).
    44  		SetAdminKey(env.Client.GetOperatorPublicKey()).
    45  		SetFreezeKey(env.Client.GetOperatorPublicKey()).
    46  		SetWipeKey(env.Client.GetOperatorPublicKey()).
    47  		SetKycKey(env.Client.GetOperatorPublicKey()).
    48  		SetSupplyKey(env.Client.GetOperatorPublicKey()).
    49  		SetFreezeDefault(false).
    50  		Execute(env.Client)
    51  	require.NoError(t, err)
    52  
    53  	receipt, err := resp.SetValidateStatus(true).GetReceipt(env.Client)
    54  	require.NoError(t, err)
    55  
    56  	tokenID := *receipt.TokenID
    57  
    58  	info, err := NewTokenInfoQuery().
    59  		SetNodeAccountIDs([]AccountID{resp.NodeID}).
    60  		SetMaxQueryPayment(NewHbar(2)).
    61  		SetTokenID(tokenID).
    62  		SetQueryPayment(NewHbar(1)).
    63  		Execute(env.Client)
    64  	require.NoError(t, err)
    65  
    66  	assert.Equal(t, info.TokenID, tokenID)
    67  	assert.Equal(t, info.Name, "ffff")
    68  	assert.Equal(t, info.Symbol, "F")
    69  	assert.Equal(t, info.Decimals, uint32(3))
    70  	assert.Equal(t, info.Treasury, env.Client.GetOperatorAccountID())
    71  	assert.NotNil(t, info.AdminKey)
    72  	assert.NotNil(t, info.KycKey)
    73  	assert.NotNil(t, info.FreezeKey)
    74  	assert.NotNil(t, info.WipeKey)
    75  	assert.NotNil(t, info.SupplyKey)
    76  	assert.Equal(t, info.AdminKey.String(), env.Client.GetOperatorPublicKey().String())
    77  	assert.Equal(t, info.KycKey.String(), env.Client.GetOperatorPublicKey().String())
    78  	assert.Equal(t, info.FreezeKey.String(), env.Client.GetOperatorPublicKey().String())
    79  	assert.Equal(t, info.WipeKey.String(), env.Client.GetOperatorPublicKey().String())
    80  	assert.Equal(t, info.SupplyKey.String(), env.Client.GetOperatorPublicKey().String())
    81  	assert.False(t, *info.DefaultFreezeStatus)
    82  	assert.False(t, *info.DefaultKycStatus)
    83  
    84  	err = CloseIntegrationTestEnv(env, &tokenID)
    85  	require.NoError(t, err)
    86  }
