    32  // TokenUpdateTransaction
    33  // At consensus, updates an already created token to the given values.
    34  //
    35  // If no value is given for a field, that field is left unchanged. For an immutable tokens (that is,
    36  // a token without an admin key), only the expiry may be updated. Setting any other field in that
    37  // case will cause the transaction status to resolve to TOKEN_IS_IMMUTABLE.
    38  //
    39  // --- Signing Requirements ---
    40  //  1. Whether or not a token has an admin key, its expiry can be extended with only the transaction
    41  //     payer's signature.
    42  //  2. Updating any other field of a mutable token requires the admin key's signature.
    43  //  3. If a new admin key is set, this new key must sign <b>unless</b> it is exactly an empty
    44  //     <tt>KeyList</tt>. This special sentinel key removes the existing admin key and causes the
    45  //     token to become immutable. (Other <tt>Key</tt> structures without a constituent
    46  //     <tt>Ed25519</tt> key will be rejected with <tt>INVALID_ADMIN_KEY</tt>.)
    47  //  4. If a new treasury is set, the new treasury account's key must sign the transaction.
    48  //
    49  // --- Nft Requirements ---
    50  //  1. If a non fungible token has a positive treasury balance, the operation will abort with
    51  //     CURRENT_TREASURY_STILL_OWNS_NFTS.

   556  // Execute executes the Transaction with the provided client
   557  func (transaction *TokenUpdateTransaction) Execute(
   558  	client *Client,
   559  ) (TransactionResponse, error) {
   560  	if client == nil {
   561  		return TransactionResponse{}, errNoClientProvided
   562  	}
   563  
   564  	if transaction.freezeError != nil {
   565  		return TransactionResponse{}, transaction.freezeError
   566  	}
   567  
   568  	if !transaction.IsFrozen() {
   569  		_, err := transaction.FreezeWith(client)
   570  		if err != nil {
   571  			return TransactionResponse{}, err
   572  		}
   573  	}
   574
