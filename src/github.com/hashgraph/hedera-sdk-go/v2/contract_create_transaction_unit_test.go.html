    75  func TestUnitContractCreateTransactionMock(t *testing.T) {
    76  	key, err := PrivateKeyFromStringEd25519("302e020100300506032b657004220420d45e1557156908c967804615af59a000be88c7aa7058bfcbe0f46b16c28f887d")
    77  	require.NoError(t, err)
    78  
    79  	call := func(request *services.Transaction) *services.TransactionResponse {
    80  		require.NotEmpty(t, request.SignedTransactionBytes)
    81  		signedTransaction := services.SignedTransaction{}
    82  		_ = protobuf.Unmarshal(request.SignedTransactionBytes, &signedTransaction)
    83  
    84  		require.NotEmpty(t, signedTransaction.BodyBytes)
    85  		transactionBody := services.TransactionBody{}
    86  		_ = protobuf.Unmarshal(signedTransaction.BodyBytes, &transactionBody)
    87  
    88  		require.NotNil(t, transactionBody.TransactionID)
    89  		transactionId := transactionBody.TransactionID.String()
    90  		require.NotEqual(t, "", transactionId)
    91  
    92  		sigMap := signedTransaction.GetSigMap()
    93  		require.NotNil(t, sigMap)
    94  
    95  		for _, sigPair := range sigMap.SigPair {
    96  			verified := false
    97  
    98  			switch k := sigPair.Signature.(type) {
    99  			case *services.SignaturePair_Ed25519:
   100  				pbTemp, _ := PublicKeyFromBytesEd25519(sigPair.PubKeyPrefix)
   101  				verified = pbTemp.Verify(signedTransaction.BodyBytes, k.Ed25519)
   102  			case *services.SignaturePair_ECDSASecp256K1:
   103  				pbTemp, _ := PublicKeyFromBytesECDSA(sigPair.PubKeyPrefix)
   104  				verified = pbTemp.Verify(signedTransaction.BodyBytes, k.ECDSASecp256K1)
   105  			}
   106  			require.True(t, verified)
   107  		}
   108  
   109  		if bod, ok := transactionBody.Data.(*services.TransactionBody_ContractCreateInstance); ok {
   110  			require.Equal(t, bod.ContractCreateInstance.InitcodeSource.(*services.ContractCreateTransactionBody_FileID).FileID.FileNum, int64(123))
   111  			params := NewContractFunctionParameters().AddString("hello from hedera")
   112  			require.Equal(t, bytes.Compare(bod.ContractCreateInstance.ConstructorParameters, params._Build(nil)), 0)
   113  			require.Equal(t, bod.ContractCreateInstance.Memo, "hedera-sdk-go::TestContractCreateTransaction_Execute")
   114  			require.Equal(t, bod.ContractCreateInstance.Gas, int64(100000))
   115  			require.Equal(t, hex.EncodeToString(bod.ContractCreateInstance.AdminKey.GetEd25519()), key.PublicKey().StringRaw())
   116  		}
   117  
   118  		return &services.TransactionResponse{
   119  			NodeTransactionPrecheckCode: services.ResponseCodeEnum_OK,
   120  		}
   121  	}
   122  	responses := [][]interface{}{{
   123  		call,
   124  	}}
   125  
   126  	client, server := NewMockClientAndServer(responses)
   127  	defer server.Close()
   128  
   129  	_, err = NewContractCreateTransaction().
   130  		SetAdminKey(client.GetOperatorPublicKey()).
   131  		SetNodeAccountIDs([]AccountID{{Account: 3}}).
   132  		SetGas(100000).
   133  		SetConstructorParameters(NewContractFunctionParameters().AddString("hello from hedera")).
   134  		SetBytecodeFileID(FileID{File: 123}).
   135  		SetContractMemo("hedera-sdk-go::TestContractCreateTransaction_Execute").
   136  		Execute(client)
   137  	require.NoError(t, err)
   138  }
