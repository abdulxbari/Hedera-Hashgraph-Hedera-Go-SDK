    10  	var client *hedera.Client
    11  	var err error
    12  
    13  	// Retrieving network type from environment variable HEDERA_NETWORK
    14  	client, err = hedera.ClientForName(os.Getenv("HEDERA_NETWORK"))
    15  	if err != nil {
    16  		println(err.Error(), ": error creating client")
    17  		return
    18  	}
    19  
    20  	// Retrieving operator ID from environment variable OPERATOR_ID
    21  	operatorAccountID, err := hedera.AccountIDFromString(os.Getenv("OPERATOR_ID"))
    22  	if err != nil {
    23  		println(err.Error(), ": error converting string to AccountID")
    24  		return
    25  	}
    26  
    27  	// Retrieving operator key from environment variable OPERATOR_KEY
    28  	operatorKey, err := hedera.PrivateKeyFromString(os.Getenv("OPERATOR_KEY"))
    29  	if err != nil {
    30  		println(err.Error(), ": error converting string to PrivateKey")
    31  		return
    32  	}
    33  
    34  	// Setting the client operator ID and key
    35  	client.SetOperator(operatorAccountID, operatorKey)
    36  
    37  	// Create a new file
    38  	newFileResponse, err := hedera.NewFileCreateTransaction().
    39  		// Accepts both Key and []Key
    40  		// All keys at the top level of a key list must sign to create or modify the file. Any one of
    41  		// the keys at the top level key list can sign to delete the file.
    42  		SetKeys(client.GetOperatorPublicKey()).
    43  		// Basic starting file content
    44  		SetContents([]byte("Hello from Hedera.")).
    45  		SetMemo("go file append test").
    46  		// Set max fee if we don't want to get overcharged
    47  		SetMaxTransactionFee(hedera.NewHbar(2)).
    48  		Execute(client)
    49  	if err != nil {
    50  		println(err.Error(), ": error creating file")
    51  		return
    52  	}
    53  
    54  	// Get receipt to make sure the transaction worked
    55  	receipt, err := newFileResponse.GetReceipt(client)
    56  	if err != nil {
    57  		println(err.Error(), ": error retrieving file creation receipt")
    58  		return
    59  	}
    60  
    61  	// Retrieve file ID from the receipt
    62  	fileID := *receipt.FileID
    63  
    64  	// File append
    65  	fileResponse, err := hedera.NewFileAppendTransaction().
    66  		// Make sure the node is the same as the new file, as we don't have to wait for propagation
    67  		SetNodeAccountIDs([]hedera.AccountID{newFileResponse.NodeID}).
    68  		// File ID to append to
    69  		SetFileID(fileID).
    70  		// Contents that will be appended to the end of the file
    71  		SetContents([]byte(bigContents)).
    72  		// Set max transaction when you are not sure how much it will cost.
    73  		SetMaxTransactionFee(hedera.NewHbar(5)).
    74  		Execute(client)
    75  	if err != nil {
    76  		println(err.Error(), ": error executing file append transaction")
    77  		return
    78  	}
    79  
    80  	// Checking if transaction went through
    81  	receipt, err = fileResponse.GetReceipt(client)
    82  	if err != nil {
    83  		println(err.Error(), ": error retrieving file append transaction receipt")
    84  		return
    85  	}
    86  
    87  	// Checking if append succeeded
    88  	println(receipt.Status.String())
    89  	info, err := hedera.NewFileInfoQuery().
    90  		// Once again same node account ID
    91  		SetNodeAccountIDs([]hedera.AccountID{fileResponse.NodeID}).
    92  		// Only the file ID is required for this
    93  		SetFileID(fileID).
    94  		Execute(client)
    95  
    96  	if err != nil {
    97  		println(err.Error(), ": error executing file info query")
    98  		return
    99  	}
   100  
   101  	println("File size according to `FileInfoQuery`:", info.Size)
   102  	println("File memo according to `FileInfoQuery`:", info.FileMemo)
   103  }
