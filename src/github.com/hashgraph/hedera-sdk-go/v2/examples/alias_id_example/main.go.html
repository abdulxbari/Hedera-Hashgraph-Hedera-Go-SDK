    27  	}

    31  	if err != nil {

    34  	}
    35  
    36  	// Defaults the operator account ID and key such that all generated transactions will be paid for
    37  	// by this account and be signed by this key
    38  	client.SetOperator(operatorAccountID, operatorKey)
    39  
    40  	/*
    41  	 * Hedera supports a form of auto account creation.
    42  	 *
    43  	 * You can "create" an account by generating a private key, and then deriving the public key,
    44  	 * without any need to interact with the Hedera network.  The public key more or less acts as the user's
    45  	 * account ID.  This public key is an account's aliasKey: a public key that aliases (or will eventually alias)
    46  	 * to a Hedera account.
    47  	 *
    48  	 * An AccountId takes one of two forms: a normal AccountId with a null aliasKey member takes the form 0.0.123,
    49  	 * while an account ID with a non-null aliasKey member takes the form
    50  	 * 0.0.302a300506032b6570032100114e6abc371b82dab5c15ea149f02d34a012087b163516dd70f44acafabf7777
    51  	 * Note the prefix of "0.0." indicating the shard and realm.  Also note that the aliasKey is stringified
    52  	 * as a hex-encoded ASN1 DER representation of the key.
    53  	 *
    54  	 * An AccountId with an aliasKey can be used just like a normal AccountId for the purposes of queries and
    55  	 * transactions, however most queries and transactions involving such an AccountId won't work until Hbar has
    56  	 * been transferred to the aliasKey account.
    57  	 *
    58  	 * There is no record in the Hedera network of an account associated with a given aliasKey
    59  	 * until an amount of Hbar is transferred to the account.  The moment that Hbar is transferred to that aliasKey
    60  	 * AccountId is the moment that that account actually begins to exist in the Hedera ledger.
    61  	 */
    62  
    63  	println("Creating a new account")
    64  
    65  	key, err := hedera.GeneratePrivateKey()
    66  	if err != nil {
    67  		println(err.Error(), ": error generating private key")
    68  		return
    69  	}
    70  	publicKey := key.PublicKey()
    71  
    72  	// Assuming that the target shard and realm are known.
    73  	// For now they are virtually always 0 and 0.
    74  	aliasAccountID := publicKey.ToAccountID(0, 0)
    75  
    76  	println("New account ID:", aliasAccountID.String())
    77  	println("Just the key:", aliasAccountID.AliasKey.String())

    80  	*
    81  	* Note that no queries or transactions have taken place yet.

    83  	*
    84  	* AccountId.fromString() can construct an AccountId with an aliasKey.
    85  	* It expects a string of the form 0.0.123 in the case of a normal AccountId, or of the form
    86  	* 0.0.302a300506032b6570032100114e6abc371b82dab5c15ea149f02d34a012087b163516dd70f44acafabf7777
    87  	* in the case of an AccountId with aliasKey.  Note the prefix of "0.0." to indicate the shard and realm.
    88  	*
    89  	* If the shard and realm are known, you may use PublicKeyFromString() then PublicKey.toAccountId() to construct the
    90  	* aliasKey AccountID
    91  	* fromStr, err := hedera.AccountIDFromString("0.0.302a300506032b6570032100114e6abc371b82dab5c15ea149f02d34a012087b163516dd70f44acafabf7777")
    92  	* publicKey2, err := hedera.PublicKeyFromString("302a300506032b6570032100114e6abc371b82dab5c15ea149f02d34a012087b163516dd70f44acafabf7777")
    93  	* fromKeyString := publicKey2.ToAccountID(0,0)
    94  	 */
    95  
    96  	println("Transferring some Hbar to the new account")
    97  	resp, err := hedera.NewTransferTransaction().
    98  		AddHbarTransfer(client.GetOperatorAccountID(), hedera.NewHbar(1).Negated()).
    99  		AddHbarTransfer(*aliasAccountID, hedera.NewHbar(1)).
   100  		Execute(client)
   101  	if err != nil {
   102  		println(err.Error(), ": error executing transfer transaction")
   103  		return
   104  	}
   105  
   106  	receipt, err := resp.GetReceipt(client)
   107  	println(receipt.Status.String())
   108  	if receipt.AccountID != nil {
   109  		println(receipt.AccountID.String())
   110  	}
   111  	if err != nil {
   112  		println(err.Error(), ": error getting transfer transaction receipt")
   113  		return
   114  	}
   115  
   116  	balance, err := hedera.NewAccountBalanceQuery().
   117  		SetAccountID(*aliasAccountID).
   118  		Execute(client)
   119  	if err != nil {
   120  		println(err.Error(), ": error retrieving balance")
   121  		return
   122  	}
   123  
   124  	println("Balance of the new account:", balance.Hbars.String())
   125  
   126  	info, err := hedera.NewAccountInfoQuery().

   128  		Execute(client)
   129  	if err != nil {

   131  		return
   132  	}

   136  	 * via an AccountInfoQuery.
   137  	 *
   138  	 * Users may continue to refer to the account by its aliasKey AccountId, but they may also
   139  	 * now refer to it by its normal AccountId
   140  	 */
   141

   145  	println("Example complete")
