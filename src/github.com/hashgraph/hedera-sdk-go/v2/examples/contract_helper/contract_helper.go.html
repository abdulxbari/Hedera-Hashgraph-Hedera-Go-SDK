   117  func (this *ContractHelper) ExecuteSteps(firstStep int32, lastStep int32, client *hedera.Client) (*ContractHelper, error) {
   118  	for stepIndex := firstStep; stepIndex <= lastStep; stepIndex++ {
   119  		println("Attempting to execuite step", stepIndex)
   120  
   121  		transaction := hedera.NewContractExecuteTransaction().
   122  			SetContractID(this.contractID).
   123  			SetGas(10000000)
   124  
   125  		payableAmount := this.GetPayableAmount(stepIndex)
   126  		if payableAmount != nil {
   127  			transaction.SetPayableAmount(*payableAmount)
   128  		}
   129  
   130  		functionName := "step" + strconv.Itoa(int(stepIndex))
   131  		parameters := this.GetParameterSupplier(stepIndex)()
   132  		if parameters != nil {
   133  			transaction.SetFunction(functionName, parameters)
   134  		} else {
   135  			transaction.SetFunction(functionName, nil)
   136  		}
   137  
   138  		if feePayerAccountID, ok := this.stepFeePayers[stepIndex]; ok {
   139  			transaction.SetTransactionID(hedera.TransactionIDGenerate(*feePayerAccountID))
   140  		}
   141  
   142  		frozen, err := transaction.FreezeWith(client)
   143  		if err != nil {
   144  			return &ContractHelper{}, err
   145  		}
   146  		for _, signer := range this.GetSigners(stepIndex) {
   147  			frozen.Sign(signer)
   148  		}
   149  
   150  		response, err := frozen.Execute(client)
   151  		if err != nil {
   152  			return &ContractHelper{}, err
   153  		}
   154  
   155  		record, err := response.GetRecord(client)
   156  		if err != nil {
   157  			return &ContractHelper{}, err
   158  		}
   159  
   160  		functionResult, err := record.GetContractExecuteResult()
   161  		if err != nil {
   162  			return &ContractHelper{}, err
   163  		}
   164  
   165  		if this.GetResultValidator(stepIndex)(functionResult) {
   166  			fmt.Printf("Step %d completed, and returned valid result. (TransactionId %s)", stepIndex, record.TransactionID.String())
   167  		} else {
   168  			return &ContractHelper{}, errors.New(fmt.Sprintf("Step %d returned invalid result", stepIndex))
   169  		}
   170  	}
   171  
   172  	return this, nil
   173  }
