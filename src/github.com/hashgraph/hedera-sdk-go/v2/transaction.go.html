   649  func (this *Transaction) _SignTransaction(index int) {
   650  	initialTx := this.signedTransactions._Get(index).(*services.SignedTransaction)
   651  	bodyBytes := initialTx.GetBodyBytes()
   652  	if len(initialTx.SigMap.SigPair) != 0 {
   653  		for i, key := range this.publicKeys {
   654  			if this.transactionSigners[i] != nil {
   655  				if key.ed25519PublicKey != nil {
   656  					if bytes.Equal(initialTx.SigMap.SigPair[0].PubKeyPrefix, key.ed25519PublicKey.keyData) {
   657  						if !this.regenerateTransactionID {
   658  							return
   659  						}
   660  						switch t := initialTx.SigMap.SigPair[0].Signature.(type) { //nolint
   661  						case *services.SignaturePair_Ed25519:
   662  							if bytes.Equal(t.Ed25519, this.transactionSigners[0](bodyBytes)) && len(t.Ed25519) > 0 {
   663  								return
   664  							}
   665  						}
   666  					}
   667  				}
   668  				if key.ecdsaPublicKey != nil {
   669  					if bytes.Equal(initialTx.SigMap.SigPair[0].PubKeyPrefix, key.ecdsaPublicKey._BytesRaw()) {
   670  						if !this.regenerateTransactionID {
   671  							return
   672  						}
   673  						switch t := initialTx.SigMap.SigPair[0].Signature.(type) { //nolint
   674  						case *services.SignaturePair_ECDSASecp256K1:
   675  							if bytes.Equal(t.ECDSASecp256K1, this.transactionSigners[0](bodyBytes)) && len(t.ECDSASecp256K1) > 0 {
   676  								return
   677  							}
   678  						}
   679  					}
   680  				}
   681  			}
   682  		}
   683  	}
   684  
   685  	if this.regenerateTransactionID && !this.transactionIDs.locked {
   686  		modifiedTx := this.signedTransactions._Get(index).(*services.SignedTransaction)
   687  		modifiedTx.SigMap.SigPair = make([]*services.SignaturePair, 0)
   688  		this.signedTransactions._Set(index, modifiedTx)
   689  	}
   690  
   691  	for i := 0; i < len(this.publicKeys); i++ {
   692  		publicKey := this.publicKeys[i]
   693  		signer := this.transactionSigners[i]
   694  
   695  		if signer == nil {
   696  			continue
   697  		}
   698  
   699  		modifiedTx := this.signedTransactions._Get(index).(*services.SignedTransaction)
   700  		modifiedTx.SigMap.SigPair = append(modifiedTx.SigMap.SigPair, publicKey._ToSignaturePairProtobuf(signer(bodyBytes)))
   701  		this.signedTransactions._Set(index, modifiedTx)
   702  	}
   703  }

   719  func (this *Transaction) _BuildTransaction(index int) (*services.Transaction, error) {
   720  	signedTx := this.signedTransactions._Get(index).(*services.SignedTransaction)
   721  
   722  	txID := this.transactionIDs._GetCurrent().(TransactionID)
   723  	originalBody := services.TransactionBody{}
   724  	_ = protobuf.Unmarshal(signedTx.BodyBytes, &originalBody)
   725  
   726  	if originalBody.NodeAccountID == nil {
   727  		originalBody.NodeAccountID = this.nodeAccountIDs._GetCurrent().(AccountID)._ToProtobuf()
   728  	}
   729  
   730  	if originalBody.TransactionID.String() != txID._ToProtobuf().String() {
   731  		originalBody.TransactionID = txID._ToProtobuf()
   732  	}
   733  
   734  	originalBody.Memo = this.memo
   735  	if this.transactionFee != 0 {
   736  		originalBody.TransactionFee = this.transactionFee
   737  	} else {
   738  		originalBody.TransactionFee = this.defaultMaxTransactionFee
   739  	}
   740  
   741  	updatedBody, err := protobuf.Marshal(&originalBody)
   742  	if err != nil {
   743  		return &services.Transaction{}, errors.Wrap(err, "failed to update this ID")
   744  	}
   745  
   746  	// Bellow are checks whether we need to sign the transaction or we already have the same signed
   747  	if bytes.Equal(signedTx.BodyBytes, updatedBody) {
   748  		sigPairLen := len(signedTx.SigMap.GetSigPair())
   749  		// For cases where we need more than 1 signature
   750  		if sigPairLen > 0 && sigPairLen == len(this.publicKeys) {
   751  			data, err := protobuf.Marshal(signedTx)
   752  			if err != nil {
   753  				return &services.Transaction{}, errors.Wrap(err, "failed to serialize transactions for building")
   754  			}
   755  			transaction := &services.Transaction{
   756  				SignedTransactionBytes: data,
   757  			}
   758  
   759  			return transaction, nil
   760  		}
   761  	}
   762  
   763  	signedTx.BodyBytes = updatedBody
   764  	this.signedTransactions._Set(index, signedTx)
   765  	this._SignTransaction(index)
   766  
   767  	tx := this.signedTransactions._Get(index).(*services.SignedTransaction)
   768  	data, err := protobuf.Marshal(tx)
   769  	if err != nil {
   770  		return &services.Transaction{}, errors.Wrap(err, "failed to serialize transactions for building")
   771  	}
   772  
   773  	transaction := &services.Transaction{
   774  		SignedTransactionBytes: data,
   775  	}
   776  
   777  	return transaction, nil
   778  }
