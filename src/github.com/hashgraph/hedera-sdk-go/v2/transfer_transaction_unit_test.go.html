    91  	expectedHbar := float64(3)

    96  		AddHbarTransfer(accountID, NewHbar(1)).
    97  		AddHbarTransfer(accountID, NewHbar(1))
    98  
    99  	transfer.AddHbarTransfer(accountID, NewHbar(1))
   100  	transfer.AddHbarTransfer(AccountID{Account: 1}, NewHbar(1))
   101  
   102  	found := false
   103  	for _, s := range transfer.hbarTransfers {
   104  		if s.accountID.Compare(accountID) == 0 {
   105  			if s.Amount.As("hbar") == expectedHbar {
   106  				found = true
   107  			}
   108  		}
   109  	}
   110  
   111  	require.True(t, found)

   113  	transferTransaction, err := NewTransferTransaction().

   116  		SetTransactionID(NewTransactionIDWithValidStart(AccountID{Shard: 3, Realm: 3, Account: 3, checksum: nil}, time.Unix(4, 4))).
   117  		SetNodeAccountIDs([]AccountID{accountID4}).
   118  		Freeze()
   119  	require.NoError(t, err)
   120  
   121  	transferTransactionToBytes, err := transferTransaction.ToBytes()

   130  	}
