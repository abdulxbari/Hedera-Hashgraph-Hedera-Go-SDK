   123  func (id *AccountID) ValidateChecksum(client *Client) error {
   124  	if id.AliasKey != nil {
   125  		return errors.New("Account ID contains alias key, unable to validate")
   126  	}
   127  	if !id._IsZero() && client != nil && client.network.ledgerID != nil {
   128  		var tempChecksum _ParseAddressResult
   129  		var err error
   130  		tempChecksum, err = _ChecksumParseAddress(client.GetLedgerID(), fmt.Sprintf("%d.%d.%d", id.Shard, id.Realm, id.Account))
   131  		if err != nil {
   132  			return err
   133  		}
   134  		err = _ChecksumVerify(tempChecksum.status)
   135  		if err != nil {
   136  			return err
   137  		}
   138  		if id.checksum == nil {
   139  			return errChecksumMissing
   140  		}
   141  		if tempChecksum.correctChecksum != *id.checksum {
   142  			temp, _ := client.network.ledgerID.ToNetworkName()
   143  			return errors.New(fmt.Sprintf("network mismatch or wrong checksum given, given checksum: %s, correct checksum %s, network: %s",
   144  				*id.checksum,
   145  				tempChecksum.correctChecksum,
   146  				temp))
   147  		}
   148  	}
   149  
   150  	return nil
   151  }

   260  func _AccountIDFromProtobuf(accountID *services.AccountID) *AccountID {
   261  	if accountID == nil {
   262  		return nil
   263  	}
   264  	resultAccountID := &AccountID{
   265  		Shard: uint64(accountID.ShardNum),
   266  		Realm: uint64(accountID.RealmNum),
   267  	}
   268  
   269  	switch t := accountID.Account.(type) {
   270  	case *services.AccountID_Alias:
   271  		pb := services.Key{}
   272  		_ = protobuf.Unmarshal(t.Alias, &pb)
   273  		initialKey, err := _KeyFromProtobuf(&pb)
   274  		if err != nil && t.Alias != nil {
   275  			resultAccountID.Account = 0
   276  			resultAccountID.AliasEvmAddress = &t.Alias
   277  			return resultAccountID
   278  		}
   279  		if evm, ok := pb.Key.(*services.Key_ECDSASecp256K1); ok && len(evm.ECDSASecp256K1) == 20 {
   280  			resultAccountID.Account = 0
   281  			resultAccountID.AliasEvmAddress = &evm.ECDSASecp256K1
   282  			return resultAccountID
   283  		}
   284  		switch t2 := initialKey.(type) {
   285  		case PublicKey:
   286  			resultAccountID.Account = 0
   287  			resultAccountID.AliasKey = &t2
   288  			return resultAccountID
   289  		default:
   290  			return &AccountID{}
   291  		}
   292  	case *services.AccountID_AccountNum:
   293  		resultAccountID.Account = uint64(t.AccountNum)
   294  		resultAccountID.AliasKey = nil
   295  		return resultAccountID
   296  	default:
   297  		return &AccountID{}
