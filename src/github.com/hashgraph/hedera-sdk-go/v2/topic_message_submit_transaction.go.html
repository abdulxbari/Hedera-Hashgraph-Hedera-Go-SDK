   306  func (transaction *TopicMessageSubmitTransaction) FreezeWith(client *Client) (*TopicMessageSubmitTransaction, error) {
   307  	var err error
   308  	if transaction.nodeAccountIDs._Length() == 0 {
   309  		if client == nil {
   310  			return transaction, errNoClientOrTransactionIDOrNodeId
   311  		}
   312  
   313  		transaction.SetNodeAccountIDs(client.network._GetNodeAccountIDsForExecute())
   314  	}
   315  
   316  	transaction._InitFee(client)
   317  	err = transaction._ValidateNetworkOnIDs(client)
   318  	if err != nil {
   319  		return &TopicMessageSubmitTransaction{}, err
   320  	}
   321  	if err := transaction._InitTransactionID(client); err != nil {
   322  		return transaction, err
   323  	}
   324  	body := transaction._Build()
   325  
   326  	chunks := uint64((len(transaction.message) + (chunkSize - 1)) / chunkSize)
   327  	if chunks > transaction.maxChunks {
   328  		return transaction, ErrMaxChunksExceeded{
   329  			Chunks:    chunks,
   330  			MaxChunks: transaction.maxChunks,
   331  		}
   332  	}
   333  
   334  	initialTransactionID := transaction.transactionIDs._GetCurrent().(TransactionID)
   335  	nextTransactionID, _ := TransactionIdFromString(initialTransactionID.String())
   336  
   337  	transaction.transactionIDs = _NewLockableSlice()
   338  	transaction.transactions = _NewLockableSlice()
   339  	transaction.signedTransactions = _NewLockableSlice()
   340  	if b, ok := body.Data.(*services.TransactionBody_ConsensusSubmitMessage); ok {
   341  		for i := 0; uint64(i) < chunks; i++ {
   342  			start := i * chunkSize
   343  			end := start + chunkSize
   344  
   345  			if end > len(transaction.message) {
   346  				end = len(transaction.message)
   347  			}
   348  
   349  			transaction.transactionIDs._Push(_TransactionIDFromProtobuf(nextTransactionID._ToProtobuf()))
   350  
   351  			b.ConsensusSubmitMessage.Message = transaction.message[start:end]
   352  			b.ConsensusSubmitMessage.ChunkInfo = &services.ConsensusMessageChunkInfo{
   353  				InitialTransactionID: initialTransactionID._ToProtobuf(),
   354  				Total:                int32(chunks),
   355  				Number:               int32(i) + 1,
   356  			}
   357  
   358  			body.TransactionID = nextTransactionID._ToProtobuf()
   359  			body.Data = &services.TransactionBody_ConsensusSubmitMessage{
   360  				ConsensusSubmitMessage: b.ConsensusSubmitMessage,
   361  			}
   362  
   363  			for _, nodeAccountID := range transaction.nodeAccountIDs.slice {
   364  				body.NodeAccountID = nodeAccountID.(AccountID)._ToProtobuf()
   365  
   366  				bodyBytes, err := protobuf.Marshal(body)
   367  				if err != nil {
   368  					return transaction, errors.Wrap(err, "error serializing transaction body for topic submission")
   369  				}
   370  
   371  				transaction.signedTransactions._Push(&services.SignedTransaction{
   372  					BodyBytes: bodyBytes,
   373  					SigMap:    &services.SignatureMap{},
   374  				})
   375  			}
   376  
   377  			validStart := *nextTransactionID.ValidStart
   378  
   379  			*nextTransactionID.ValidStart = validStart.Add(1 * time.Nanosecond)
   380  		}
   381  	}
   382  
   383  	return transaction, nil
   384  }

   461  func (transaction *TopicMessageSubmitTransaction) AddSignature(publicKey PublicKey, signature []byte) *TopicMessageSubmitTransaction {
   462  	transaction._RequireOneNodeAccountID()
   463  
   464  	if transaction._KeyAlreadySigned(publicKey) {
   465  		return transaction
   466  	}
   467  
   468  	if transaction.signedTransactions._Length() == 0 {
   469  		return transaction
   470  	}
   471  
   472  	transaction.transactions = _NewLockableSlice()
   473  	transaction.publicKeys = append(transaction.publicKeys, publicKey)
   474  	transaction.transactionSigners = append(transaction.transactionSigners, nil)
   475  	transaction.transactionIDs.locked = true
   476  
   477  	for index := 0; index < transaction.signedTransactions._Length(); index++ {
   478  		var temp *services.SignedTransaction
   479  		switch t := transaction.signedTransactions._Get(index).(type) { //nolint
   480  		case *services.SignedTransaction:

   487  		transaction.signedTransactions._Set(index, temp)
   488  	}
