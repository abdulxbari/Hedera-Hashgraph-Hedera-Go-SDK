    66  func TestUnitMockContractCallQuery(t *testing.T) {
    67  	message := "getMessage"
    68  	params := NewContractFunctionParameters()
    69  	params._Build(&message)
    70  
    71  	responses := [][]interface{}{{
    72  		&services.Response{
    73  			Response: &services.Response_ContractCallLocal{
    74  				ContractCallLocal: &services.ContractCallLocalResponse{
    75  					Header: &services.ResponseHeader{NodeTransactionPrecheckCode: services.ResponseCodeEnum_BUSY, ResponseType: services.ResponseType_ANSWER_ONLY},
    76  				},
    77  			},
    78  		},
    79  		&services.Response{
    80  			Response: &services.Response_ContractCallLocal{
    81  				ContractCallLocal: &services.ContractCallLocalResponse{
    82  					Header: &services.ResponseHeader{NodeTransactionPrecheckCode: services.ResponseCodeEnum_OK, ResponseType: services.ResponseType_ANSWER_ONLY, Cost: 2},
    83  					FunctionResult: &services.ContractFunctionResult{
    84  						ContractID:         &services.ContractID{Contract: &services.ContractID_ContractNum{ContractNum: 123}},
    85  						GasUsed:            75000,
    86  						ContractCallResult: params._Build(&message),
    87  					},
    88  				},
    89  			},
    90  		},
    91  	}}
    92  
    93  	client, server := NewMockClientAndServer(responses)
    94  	defer server.Close()
    95  
    96  	result, err := NewContractCallQuery().
    97  		SetNodeAccountIDs([]AccountID{{Account: 3}}).
    98  		SetContractID(ContractID{Contract: 123}).
    99  		SetQueryPayment(NewHbar(1)).
   100  		SetGas(100000).
   101  		SetFunction(message, nil).
   102  		SetMaxQueryPayment(NewHbar(5)).
   103  		Execute(client)
   104  	require.NoError(t, err)
   105  
   106  	require.Equal(t, bytes.Compare(result.ContractCallResult, params._Build(&message)), 0)
   107  	require.Equal(t, result.GasUsed, uint64(75000))
   108  	require.Equal(t, result.ContractID.Contract, uint64(123))
   109  }
