   332  func (transaction *LiveHashDeleteTransaction) AddSignature(publicKey PublicKey, signature []byte) *LiveHashDeleteTransaction {
   333  	transaction._RequireOneNodeAccountID()
   334  
   335  	if transaction._KeyAlreadySigned(publicKey) {
   336  		return transaction
   337  	}
   338  
   339  	if transaction.signedTransactions._Length() == 0 {
   340  		return transaction
   341  	}
   342  
   343  	transaction.transactions = _NewLockableSlice()
   344  	transaction.publicKeys = append(transaction.publicKeys, publicKey)
   345  	transaction.transactionSigners = append(transaction.transactionSigners, nil)
   346  	transaction.transactionIDs.locked = true
   347  
   348  	for index := 0; index < transaction.signedTransactions._Length(); index++ {
   349  		var temp *services.SignedTransaction
   350  		switch t := transaction.signedTransactions._Get(index).(type) { //nolint
   351  		case *services.SignedTransaction:
   352  			temp = t
   353  		}
   354  		temp.SigMap.SigPair = append(
   355  			temp.SigMap.SigPair,
   356  			publicKey._ToSignaturePairProtobuf(signature),
   357  		)
   358  		transaction.signedTransactions._Set(index, temp)
   359  	}
   360  
   361  	return transaction
