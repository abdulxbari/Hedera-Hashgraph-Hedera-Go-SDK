    36  func TestIntegrationTopicUpdateTransactionCanExecute(t *testing.T) {
    37  	env := NewIntegrationTestEnv(t)
    38  
    39  	resp, err := NewTopicCreateTransaction().
    40  		SetAdminKey(env.Client.GetOperatorPublicKey()).
    41  		SetNodeAccountIDs(env.NodeAccountIDs).
    42  		SetTopicMemo(oldTopicMemo).
    43  		Execute(env.Client)
    44  
    45  	require.NoError(t, err)
    46  
    47  	receipt, err := resp.SetValidateStatus(true).GetReceipt(env.Client)
    48  	require.NoError(t, err)
    49  
    50  	topicID := *receipt.TopicID
    51  	require.NoError(t, err)
    52  
    53  	info, err := NewTopicInfoQuery().
    54  		SetTopicID(topicID).
    55  		SetNodeAccountIDs([]AccountID{resp.NodeID}).
    56  		SetMaxQueryPayment(NewHbar(1)).
    57  		Execute(env.Client)
    58  	require.NoError(t, err)
    59  	assert.NotNil(t, info)
    60  
    61  	assert.Equal(t, oldTopicMemo, info.TopicMemo)
    62  	assert.Equal(t, uint64(0), info.SequenceNumber)
    63  	assert.Equal(t, env.Client.GetOperatorPublicKey().String(), info.AdminKey.String())
    64  
    65  	newTopicMemo := "go-sdk::TestConsensusTopicUpdateTransaction_Execute::updated"
    66  
    67  	resp, err = NewTopicUpdateTransaction().
    68  		SetTopicID(topicID).
    69  		SetNodeAccountIDs([]AccountID{resp.NodeID}).
    70  		SetTopicMemo(newTopicMemo).
    71  		Execute(env.Client)
    72  	require.NoError(t, err)
    73  
    74  	_, err = resp.SetValidateStatus(true).GetReceipt(env.Client)
    75  	require.NoError(t, err)
    76  
    77  	info, err = NewTopicInfoQuery().
    78  		SetTopicID(topicID).
    79  		SetNodeAccountIDs([]AccountID{resp.NodeID}).
    80  		SetMaxQueryPayment(NewHbar(1)).
    81  		Execute(env.Client)
    82  	require.NoError(t, err)
    83  	assert.NotNil(t, info)
    84  
    85  	assert.Equal(t, newTopicMemo, info.TopicMemo)
    86  	assert.Equal(t, uint64(0), info.SequenceNumber)
    87  	assert.Equal(t, env.Client.GetOperatorPublicKey().String(), info.AdminKey.String())
    88  
    89  	resp, err = NewTopicDeleteTransaction().
    90  		SetTopicID(topicID).
    91  		SetNodeAccountIDs([]AccountID{resp.NodeID}).
    92  		Execute(env.Client)
    93  	require.NoError(t, err)
    94  
    95  	_, err = resp.SetValidateStatus(true).GetReceipt(env.Client)
    96  	require.NoError(t, err)
    97  
    98  	err = CloseIntegrationTestEnv(env, nil)
    99  	require.NoError(t, err)
   100  }
