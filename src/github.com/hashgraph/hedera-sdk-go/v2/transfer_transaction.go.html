    33  // TransferTransaction
    34  // Transfers cryptocurrency among two or more accounts by making the desired adjustments to their
    35  // balances. Each transfer list can specify up to 10 adjustments. Each negative amount is withdrawn
    36  // from the corresponding account (a sender), and each positive one is added to the corresponding
    37  // account (a receiver). The amounts list must sum to zero. Each amount is a number of tinybars
    38  // (there are 100,000,000 tinybars in one hbar).  If any sender account fails to have sufficient
    39  // hbars, then the entire transaction fails, and none of those transfers occur, though the
    40  // transaction fee is still charged. This transaction must be signed by the keys for all the sending
    41  // accounts, and for any receiving accounts that have receiverSigRequired == true. The signatures
    42  // are in the same order as the accounts, skipping those accounts that don't need a signature.
    43  type TransferTransaction struct {
    44  	Transaction
    45  	tokenTransfers map[TokenID]*_TokenTransfer
    46  	hbarTransfers  []*_HbarTransfer
    47  	nftTransfers   map[TokenID][]*TokenNftTransfer
    48  }

    73  func _TransferTransactionFromProtobuf(transaction Transaction, pb *services.TransactionBody) *TransferTransaction {
    74  	tokenTransfers := make(map[TokenID]*_TokenTransfer)
    75  	nftTransfers := make(map[TokenID][]*TokenNftTransfer)
    76  
    77  	for _, tokenTransfersList := range pb.GetCryptoTransfer().GetTokenTransfers() {
    78  		tok := _TokenIDFromProtobuf(tokenTransfersList.Token)
    79  		tokenTransfers[*tok] = _TokenTransferPrivateFromProtobuf(tokenTransfersList)
    80  	}
    81  
    82  	for _, tokenTransfersList := range pb.GetCryptoTransfer().GetTokenTransfers() {
    83  		if tokenID := _TokenIDFromProtobuf(tokenTransfersList.Token); tokenID != nil {
    84  			for _, aa := range tokenTransfersList.GetNftTransfers() {
    85  				if nftTransfers[*tokenID] == nil {
    86  					nftTransfers[*tokenID] = make([]*TokenNftTransfer, 0)
    87  				}
    88  				nftTransfer := _NftTransferFromProtobuf(aa)
    89  				nftTransfers[*tokenID] = append(nftTransfers[*tokenID], &nftTransfer)
    90  			}
    91  		}
    92  	}
    93  
    94  	return &TransferTransaction{
    95  		Transaction:    transaction,
    96  		hbarTransfers:  _HbarTransferFromProtobuf(pb.GetCryptoTransfer().GetTransfers().GetAccountAmounts()),
    97  		tokenTransfers: tokenTransfers,
    98  		nftTransfers:   nftTransfers,
    99  	}

   441  func (transaction *TransferTransaction) _ValidateNetworkOnIDs(client *Client) error {
   442  	if client == nil || !client.autoValidateChecksums {
   443  		return nil
   444  	}
   445  	var err error
   446  	for token, tokenTransfer := range transaction.tokenTransfers {
   447  		err = token.ValidateChecksum(client)
   448  		for _, transfer := range tokenTransfer.Transfers {
   449  			err = transfer.accountID.ValidateChecksum(client)
   450  			if err != nil {
   451  				return err
   452  			}
   453  		}
   454  		if err != nil {
   455  			return err
   456  		}
   457  	}
   458  	for token, nftTransfers := range transaction.nftTransfers {
   459  		err = token.ValidateChecksum(client)
   460  		if err != nil {
   461  			return err
   462  		}
   463  		for _, nftTransfer := range nftTransfers {
   464  			err = nftTransfer.SenderAccountID.ValidateChecksum(client)
   465  			if err != nil {
   466  				return err
   467  			}
   468  			err = nftTransfer.ReceiverAccountID.ValidateChecksum(client)

   474  	for _, hbarTransfer := range transaction.hbarTransfers {

   495  func (transaction *TransferTransaction) _ConstructScheduleProtobuf() (*services.SchedulableTransactionBody, error) {
   496  	body := &services.CryptoTransferTransactionBody{
   497  		Transfers: &services.TransferList{
   498  			AccountAmounts: []*services.AccountAmount{},
   499  		},
   500  		TokenTransfers: []*services.TokenTransferList{},
   501  	}
   502  
   503  	sort.Sort(&_HbarTransfers{transaction.hbarTransfers})
   504  
   505  	if len(transaction.hbarTransfers) > 0 {
   506  		body.Transfers.AccountAmounts = make([]*services.AccountAmount, 0)
   507  		for _, hbarTransfer := range transaction.hbarTransfers {
   508  			body.Transfers.AccountAmounts = append(body.Transfers.AccountAmounts, &services.AccountAmount{
   509  				AccountID:  hbarTransfer.accountID._ToProtobuf(),
   510  				Amount:     hbarTransfer.Amount.AsTinybar(),
   511  				IsApproval: hbarTransfer.IsApproved,
   512  			})
   513  		}
   514  	}
   515  
   516  	tempTokenIDarray := make([]TokenID, 0)
   517  	for k := range transaction.tokenTransfers {
   518  		tempTokenIDarray = append(tempTokenIDarray, k)
   519  	}
   520  	sort.Sort(_TokenIDs{tokenIDs: tempTokenIDarray})
   521  
   522  	for _, k := range tempTokenIDarray {
   523  		sort.Sort(&_HbarTransfers{transfers: transaction.tokenTransfers[k].Transfers})
   524  	}
   525  
   526  	if len(tempTokenIDarray) > 0 {
   527  		if body.TokenTransfers == nil {
   528  			body.TokenTransfers = make([]*services.TokenTransferList, 0)
   529  		}
   530  
   531  		for _, tokenID := range tempTokenIDarray {
   532  			transfers := transaction.tokenTransfers[tokenID]._ToProtobuf()
   533  
   534  			bod := &services.TokenTransferList{
   535  				Token:     tokenID._ToProtobuf(),
   536  				Transfers: transfers,
   537  			}
   538  
   539  			if transaction.tokenTransfers[tokenID].ExpectedDecimals != nil {
   540  				bod.ExpectedDecimals = &wrapperspb.UInt32Value{Value: *transaction.tokenTransfers[tokenID].ExpectedDecimals}
   541  			}
   542  
   543  			body.TokenTransfers = append(body.TokenTransfers, bod)
   544  		}
   545  	}
   546  
   547  	tempTokenIDarray = make([]TokenID, 0)
   548  	for k := range transaction.nftTransfers {
   549  		tempTokenIDarray = append(tempTokenIDarray, k)
   550  	}
   551  	sort.Sort(_TokenIDs{tokenIDs: tempTokenIDarray})
   552  
   553  	tempNftTransfers := make(map[TokenID][]*TokenNftTransfer)
   554  	for _, k := range tempTokenIDarray {
   555  		tempTokenNftTransfer := transaction.nftTransfers[k]
   556  
   557  		sort.Sort(&_TokenNftTransfers{tempTokenNftTransfer})
   558  
   559  		tempNftTransfers[k] = tempTokenNftTransfer
   560  	}
   561  
   562  	if len(tempTokenIDarray) > 0 {
   563  		if body.TokenTransfers == nil {
   564  			body.TokenTransfers = make([]*services.TokenTransferList, 0)
   565  		}
   566  
   567  		for _, tokenID := range tempTokenIDarray {
   568  			nftTransfers := make([]*services.NftTransfer, 0)
   569  
   570  			for _, nftT := range tempNftTransfers[tokenID] {
   571  				nftTransfers = append(nftTransfers, nftT._ToProtobuf())

   639  func (transaction *TransferTransaction) SignWithOperator(
   640  	client *Client,
   641  ) (*TransferTransaction, error) {
   642  	// If the transaction is not signed by the _Operator, we need
   643  	// to sign the transaction with the _Operator
   644

   673  // Execute executes the Transaction with the provided client
   674  func (transaction *TransferTransaction) Execute(
   675  	client *Client,
   676  ) (TransactionResponse, error) {
   677  	if client == nil {
   678  		return TransactionResponse{}, errNoClientProvided
   679  	}
   680  
   681  	if transaction.freezeError != nil {
   682  		return TransactionResponse{}, transaction.freezeError
   683  	}
   684  
   685  	if !transaction.IsFrozen() {
   686  		_, err := transaction.FreezeWith(client)
   687  		if err != nil {
   688  			return TransactionResponse{}, err
   689  		}
   690  	}
   691  
   692  	transactionID := transaction.transactionIDs._GetCurrent().(TransactionID)
   693  
   694  	if !client.GetOperatorAccountID()._IsZero() && client.GetOperatorAccountID()._Equals(*transactionID.AccountID) {
   695  		transaction.SignWith(
   696  			client.GetOperatorPublicKey(),
   697  			client.operator.signer,
   698  		)
   699  	}
   700  
   701  	resp, err := _Execute(
   702  		client,
   703  		&transaction.Transaction,
   704  		_TransactionShouldRetry,
   705  		_TransactionMakeRequest,
   706  		_TransactionAdvanceRequest,
   707  		_TransactionGetNodeAccountID,
   708  		_TransferTransactionGetMethod,
   709  		_TransactionMapStatusError,
   710  		_TransactionMapResponse,
   711  		transaction._GetLogID(),
   712  		transaction.grpcDeadline,
   713  		transaction.maxBackoff,
   714  		transaction.minBackoff,
   715  		transaction.maxRetry,
   716  	)
   717  
   718  	if err != nil {
   719  		return TransactionResponse{
   720  			TransactionID:  transaction.GetTransactionID(),
   721  			NodeID:         resp.(TransactionResponse).NodeID,
   722  			ValidateStatus: true,
   723  		}, err
   724  	}
   725  
   726  	return TransactionResponse{
   727  		TransactionID:  transaction.GetTransactionID(),
   728  		NodeID:         resp.(TransactionResponse).NodeID,
   729  		Hash:           resp.(TransactionResponse).Hash,
   730  		ValidateStatus: true,
   731  	}, nil
   732  }

   734  func (transaction *TransferTransaction) _Build() *services.TransactionBody {
   735  	body := &services.CryptoTransferTransactionBody{
   736  		Transfers: &services.TransferList{
   737  			AccountAmounts: []*services.AccountAmount{},
   738  		},
   739  		TokenTransfers: []*services.TokenTransferList{},
   740  	}
   741  
   742  	sort.Sort(&_HbarTransfers{transaction.hbarTransfers})
   743  
   744  	if len(transaction.hbarTransfers) > 0 {
   745  		body.Transfers.AccountAmounts = make([]*services.AccountAmount, 0)
   746  		for _, hbarTransfer := range transaction.hbarTransfers {
   747  			body.Transfers.AccountAmounts = append(body.Transfers.AccountAmounts, &services.AccountAmount{
   748  				AccountID:  hbarTransfer.accountID._ToProtobuf(),
   749  				Amount:     hbarTransfer.Amount.AsTinybar(),
   750  				IsApproval: hbarTransfer.IsApproved,
   751  			})
   752  		}
   753  	}
   754  
   755  	tempTokenIDarray := make([]TokenID, 0)
   756  	for k := range transaction.tokenTransfers {
   757  		tempTokenIDarray = append(tempTokenIDarray, k)
   758  	}
   759  	sort.Sort(_TokenIDs{tokenIDs: tempTokenIDarray})
   760  
   761  	for _, k := range tempTokenIDarray {
   762  		sort.Sort(&_HbarTransfers{transfers: transaction.tokenTransfers[k].Transfers})
   763  	}
   764  
   765  	if len(tempTokenIDarray) > 0 {
   766  		if body.TokenTransfers == nil {
   767  			body.TokenTransfers = make([]*services.TokenTransferList, 0)
   768  		}
   769  
   770  		for _, tokenID := range tempTokenIDarray {
   771  			transfers := transaction.tokenTransfers[tokenID]._ToProtobuf()
   772  
   773  			bod := &services.TokenTransferList{
   774  				Token:     tokenID._ToProtobuf(),
   775  				Transfers: transfers,
   776  			}
   777  
   778  			if transaction.tokenTransfers[tokenID].ExpectedDecimals != nil {
   779  				bod.ExpectedDecimals = &wrapperspb.UInt32Value{Value: *transaction.tokenTransfers[tokenID].ExpectedDecimals}
   780  			}
   781  
   782  			body.TokenTransfers = append(body.TokenTransfers, bod)
   783  		}
   784  	}
   785  
   786  	tempTokenIDarray = make([]TokenID, 0)
   787  	for k := range transaction.nftTransfers {
   788  		tempTokenIDarray = append(tempTokenIDarray, k)
   789  	}
   790  	sort.Sort(_TokenIDs{tokenIDs: tempTokenIDarray})
   791  
   792  	tempNftTransfers := make(map[TokenID][]*TokenNftTransfer)
   793  	for _, k := range tempTokenIDarray {
   794  		tempTokenNftTransfer := transaction.nftTransfers[k]
   795  
   796  		sort.Sort(&_TokenNftTransfers{tempTokenNftTransfer})
   797  
   798  		tempNftTransfers[k] = tempTokenNftTransfer
   799  	}
   800  
   801  	if len(tempTokenIDarray) > 0 {
   802  		if body.TokenTransfers == nil {
   803  			body.TokenTransfers = make([]*services.TokenTransferList, 0)
   804  		}
   805  
   806  		for _, tokenID := range tempTokenIDarray {
   807  			nftTransfers := make([]*services.NftTransfer, 0)
   808  
   809  			for _, nftT := range tempNftTransfers[tokenID] {
   810  				nftTransfers = append(nftTransfers, nftT._ToProtobuf())
   811  			}
   812  
   813  			body.TokenTransfers = append(body.TokenTransfers, &services.TokenTransferList{
   814  				Token:        tokenID._ToProtobuf(),
   815  				NftTransfers: nftTransfers,
   816  			})
   817  		}
   818  	}
   819  
   820  	return &services.TransactionBody{
   821  		TransactionFee:           transaction.transactionFee,
   822  		Memo:                     transaction.Transaction.memo,
   823  		TransactionValidDuration: _DurationToProtobuf(transaction.GetTransactionValidDuration()),
   824  		TransactionID:            transaction.transactionID._ToProtobuf(),
   825  		Data: &services.TransactionBody_CryptoTransfer{
   826  			CryptoTransfer: body,
   827  		},
   828  	}
   829  }

   835  func (transaction *TransferTransaction) FreezeWith(client *Client) (*TransferTransaction, error) {
   836  	if transaction.IsFrozen() {
   837  		return transaction, nil
   838  	}
   839  
   840  	transaction._InitFee(client)
   841  	if err := transaction._InitTransactionID(client); err != nil {
   842  		return transaction, err
   843  	}
   844  	err := transaction._ValidateNetworkOnIDs(client)
   845  	if err != nil {
   846  		return &TransferTransaction{}, err
   847  	}
   848  	body := transaction._Build()
   849  
   850  	return transaction, _TransactionFreezeWith(&transaction.Transaction, client, body)
   851  }
