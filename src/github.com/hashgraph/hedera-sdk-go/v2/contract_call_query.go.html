    30  // ContractCallQuery calls a function of the given smart contract instance, giving it ContractFunctionParameters as its
    31  // inputs. It will consume the entire given amount of gas.
    32  //
    33  // This is performed locally on the particular _Node that the client is communicating with. It cannot change the state of
    34  // the contract instance (and so, cannot spend anything from the instance's Hedera account). It will not have a
    35  // consensus timestamp. It cannot generate a record or a receipt. This is useful for calling getter functions, which
    36  // purely read the state and don't change it. It is faster and cheaper than a ContractExecuteTransaction, because it is
    37  // purely local to a single  _Node.
    38  type ContractCallQuery struct {
    39  	Query
    40  	contractID         *ContractID
    41  	gas                uint64
    42  	maxResultSize      uint64
    43  	functionParameters []byte
    44  	senderID           *AccountID
    45  }

   129  func (query *ContractCallQuery) _ValidateNetworkOnIDs(client *Client) error {
   130  	if client == nil || !client.autoValidateChecksums {
   131  		return nil
   132  	}
   133  
   134  	if query.contractID != nil {
   135  		if err := query.contractID.ValidateChecksum(client); err != nil {
   136  			return err
   137  		}
   138  	}
   139  
   140  	if query.senderID != nil {
   141  		if err := query.senderID.ValidateChecksum(client); err != nil {
   142  			return err
   143  		}
   144  	}
   145  
   146  	return nil
   147  }

   172  func (query *ContractCallQuery) GetCost(client *Client) (Hbar, error) {
   173  	if client == nil || client.operator == nil {
   174  		return Hbar{}, errNoClientProvided
   175  	}
   176  
   177  	var err error
   178  
   179  	err = query._ValidateNetworkOnIDs(client)
   180  	if err != nil {
   181  		return Hbar{}, err
   182  	}
   183  
   184  	for range query.nodeAccountIDs.slice {
   185  		paymentTransaction, err := _QueryMakePaymentTransaction(TransactionID{}, AccountID{}, client.operator, Hbar{})
   186  		if err != nil {
   187  			return Hbar{}, err
   188  		}
   189  		query.paymentTransactions = append(query.paymentTransactions, paymentTransaction)
   190  	}
   191  
   192  	pb := query._Build()
   193  	pb.ContractCallLocal.Header = query.pbHeader
   194  
   195  	query.pb = &services.Query{
   196  		Query: pb,
   197  	}
   198  
   199  	resp, err := _Execute(
   200  		client,
   201  		&query.Query,
   202  		_ContractCallQueryShouldRetry,
   203  		_CostQueryMakeRequest,
   204  		_CostQueryAdvanceRequest,
   205  		_QueryGetNodeAccountID,
   206  		_ContractCallQueryGetMethod,
   207  		_ContractCallQueryMapStatusError,
   208  		_QueryMapResponse,
   209  		query._GetLogID(),
   210  		query.grpcDeadline,
   211  		query.maxBackoff,
   212  		query.minBackoff,
   213  		query.maxRetry,
   214  	)
   215  
   216  	if err != nil {
   217  		return Hbar{}, err
   218  	}
   219  
   220  	cost := int64(resp.(*services.Response).GetContractCallLocal().Header.Cost)
   221  	return HbarFromTinybar(cost), nil
   222  }

   240  func (query *ContractCallQuery) Execute(client *Client) (ContractFunctionResult, error) {
   241  	if client == nil || client.operator == nil {
   242  		return ContractFunctionResult{}, errNoClientProvided
   243  	}
   244  
   245  	var err error
   246  
   247  	err = query._ValidateNetworkOnIDs(client)
   248  	if err != nil {
   249  		return ContractFunctionResult{}, err
   250  	}
   251  
   252  	if !query.paymentTransactionIDs.locked {
   253  		query.paymentTransactionIDs._Clear()._Push(TransactionIDGenerate(client.operator.accountID))
   254  	}
   255  
   256  	var cost Hbar
   257  	if query.queryPayment.tinybar != 0 {
   258  		cost = query.queryPayment
   259  	} else {
   260  		if query.maxQueryPayment.tinybar == 0 {
   261  			cost = client.GetDefaultMaxQueryPayment()
   262  		} else {
   263  			cost = query.maxQueryPayment
   264  		}
   265  
   266  		actualCost, err := query.GetCost(client)
   267  		if err != nil {
   268  			return ContractFunctionResult{}, err
   269  		}
   270  
   271  		if cost.tinybar < actualCost.tinybar {
   272  			return ContractFunctionResult{}, ErrMaxQueryPaymentExceeded{
   273  				QueryCost:       actualCost,
   274  				MaxQueryPayment: cost,
   275  				query:           "ContractFunctionResultQuery",
   276  			}
   277  		}
   278  
   279  		cost = actualCost
   280  	}
   281  
   282  	query.paymentTransactions = make([]*services.Transaction, 0)
   283  
   284  	if query.nodeAccountIDs.locked {
   285  		err = _QueryGeneratePayments(&query.Query, client, cost)
   286  		if err != nil {
   287  			return ContractFunctionResult{}, err
   288  		}
   289  	} else {
   290  		paymentTransaction, err := _QueryMakePaymentTransaction(query.paymentTransactionIDs._GetCurrent().(TransactionID), AccountID{}, client.operator, cost)
   291  		if err != nil {
   292  			return ContractFunctionResult{}, err
   293  		}
   294  		query.paymentTransactions = append(query.paymentTransactions, paymentTransaction)
   295  	}
   296  
   297  	pb := query._Build()
   298  	pb.ContractCallLocal.Header = query.pbHeader
   299  	query.pb = &services.Query{
   300  		Query: pb,
   301  	}
   302  
   303  	resp, err := _Execute(
   304  		client,
   305  		&query.Query,
   306  		_ContractCallQueryShouldRetry,
   307  		_QueryMakeRequest,
   308  		_QueryAdvanceRequest,
   309  		_QueryGetNodeAccountID,
   310  		_ContractCallQueryGetMethod,
   311  		_ContractCallQueryMapStatusError,
   312  		_QueryMapResponse,
   313  		query._GetLogID(),
   314  		query.grpcDeadline,
   315  		query.maxBackoff,
   316  		query.minBackoff,
   317  		query.maxRetry,
   318  	)
   319  
   320  	if err != nil {
   321  		return ContractFunctionResult{}, err
   322  	}
   323  
   324  	return _ContractFunctionResultFromProtobuf(resp.(*services.Response).GetContractCallLocal().FunctionResult), nil
   325  }
