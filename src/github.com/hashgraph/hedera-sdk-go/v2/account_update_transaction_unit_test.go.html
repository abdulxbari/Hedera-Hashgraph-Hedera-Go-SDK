   222  		SetKey(key).

   225  		SetReceiverSignatureRequired(true).
   226  		SetMaxAutomaticTokenAssociations(2).

   245  	require.Equal(t, proto.StakedId.(*services.CryptoUpdateTransactionBody_StakedAccountId).StakedAccountId.String(),

   247  	require.Equal(t, proto.DeclineReward.Value, true)
   248  	require.Equal(t, proto.AutoRenewPeriod.String(), _DurationToProtobuf(60*time.Second).String())
   249  	require.Equal(t, proto.ExpirationTime.String(), _TimeToProtobuf(time.Unix(34, 56)).String())
