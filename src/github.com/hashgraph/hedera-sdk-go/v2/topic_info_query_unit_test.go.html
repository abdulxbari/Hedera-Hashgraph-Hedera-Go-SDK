    65  func TestUnitTopicInfoQueryGet(t *testing.T) {
    66  	checksum := "dmqui"
    67  	topicID := TopicID{Topic: 3, checksum: &checksum}
    68  	deadline := time.Duration(time.Minute)
    69  	accountId := AccountID{Account: 123}
    70  	transactionID := TransactionIDGenerate(accountId)
    71  	query := NewTopicInfoQuery().
    72  		SetTopicID(topicID).
    73  		SetQueryPayment(NewHbar(2)).
    74  		SetMaxQueryPayment(NewHbar(1)).
    75  		SetQueryPayment(HbarFromTinybar(25)).
    76  		SetNodeAccountIDs([]AccountID{{Account: 10}, {Account: 11}, {Account: 12}}).
    77  		SetMaxRetry(3).
    78  		SetMinBackoff(300 * time.Millisecond).
    79  		SetMaxBackoff(10 * time.Second).
    80  		SetPaymentTransactionID(transactionID).
    81  		SetMaxQueryPayment(NewHbar(500)).
    82  		SetGrpcDeadline(&deadline)
    83  	client := ClientForTestnet()
    84  	client.SetAutoValidateChecksums(true)
    85  	err := query._ValidateNetworkOnIDs(client)
    86  	require.NoError(t, err)
    87  	require.Equal(t, topicID, query.GetTopicID())
    88  	require.Equal(t, []AccountID{{Account: 10}, {Account: 11}, {Account: 12}}, query.GetNodeAccountIDs())
    89  	require.Equal(t, 300*time.Millisecond, query.GetMinBackoff())
    90  	require.Equal(t, 10*time.Second, query.GetMaxBackoff())
    91  	require.Equal(t, 3, query.GetMaxRetryCount())
    92  	require.Equal(t, transactionID, query.GetPaymentTransactionID())
    93  	require.Equal(t, HbarFromTinybar(25), query.GetQueryPayment())
    94  	require.Equal(t, NewHbar(500), query.GetMaxQueryPayment())
    95  	require.Equal(t, &deadline, query.GetGrpcDeadline())
    96  	require.Equal(t, fmt.Sprintf("TopicInfoQuery:%v", transactionID.ValidStart.UnixNano()), query._GetLogID())
    97  }
