   193  func (transaction *EthereumTransaction) Execute(
   194  	client *Client,
   195  ) (TransactionResponse, error) {
   196  	if client == nil {
   197  		return TransactionResponse{}, errNoClientProvided
   198  	}
   199  
   200  	if transaction.freezeError != nil {
   201  		return TransactionResponse{}, transaction.freezeError
   202  	}
   203  
   204  	if !transaction.IsFrozen() {
   205  		_, err := transaction.FreezeWith(client)
   206  		if err != nil {
   207  			return TransactionResponse{}, err
   208  		}
   209  	}
   210  
   211  	transactionID := transaction.transactionIDs._GetCurrent().(TransactionID)
   212  
   213  	if !client.GetOperatorAccountID()._IsZero() && client.GetOperatorAccountID()._Equals(*transactionID.AccountID) {
   214  		transaction.SignWith(
   215  			client.GetOperatorPublicKey(),
   216  			client.operator.signer,
   217  		)
   218  	}
   219  
   220  	if transaction.grpcDeadline == nil {
   221  		transaction.grpcDeadline = client.requestTimeout
   222  	}
   223  
   224  	resp, err := _Execute(
   225  		client,
   226  		&transaction.Transaction,
   227  		_TransactionShouldRetry,
   228  		_TransactionMakeRequest,
   229  		_TransactionAdvanceRequest,

   347  func (transaction *EthereumTransaction) AddSignature(publicKey PublicKey, signature []byte) *EthereumTransaction {
   348  	transaction._RequireOneNodeAccountID()
   349  
   350  	if transaction._KeyAlreadySigned(publicKey) {
   351  		return transaction
   352  	}
   353  
   354  	if transaction.signedTransactions._Length() == 0 {
   355  		return transaction
   356  	}
   357  
   358  	transaction.transactions = _NewLockableSlice()
   359  	transaction.publicKeys = append(transaction.publicKeys, publicKey)
   360  	transaction.transactionSigners = append(transaction.transactionSigners, nil)
   361  	transaction.transactionIDs.locked = true
   362  
   363  	for index := 0; index < transaction.signedTransactions._Length(); index++ {
   364  		var temp *services.SignedTransaction
   365  		switch t := transaction.signedTransactions._Get(index).(type) { //nolint
   366  		case *services.SignedTransaction:
   367  			temp = t
   368  		}
   369  		temp.SigMap.SigPair = append(
   370  			temp.SigMap.SigPair,
   371  			publicKey._ToSignaturePairProtobuf(signature),
   372  		)
   373  		transaction.signedTransactions._Set(index, temp)
   374  	}
   375  
   376  	return transaction
   377  }
