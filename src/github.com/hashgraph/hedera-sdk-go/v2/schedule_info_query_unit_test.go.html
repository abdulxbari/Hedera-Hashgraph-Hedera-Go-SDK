    66  func TestUnitScheduleInfoQueryGet(t *testing.T) {
    67  	checksum := "dmqui"
    68  	accountId := AccountID{Account: 123}
    69  	deadline := time.Duration(time.Minute)
    70  	validStart := time.Now().Add(10 * time.Minute)
    71  	scheduleID := ScheduleID{Schedule: 3, checksum: &checksum}
    72  
    73  	query := NewScheduleInfoQuery().
    74  		SetScheduleID(scheduleID).
    75  		SetQueryPayment(NewHbar(2)).
    76  		SetMaxQueryPayment(NewHbar(1)).
    77  		SetQueryPayment(HbarFromTinybar(25)).
    78  		SetNodeAccountIDs([]AccountID{{Account: 10}, {Account: 11}, {Account: 12}}).
    79  		SetMaxRetry(3).
    80  		SetMinBackoff(300 * time.Millisecond).
    81  		SetMaxBackoff(10 * time.Second).
    82  		SetPaymentTransactionID(TransactionID{AccountID: &accountId, ValidStart: &validStart}).
    83  		SetMaxQueryPayment(NewHbar(500)).
    84  		SetGrpcDeadline(&deadline)
    85  	client := ClientForTestnet()
    86  	client.SetAutoValidateChecksums(true)
    87  	err := query._ValidateNetworkOnIDs(client)
    88  	require.NoError(t, err)
    89  	require.Equal(t, scheduleID, query.GetScheduleID())
    90  	require.Equal(t, []AccountID{{Account: 10}, {Account: 11}, {Account: 12}}, query.GetNodeAccountIDs())
    91  	require.Equal(t, 300*time.Millisecond, query.GetMinBackoff())
    92  	require.Equal(t, 10*time.Second, query.GetMaxBackoff())
    93  	require.Equal(t, 3, query.GetMaxRetryCount())
    94  	require.Equal(t, TransactionID{AccountID: &AccountID{Account: 123}, ValidStart: &validStart}, query.GetPaymentTransactionID())
    95  	require.Equal(t, HbarFromTinybar(25), query.GetQueryPayment())
    96  	require.Equal(t, NewHbar(500), query.GetMaxQueryPayment())
    97  	require.Equal(t, &deadline, query.GetGrpcDeadline())
    98  }

   128  		SetNodeAccountIDs(nodeAccountID).
   129  		SetPaymentTransactionID(transactionID).
   130  		SetMaxQueryPayment(NewHbar(23)).
   131  		SetQueryPayment(NewHbar(3)).
