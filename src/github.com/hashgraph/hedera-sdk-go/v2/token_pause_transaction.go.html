    30  // TokenPauseTransaction
    31  // Pauses the Token from being involved in any kind of Transaction until it is unpaused.
    32  // Must be signed with the Token's pause key.
    33  // If the provided token is not found, the transaction will resolve to INVALID_TOKEN_ID.
    34  // If the provided token has been deleted, the transaction will resolve to TOKEN_WAS_DELETED.
    35  // If no Pause Key is defined, the transaction will resolve to TOKEN_HAS_NO_PAUSE_KEY.
    36  // Once executed the Token is marked as paused and will be not able to be a part of any transaction.
    37  // The operation is idempotent - becomes a no-op if the Token is already Paused.
    38  type TokenPauseTransaction struct {
    39  	Transaction
    40  	tokenID *TokenID
    41  }

   161  	client *Client,
   162  ) (*TokenPauseTransaction, error) {
   163  	// If the transaction is not signed by the _Operator, we need
   164  	// to sign the transaction with the _Operator

   194  // Execute executes the Transaction with the provided client
   195  func (transaction *TokenPauseTransaction) Execute(
   196  	client *Client,
   197  ) (TransactionResponse, error) {
   198  	if client == nil {
   199  		return TransactionResponse{}, errNoClientProvided
   200  	}
   201  
   202  	if transaction.freezeError != nil {
   203  		return TransactionResponse{}, transaction.freezeError
   204  	}
   205  
   206  	if !transaction.IsFrozen() {
   207  		_, err := transaction.FreezeWith(client)
   208  		if err != nil {
   209  			return TransactionResponse{}, err
   210  		}
   211  	}
   212  
   213  	transactionID := transaction.transactionIDs._GetCurrent().(TransactionID)
   214  
   215  	if !client.GetOperatorAccountID()._IsZero() && client.GetOperatorAccountID()._Equals(*transactionID.AccountID) {
   216  		transaction.SignWith(
   217  			client.GetOperatorPublicKey(),
   218  			client.operator.signer,
   219  		)
   220  	}
   221  
   222  	resp, err := _Execute(
   223  		client,
   224  		&transaction.Transaction,
   225  		_TransactionShouldRetry,
   226  		_TransactionMakeRequest,
   227  		_TransactionAdvanceRequest,
   228  		_TransactionGetNodeAccountID,
   229  		_TokenPauseTransactionGetMethod,
   230  		_TransactionMapStatusError,
   231  		_TransactionMapResponse,
   232  		transaction._GetLogID(),
   233  		transaction.grpcDeadline,
   234  		transaction.maxBackoff,
   235  		transaction.minBackoff,
   236  		transaction.maxRetry,
   237  	)
   238  
   239  	if err != nil {
   240  		return TransactionResponse{
   241  			TransactionID:  transaction.GetTransactionID(),
   242  			NodeID:         resp.(TransactionResponse).NodeID,
   243  			ValidateStatus: true,
   244  		}, err
   245  	}
   246  
   247  	return TransactionResponse{
   248  		TransactionID:  transaction.GetTransactionID(),
   249  		NodeID:         resp.(TransactionResponse).NodeID,
   250  		Hash:           resp.(TransactionResponse).Hash,
   251  		ValidateStatus: true,
   252  	}, nil
   253  }

   259  func (transaction *TokenPauseTransaction) FreezeWith(client *Client) (*TokenPauseTransaction, error) {
   260  	if transaction.IsFrozen() {
   261  		return transaction, nil
   262  	}
   263  	transaction._InitFee(client)
   264  	err := transaction._ValidateNetworkOnIDs(client)
   265  	if err != nil {
   266  		return &TokenPauseTransaction{}, err
   267  	}
   268  	if err := transaction._InitTransactionID(client); err != nil {
   269  		return transaction, err
   270  	}
   271  	body := transaction._Build()
   272  
   273  	return transaction, _TransactionFreezeWith(&transaction.Transaction, client, body)
   274  }

   345  func (transaction *TokenPauseTransaction) AddSignature(publicKey PublicKey, signature []byte) *TokenPauseTransaction {
   346  	transaction._RequireOneNodeAccountID()
   347  
   348  	if transaction._KeyAlreadySigned(publicKey) {
   349  		return transaction
   350  	}
   351  
   352  	if transaction.signedTransactions._Length() == 0 {
   353  		return transaction
   354  	}
   355  
   356  	transaction.transactions = _NewLockableSlice()
   357  	transaction.publicKeys = append(transaction.publicKeys, publicKey)
   358  	transaction.transactionSigners = append(transaction.transactionSigners, nil)
   359  	transaction.transactionIDs.locked = true
   360  
   361  	for index := 0; index < transaction.signedTransactions._Length(); index++ {
   362  		var temp *services.SignedTransaction
   363  		switch t := transaction.signedTransactions._Get(index).(type) { //nolint
   364  		case *services.SignedTransaction:
   365  			temp = t
   366  		}
   367  		temp.SigMap.SigPair = append(
   368  			temp.SigMap.SigPair,
   369  			publicKey._ToSignaturePairProtobuf(signature),
   370  		)
   371  		transaction.signedTransactions._Set(index, temp)
   372  	}
   373  
   374  	return transaction
   375  }
