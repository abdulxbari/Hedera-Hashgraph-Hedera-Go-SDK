   193  func (transaction *FreezeTransaction) SignWithOperator(
   194  	client *Client,
   195  ) (*FreezeTransaction, error) {
   196  	// If the transaction is not signed by the _Operator, we need
   197  	// to sign the transaction with the _Operator
   198  
   199  	if client == nil {
   200  		return nil, errNoClientProvided
   201  	} else if client.operator == nil {
   202  		return nil, errClientOperatorSigning
   203  	}
   204  
   205  	if !transaction.IsFrozen() {
   206  		_, err := transaction.FreezeWith(client)
   207  		if err != nil {
   208  			return transaction, err
   209  		}
   210  	}
   211  	return transaction.SignWith(client.operator.publicKey, client.operator.signer), nil
   212  }

   228  func (transaction *FreezeTransaction) Execute(
   229  	client *Client,
   230  ) (TransactionResponse, error) {
   231  	if client == nil {
   232  		return TransactionResponse{}, errNoClientProvided
   233  	}
   234  
   235  	if !transaction.IsFrozen() {
   236  		_, err := transaction.FreezeWith(client)
   237  		if err != nil {
   238  			return TransactionResponse{}, err
   239  		}
   240  	}
   241  
   242  	if transaction.freezeError != nil {
   243  		return TransactionResponse{}, transaction.freezeError
   244  	}
   245  
   246  	transactionID := transaction.transactionIDs._GetCurrent().(TransactionID)
   247  
   248  	if !client.GetOperatorAccountID()._IsZero() && client.GetOperatorAccountID()._Equals(*transactionID.AccountID) {
   249  		transaction.SignWith(
   250  			client.GetOperatorPublicKey(),
   251  			client.operator.signer,
   252  		)
   253  	}
   254  
   255  	resp, err := _Execute(
   256  		client,
   257  		&transaction.Transaction,
   258  		_TransactionShouldRetry,
   259  		_TransactionMakeRequest,
   260  		_TransactionAdvanceRequest,
   261  		_TransactionGetNodeAccountID,
   262  		_FreezeTransactionGetMethod,
   263  		_TransactionMapStatusError,
   264  		_TransactionMapResponse,
   265  		transaction._GetLogID(),
   266  		transaction.grpcDeadline,
   267  		transaction.maxBackoff,
   268  		transaction.minBackoff,
   269  		transaction.maxRetry,
   270  	)
   271  
   272  	if err != nil {
   273  		return TransactionResponse{
   274  			TransactionID:  transaction.GetTransactionID(),
   275  			NodeID:         resp.(TransactionResponse).NodeID,
   276  			ValidateStatus: true,
   277  		}, err
   278  	}
   279  
   280  	return TransactionResponse{
   281  		TransactionID:  transaction.GetTransactionID(),
   282  		NodeID:         resp.(TransactionResponse).NodeID,
   283  		Hash:           resp.(TransactionResponse).Hash,
   284  		ValidateStatus: true,
   285  	}, nil
   286  }

   374  func (transaction *FreezeTransaction) AddSignature(publicKey PublicKey, signature []byte) *FreezeTransaction {
   375  	transaction._RequireOneNodeAccountID()
   376  
   377  	if transaction._KeyAlreadySigned(publicKey) {
   378  		return transaction
   379  	}
   380  
   381  	if transaction.signedTransactions._Length() == 0 {
   382  		return transaction
   383  	}
   384  
   385  	transaction.transactions = _NewLockableSlice()
   386  	transaction.publicKeys = append(transaction.publicKeys, publicKey)
   387  	transaction.transactionSigners = append(transaction.transactionSigners, nil)
   388  	transaction.transactionIDs.locked = true
   389  
   390  	for index := 0; index < transaction.signedTransactions._Length(); index++ {
   391  		var temp *services.SignedTransaction
   392  		switch t := transaction.signedTransactions._Get(index).(type) { //nolint
   393  		case *services.SignedTransaction:
   394  			temp = t
   395  		}
   396  		temp.SigMap.SigPair = append(
   397  			temp.SigMap.SigPair,
   398  			publicKey._ToSignaturePairProtobuf(signature),
   399  		)
   400  		transaction.signedTransactions._Set(index, temp)
   401  	}
   402  
   403  	return transaction
   404  }
