   300  func (transaction *FileUpdateTransaction) Execute(
   301  	client *Client,
   302  ) (TransactionResponse, error) {
   303  	if client == nil {
   304  		return TransactionResponse{}, errNoClientProvided
   305  	}
   306  
   307  	if !transaction.IsFrozen() {
   308  		_, err := transaction.FreezeWith(client)
   309  		if err != nil {
   310  			return TransactionResponse{}, err
   311  		}
   312  	}
   313  
   314  	if transaction.freezeError != nil {
   315  		return TransactionResponse{}, transaction.freezeError
   316  	}
   317  
   318  	transactionID := transaction.transactionIDs._GetCurrent().(TransactionID)
   319  
   320  	if !client.GetOperatorAccountID()._IsZero() && client.GetOperatorAccountID()._Equals(*transactionID.AccountID) {
   321  		transaction.SignWith(
   322  			client.GetOperatorPublicKey(),
   323  			client.operator.signer,
   324  		)
   325  	}
   326  
   327  	resp, err := _Execute(
   328  		client,
   329  		&transaction.Transaction,
   330  		_TransactionShouldRetry,
   331  		_TransactionMakeRequest,
   332  		_TransactionAdvanceRequest,
   333  		_TransactionGetNodeAccountID,
   334  		_FileUpdateTransactionGetMethod,
   335  		_TransactionMapStatusError,
   336  		_TransactionMapResponse,
   337  		transaction._GetLogID(),
   338  		transaction.grpcDeadline,
   339  		transaction.maxBackoff,
   340  		transaction.minBackoff,
   341  		transaction.maxRetry,
   342  	)
   343  
   344  	if err != nil {
   345  		return TransactionResponse{
   346  			TransactionID:  transaction.GetTransactionID(),
   347  			NodeID:         resp.(TransactionResponse).NodeID,
   348  			ValidateStatus: true,
   349  		}, err
   350  	}
   351  
   352  	return TransactionResponse{
   353  		TransactionID:  transaction.GetTransactionID(),

   450  func (transaction *FileUpdateTransaction) AddSignature(publicKey PublicKey, signature []byte) *FileUpdateTransaction {
   451  	transaction._RequireOneNodeAccountID()
   452  
   453  	if transaction._KeyAlreadySigned(publicKey) {
   454  		return transaction
   455  	}
   456  
   457  	if transaction.signedTransactions._Length() == 0 {
   458  		return transaction
   459  	}
   460  
   461  	transaction.transactions = _NewLockableSlice()
   462  	transaction.publicKeys = append(transaction.publicKeys, publicKey)
   463  	transaction.transactionSigners = append(transaction.transactionSigners, nil)
   464  	transaction.transactionIDs.locked = true
   465  
   466  	for index := 0; index < transaction.signedTransactions._Length(); index++ {
   467  		var temp *services.SignedTransaction
   468  		switch t := transaction.signedTransactions._Get(index).(type) { //nolint
   469  		case *services.SignedTransaction:
   470  			temp = t
   471  		}
   472  		temp.SigMap.SigPair = append(
   473  			temp.SigMap.SigPair,
   474  			publicKey._ToSignaturePairProtobuf(signature),
   475  		)
   476  		transaction.signedTransactions._Set(index, temp)
   477  	}
   478  
   479  	return transaction
   480  }
