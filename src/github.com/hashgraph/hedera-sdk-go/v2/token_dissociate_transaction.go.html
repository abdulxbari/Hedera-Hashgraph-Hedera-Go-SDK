    30  // TokenDissociateTransaction
    31  // Dissociates the provided account with the provided tokens. Must be signed by the provided Account's key.
    32  // If the provided account is not found, the transaction will resolve to INVALID_ACCOUNT_ID.
    33  // If the provided account has been deleted, the transaction will resolve to ACCOUNT_DELETED.
    34  // If any of the provided tokens is not found, the transaction will resolve to INVALID_TOKEN_REF.
    35  // If any of the provided tokens has been deleted, the transaction will resolve to TOKEN_WAS_DELETED.
    36  // If an association between the provided account and any of the tokens does not exist, the
    37  // transaction will resolve to TOKEN_NOT_ASSOCIATED_TO_ACCOUNT.
    38  // If a token has not been deleted and has not expired, and the user has a nonzero balance, the
    39  // transaction will resolve to TRANSACTION_REQUIRES_ZERO_TOKEN_BALANCES.
    40  // If a <b>fungible token</b> has expired, the user can disassociate even if their token balance is
    41  // not zero.
    42  // If a <b>non fungible token</b> has expired, the user can <b>not</b> disassociate if their token
    43  // balance is not zero. The transaction will resolve to TRANSACTION_REQUIRED_ZERO_TOKEN_BALANCES.
    44  // On success, associations between the provided account and tokens are removed.
    45  type TokenDissociateTransaction struct {
    46  	Transaction
    47  	accountID *AccountID
    48  	tokens    []TokenID
    49  }

   270  // Execute executes the Transaction with the provided client
   271  func (transaction *TokenDissociateTransaction) Execute(
   272  	client *Client,
   273  ) (TransactionResponse, error) {
   274  	if client == nil {
   275  		return TransactionResponse{}, errNoClientProvided
   276  	}
   277  
   278  	if transaction.freezeError != nil {
   279  		return TransactionResponse{}, transaction.freezeError
   280  	}
   281  
   282  	if !transaction.IsFrozen() {
   283  		_, err := transaction.FreezeWith(client)
   284  		if err != nil {
   285  			return TransactionResponse{}, err
   286  		}
   287  	}
   288  
   289  	transactionID := transaction.transactionIDs._GetCurrent().(TransactionID)
   290  
   291  	if !client.GetOperatorAccountID()._IsZero() && client.GetOperatorAccountID()._Equals(*transactionID.AccountID) {
   292  		transaction.SignWith(
   293  			client.GetOperatorPublicKey(),
   294  			client.operator.signer,
   295  		)
   296  	}
   297  
   298  	resp, err := _Execute(
   299  		client,
   300  		&transaction.Transaction,
   301  		_TransactionShouldRetry,
   302  		_TransactionMakeRequest,
   303  		_TransactionAdvanceRequest,
   304  		_TransactionGetNodeAccountID,
   305  		_TokenDissociateTransactionGetMethod,
   306  		_TransactionMapStatusError,
   307  		_TransactionMapResponse,
   308  		transaction._GetLogID(),
   309  		transaction.grpcDeadline,
   310  		transaction.maxBackoff,
   311  		transaction.minBackoff,
   312  		transaction.maxRetry,
   313  	)
   314  
   315  	if err != nil {
   316  		return TransactionResponse{
   317  			TransactionID:  transaction.GetTransactionID(),
   318  			NodeID:         resp.(TransactionResponse).NodeID,
   319  			ValidateStatus: true,
   320  		}, err
   321  	}
   322  
   323  	return TransactionResponse{
   324  		TransactionID:  transaction.GetTransactionID(),
   325  		NodeID:         resp.(TransactionResponse).NodeID,
   326  		Hash:           resp.(TransactionResponse).Hash,
   327  		ValidateStatus: true,
   328  	}, nil
   329  }

   421  func (transaction *TokenDissociateTransaction) AddSignature(publicKey PublicKey, signature []byte) *TokenDissociateTransaction {
   422  	transaction._RequireOneNodeAccountID()
   423  
   424  	if transaction._KeyAlreadySigned(publicKey) {
   425  		return transaction
   426  	}
   427  
   428  	if transaction.signedTransactions._Length() == 0 {
   429  		return transaction
   430  	}
   431  
   432  	transaction.transactions = _NewLockableSlice()
   433  	transaction.publicKeys = append(transaction.publicKeys, publicKey)
   434  	transaction.transactionSigners = append(transaction.transactionSigners, nil)
   435  	transaction.transactionIDs.locked = true
   436  
   437  	for index := 0; index < transaction.signedTransactions._Length(); index++ {
   438  		var temp *services.SignedTransaction
   439  		switch t := transaction.signedTransactions._Get(index).(type) { //nolint
   440  		case *services.SignedTransaction:
   441  			temp = t
   442  		}
   443  		temp.SigMap.SigPair = append(
   444  			temp.SigMap.SigPair,
   445  			publicKey._ToSignaturePairProtobuf(signature),
   446  		)
   447  		transaction.signedTransactions._Set(index, temp)
   448  	}
   449  
   450  	return transaction
   451  }
