   134  func TestUnitFileInfoQueryGet(t *testing.T) {
   135  	checksum := "dmqui"
   136  	fileID := FileID{File: 3, checksum: &checksum}
   137  	deadline := time.Duration(time.Minute)
   138  	accountId := AccountID{Account: 123}
   139  	transactionID := TransactionIDGenerate(accountId)
   140  	query := NewFileInfoQuery().
   141  		SetFileID(fileID).
   142  		SetQueryPayment(NewHbar(2)).
   143  		SetMaxQueryPayment(NewHbar(1)).
   144  		SetQueryPayment(HbarFromTinybar(25)).
   145  		SetNodeAccountIDs([]AccountID{{Account: 10}, {Account: 11}, {Account: 12}}).
   146  		SetGrpcDeadline(&deadline).
   147  		SetMaxBackoff(1 * time.Minute).
   148  		SetMinBackoff(500 * time.Millisecond).
   149  		SetMaxRetry(5).
   150  		SetPaymentTransactionID(transactionID).
   151  		SetMaxQueryPayment(NewHbar(500))
   152  	client := ClientForTestnet()
   153  	client.SetAutoValidateChecksums(true)
   154  	err := query._ValidateNetworkOnIDs(client)
   155  	require.NoError(t, err)
   156  	require.Equal(t, fileID, query.GetFileID())
   157  	require.Equal(t, []AccountID{{Account: 10}, {Account: 11}, {Account: 12}}, query.GetNodeAccountIDs())
   158  	require.Equal(t, 500*time.Millisecond, query.GetMinBackoff())
   159  	require.Equal(t, 1*time.Minute, query.GetMaxBackoff())
   160  	require.Equal(t, 5, query.GetMaxRetryCount())
   161  	require.Equal(t, transactionID, query.GetPaymentTransactionID())
   162  	require.Equal(t, HbarFromTinybar(25), query.GetQueryPayment())
   163  	require.Equal(t, NewHbar(500), query.GetMaxQueryPayment())
   164  	require.Equal(t, &deadline, query.GetGrpcDeadline())
   165  	require.Equal(t, fmt.Sprintf("FileInfoQuery:%v", transactionID.ValidStart.UnixNano()), query._GetLogID())
   166  }
