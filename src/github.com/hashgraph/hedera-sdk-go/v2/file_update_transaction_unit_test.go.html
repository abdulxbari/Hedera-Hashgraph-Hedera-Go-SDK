    69  func TestUnitFileUpdateTransactionMock(t *testing.T) {
    70  	newKey, err := PrivateKeyFromStringEd25519("302e020100300506032b657004220420a869f4c6191b9c8c99933e7f6b6611711737e4b1a1a5a4cb5370e719a1f6df98")
    71  	require.NoError(t, err)
    72  
    73  	call := func(request *services.Transaction) *services.TransactionResponse {
    74  		require.NotEmpty(t, request.SignedTransactionBytes)
    75  		signedTransaction := services.SignedTransaction{}
    76  		_ = protobuf.Unmarshal(request.SignedTransactionBytes, &signedTransaction)
    77  
    78  		require.NotEmpty(t, signedTransaction.BodyBytes)
    79  		transactionBody := services.TransactionBody{}
    80  		_ = protobuf.Unmarshal(signedTransaction.BodyBytes, &transactionBody)
    81  
    82  		require.NotNil(t, transactionBody.TransactionID)
    83  		transactionId := transactionBody.TransactionID.String()
    84  		require.NotEqual(t, "", transactionId)
    85  
    86  		sigMap := signedTransaction.GetSigMap()
    87  		require.NotNil(t, sigMap)
    88  
    89  		key, _ := PrivateKeyFromStringEd25519("302e020100300506032b657004220420d45e1557156908c967804615af59a000be88c7aa7058bfcbe0f46b16c28f887d")
    90  		for _, sigPair := range sigMap.SigPair {
    91  			verified := false
    92  
    93  			switch k := sigPair.Signature.(type) {
    94  			case *services.SignaturePair_Ed25519:
    95  				pbTemp, _ := PublicKeyFromBytesEd25519(sigPair.PubKeyPrefix)
    96  				verified = pbTemp.Verify(signedTransaction.BodyBytes, k.Ed25519)
    97  			case *services.SignaturePair_ECDSASecp256K1:
    98  				pbTemp, _ := PublicKeyFromBytesECDSA(sigPair.PubKeyPrefix)
    99  				verified = pbTemp.Verify(signedTransaction.BodyBytes, k.ECDSASecp256K1)
   100  			}
   101  			require.True(t, verified)
   102  		}
   103  
   104  		require.Equal(t, bytes.Compare(sigMap.SigPair[1].PubKeyPrefix, key.PublicKey().BytesRaw()), 0)
   105  		require.Equal(t, bytes.Compare(sigMap.SigPair[0].PubKeyPrefix, newKey.PublicKey().BytesRaw()), 0)
   106  
   107  		if bod, ok := transactionBody.Data.(*services.TransactionBody_FileUpdate); ok {
   108  			require.Equal(t, bod.FileUpdate.FileID.FileNum, int64(3))
   109  			require.Equal(t, bytes.Compare(bod.FileUpdate.Contents, []byte{123}), 0)
   110  			require.Equal(t, bod.FileUpdate.Memo, &wrapperspb.StringValue{Value: "no memo"})
   111  		}
   112  
   113  		return &services.TransactionResponse{
   114  			NodeTransactionPrecheckCode: services.ResponseCodeEnum_OK,
   115  		}
   116  	}
   117  	responses := [][]interface{}{{
   118  		call,
   119  	}}
   120  
   121  	client, server := NewMockClientAndServer(responses)
   122  	defer server.Close()
   123  
   124  	freez, err := NewFileUpdateTransaction().
   125  		SetFileID(FileID{File: 3}).
   126  		SetNodeAccountIDs([]AccountID{{Account: 3}}).
   127  		SetFileMemo("no memo").
   128  		SetKeys(newKey).
   129  		SetContents([]byte{123}).
   130  		FreezeWith(client)
   131  	require.NoError(t, err)
   132  
   133  	_, err = freez.Sign(newKey).Execute(client)
   134  	require.NoError(t, err)
   135  }
