    65  	newNodeKeys := map[string]bool{}

    67

    73  		node := newNodes[index]
    74  
    75  		if err := node._Close(); err != nil {
    76  			return err
    77  		}
    78  
    79  		if index == len(newNodes)-1 {
    80  			newNodes = newNodes[:index]
    81  		} else {
    82  			newNodes = append(newNodes[:index], newNodes[index+1:]...)
    83  		}
    84  	}
    85  
    86  	for _, node := range newNodes {
    87  		newNodeKeys[node._GetKey()] = true
    88  		newNodeValues[node._GetAddress()] = true
    89  	}
    90  
    91  	for key, value := range network {
    92  		_, keyOk := newNodeKeys[key]
    93  		_, valueOk := newNodeValues[value._GetAddress()]
    94  
    95  		if keyOk && valueOk {
    96  			continue
    97  		}
    98  
    99  		newNodes = append(newNodes, value)
   100  	}

   102  	newNetwork, newHealthyNodes := _CreateNetworkFromNodes(newNodes)
   103

   111  func (this *_ManagedNetwork) _ReadmitNodes() {
   112  	now := time.Now()
   113  	nextEarliestReadmitTime := time.Now().Add(this.maxNodeReadmitPeriod)
   114  
   115  	for _, node := range this.nodes {
   116  		if node._GetReadmitTime() != nil && node._GetReadmitTime().After(now) && node._GetReadmitTime().Before(nextEarliestReadmitTime) {
   117  			nextEarliestReadmitTime = *node._GetReadmitTime()
   118  		}
   119  	}
   120  
   121  outer:
   122  	for _, node := range this.nodes {
   123  		for _, healthyNode := range this.healthyNodes {
   124  			if node == healthyNode {
   125  				continue outer
   126  			}
   127  		}
   128  
   129  		if node._IsHealthy() {
   130  			this.healthyNodes = append(this.healthyNodes, node)
   131  		}
   132  	}
   133  }

   228  func _CreateNetworkFromNodes(nodes []_IManagedNode) (network map[string][]_IManagedNode, healthyNodes []_IManagedNode) {
   229  	healthyNodes = []_IManagedNode{}
   230  	network = map[string][]_IManagedNode{}
   231  
   232  	for _, node := range nodes {
   233  		if node._IsHealthy() {
   234  			healthyNodes = append(healthyNodes, node)
   235  		}
   236  
   237  		value, ok := network[node._GetKey()]
   238  		if !ok {
   239  			value = []_IManagedNode{}
   240  		}
   241  		value = append(value, node)
   242  		network[node._GetKey()] = value
   243  	}
   244  
   245  	return network, healthyNodes
   246  }

   277  func _GetNodesToRemove(network map[string]_IManagedNode, nodes []_IManagedNode) []int {
