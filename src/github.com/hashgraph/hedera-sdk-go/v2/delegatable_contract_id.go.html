    67  func (id *DelegatableContractID) ValidateChecksum(client *Client) error {
    68  	if !id._IsZero() && client != nil && client.network.ledgerID != nil {
    69  		var tempChecksum _ParseAddressResult
    70  		var err error
    71  		if client.network.ledgerID != nil {
    72  			tempChecksum, err = _ChecksumParseAddress(client.GetLedgerID(), fmt.Sprintf("%d.%d.%d", id.Shard, id.Realm, id.Contract))
    73  		}
    74  		if err != nil {
    75  			return err
    76  		}
    77  		err = _ChecksumVerify(tempChecksum.status)
    78  		if err != nil {
    79  			return err
    80  		}
    81  		if id.checksum == nil {
    82  			return errChecksumMissing
    83  		}
    84  		if tempChecksum.correctChecksum != *id.checksum {
    85  			temp, _ := client.network.ledgerID.ToNetworkName()
    86  			return errors.New(fmt.Sprintf("network mismatch or wrong checksum given, given checksum: %s, correct checksum %s, network: %s",
    87  				*id.checksum,
    88  				tempChecksum.correctChecksum,
    89  				temp))
    90  		}
    91  	}
    92  
    93  	return nil
    94  }

   134  func (id DelegatableContractID) ToStringWithChecksum(client Client) (string, error) {
   135  	if id.EvmAddress != nil {
   136  		return "", errors.New("EvmAddress doesn't support checksums")
   137  	}
   138  	if client.GetNetworkName() == nil && client.GetLedgerID() == nil {
   139  		return "", errNetworkNameMissing
   140  	}
   141  	var checksum _ParseAddressResult
   142  	var err error
   143  	if client.network.ledgerID != nil {
   144  		checksum, err = _ChecksumParseAddress(client.GetLedgerID(), fmt.Sprintf("%d.%d.%d", id.Shard, id.Realm, id.Contract))
   145  	}
   146  	if err != nil {
   147  		return "", err
   148  	}
   149  
   150  	return fmt.Sprintf("%d.%d.%d-%s", id.Shard, id.Realm, id.Contract, checksum.correctChecksum), nil
   151  }

   174  func _DelegatableContractIDFromProtobuf(contractID *services.ContractID) *DelegatableContractID {
   175  	if contractID == nil {
   176  		return nil
   177  	}
   178  	resultID := DelegatableContractID{
   179  		Shard: uint64(contractID.ShardNum),
   180  		Realm: uint64(contractID.RealmNum),

   182  
   183  	switch id := contractID.Contract.(type) {
   184  	case *services.ContractID_ContractNum:

   189  		resultID.EvmAddress = id.EvmAddress
   190  		resultID.Contract = 0
   191  		return &resultID
   192  	default:
   193  		return &resultID
   194  	}
   195  }
