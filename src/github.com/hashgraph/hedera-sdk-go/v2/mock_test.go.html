   799  func NewServiceDescription(handler func(interface{}, context.Context, func(interface{}) error, grpc.UnaryServerInterceptor) (interface{}, error), service *grpc.ServiceDesc) *grpc.ServiceDesc {
   800  	var methods []grpc.MethodDesc
   801  	for _, desc := range service.Methods {
   802  		methods = append(methods, grpc.MethodDesc{
   803  			MethodName: desc.MethodName,
   804  			Handler:    handler,
   805  		})
   806  	}
   807  
   808  	return &grpc.ServiceDesc{
   809  		ServiceName: service.ServiceName,
   810  		HandlerType: service.HandlerType,
   811  		Methods:     methods,
   812  		Streams:     []grpc.StreamDesc{},
   813  		Metadata:    service.Metadata,
   814  	}
   815  }
