   205  func TestUnitContractInfoQueryCoverage(t *testing.T) {
   206  	checksum := "dmqui"
   207  	deadline := time.Second * 3
   208  	contract := ContractID{Contract: 3, checksum: &checksum}
   209  	nodeAccountID := []AccountID{{Account: 10}}
   210  	transactionID := TransactionIDGenerate(AccountID{Account: 324})
   211  
   212  	client := ClientForTestnet()
   213  	client.SetAutoValidateChecksums(true)
   214  
   215  	query := NewContractInfoQuery().
   216  		SetMaxRetry(3).
   217  		SetMaxBackoff(time.Second * 30).
   218  		SetMinBackoff(time.Second * 10).
   219  		SetContractID(contract).
   220  		SetNodeAccountIDs(nodeAccountID).
   221  		SetPaymentTransactionID(transactionID).
   222  		SetMaxQueryPayment(NewHbar(23)).
   223  		SetQueryPayment(NewHbar(3)).
   224  		SetGrpcDeadline(&deadline)
   225  
   226  	err := query._ValidateNetworkOnIDs(client)
   227  	require.NoError(t, err)
   228  
   229  	require.Equal(t, nodeAccountID, query.GetNodeAccountIDs())
   230  	require.Equal(t, time.Second*30, query.GetMaxBackoff())
   231  	require.Equal(t, time.Second*10, query.GetMinBackoff())
   232  	require.Equal(t, fmt.Sprintf("ContractInfoQuery:%v", transactionID.ValidStart.UnixNano()), query._GetLogID())
   233  	require.Equal(t, contract, query.GetContractID())
   234  	require.Equal(t, &deadline, query.GetGrpcDeadline())
   235  }
