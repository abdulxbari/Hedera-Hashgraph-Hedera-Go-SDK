   197  func _QueryMakePaymentTransaction(transactionID TransactionID, nodeAccountID AccountID, operator *_Operator, cost Hbar) (*services.Transaction, error) {
   198  	accountAmounts := make([]*services.AccountAmount, 0)
   199  	accountAmounts = append(accountAmounts, &services.AccountAmount{
   200  		AccountID: nodeAccountID._ToProtobuf(),
   201  		Amount:    cost.tinybar,
   202  	})
   203  	accountAmounts = append(accountAmounts, &services.AccountAmount{
   204  		AccountID: operator.accountID._ToProtobuf(),
   205  		Amount:    -cost.tinybar,
   206  	})
   207  
   208  	body := services.TransactionBody{
   209  		TransactionID:  transactionID._ToProtobuf(),
   210  		NodeAccountID:  nodeAccountID._ToProtobuf(),
   211  		TransactionFee: uint64(NewHbar(1).tinybar),
   212  		TransactionValidDuration: &services.Duration{
   213  			Seconds: 120,
   214  		},
   215  		Data: &services.TransactionBody_CryptoTransfer{
   216  			CryptoTransfer: &services.CryptoTransferTransactionBody{
   217  				Transfers: &services.TransferList{
   218  					AccountAmounts: accountAmounts,
   219  				},
   220  			},
   221  		},
   222  	}
   223  
   224  	bodyBytes, err := protobuf.Marshal(&body)
   225  	if err != nil {
   226  		return nil, errors.Wrap(err, "error serializing query body")
   227  	}
   228  
   229  	signature := operator.signer(bodyBytes)
   230  	sigPairs := make([]*services.SignaturePair, 0)
   231  	sigPairs = append(sigPairs, operator.publicKey._ToSignaturePairProtobuf(signature))
   232  
   233  	return &services.Transaction{
   234  		BodyBytes: bodyBytes,
   235  		SigMap: &services.SignatureMap{
   236  			SigPair: sigPairs,
   237  		},
   238  	}, nil
   239  }
