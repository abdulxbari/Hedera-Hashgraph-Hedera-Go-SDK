   118  func (transaction *PrngTransaction) SignWithOperator(
   119  	client *Client,
   120  ) (*PrngTransaction, error) {
   121  	// If the transaction is not signed by the _Operator, we need
   122  	// to sign the transaction with the _Operator
   123  
   124  	if client == nil {
   125  		return nil, errNoClientProvided
   126  	} else if client.operator == nil {
   127  		return nil, errClientOperatorSigning
   128  	}
   129  	if !transaction.IsFrozen() {
   130  		_, err := transaction.FreezeWith(client)
   131  		if err != nil {
   132  			return transaction, err
   133  		}
   134  	}
   135  	return transaction.SignWith(client.operator.publicKey, client.operator.signer), nil
   136  }

   151  // Execute executes the Transaction with the provided client
   152  func (transaction *PrngTransaction) Execute(
   153  	client *Client,
   154  ) (TransactionResponse, error) {
   155  	if client == nil {
   156  		return TransactionResponse{}, errNoClientProvided
   157  	}
   158  
   159  	if transaction.freezeError != nil {
   160  		return TransactionResponse{}, transaction.freezeError
   161  	}
   162  
   163  	if !transaction.IsFrozen() {
   164  		_, err := transaction.FreezeWith(client)
   165  		if err != nil {
   166  			return TransactionResponse{}, err
   167  		}
   168  	}
   169  
   170  	transactionID := transaction.transactionIDs._GetCurrent().(TransactionID)
   171  
   172  	if !client.GetOperatorAccountID()._IsZero() && client.GetOperatorAccountID()._Equals(*transactionID.AccountID) {
   173  		transaction.SignWith(
   174  			client.GetOperatorPublicKey(),
   175  			client.operator.signer,
   176  		)
   177  	}
   178  
   179  	resp, err := _Execute(
   180  		client,
   181  		&transaction.Transaction,
   182  		_TransactionShouldRetry,
   183  		_TransactionMakeRequest,
   184  		_TransactionAdvanceRequest,
   185  		_TransactionGetNodeAccountID,
   186  		_RngTransactionGetMethod,
   187  		_TransactionMapStatusError,
   188  		_TransactionMapResponse,
   189  		transaction._GetLogID(),
   190  		transaction.grpcDeadline,
   191  		transaction.maxBackoff,
   192  		transaction.minBackoff,
   193  		transaction.maxRetry,
   194  	)
   195  
   196  	if err != nil {
   197  		return TransactionResponse{
   198  			TransactionID:  transaction.GetTransactionID(),
   199  			NodeID:         resp.(TransactionResponse).NodeID,
   200  			ValidateStatus: true,
   201  		}, err
   202  	}
   203  
   204  	return TransactionResponse{
   205  		TransactionID:          transaction.GetTransactionID(),
   206  		NodeID:                 resp.(TransactionResponse).NodeID,
   207  		Hash:                   resp.(TransactionResponse).Hash,
   208  		ScheduledTransactionId: transaction.GetTransactionID(),
   209  	}, nil
   210  }

   216  func (transaction *PrngTransaction) FreezeWith(client *Client) (*PrngTransaction, error) {
   217  	if transaction.IsFrozen() {
   218  		return transaction, nil
   219  	}
   220  	transaction._InitFee(client)
   221  	if err := transaction._InitTransactionID(client); err != nil {
   222  		return transaction, err
   223  	}
   224  	body := transaction._Build()
   225  
   226  	return transaction, _TransactionFreezeWith(&transaction.Transaction, client, body)
   227  }
