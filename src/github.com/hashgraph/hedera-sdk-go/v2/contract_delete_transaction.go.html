   285  	client *Client,
   286  ) (TransactionResponse, error) {
   287  	if client == nil {
   288  		return TransactionResponse{}, errNoClientProvided
   289  	}
   290  
   291  	if transaction.freezeError != nil {
   292  		return TransactionResponse{}, transaction.freezeError

   305  		transaction.SignWith(
   306  			client.GetOperatorPublicKey(),
   307  			client.operator.signer,

   311  	resp, err := _Execute(

   434  func (transaction *ContractDeleteTransaction) AddSignature(publicKey PublicKey, signature []byte) *ContractDeleteTransaction {
   435  	transaction._RequireOneNodeAccountID()
   436  
   437  	if transaction._KeyAlreadySigned(publicKey) {
   438  		return transaction
   439  	}
   440  
   441  	if transaction.signedTransactions._Length() == 0 {
   442  		return transaction
   443  	}
   444  
   445  	transaction.transactions = _NewLockableSlice()
   446  	transaction.publicKeys = append(transaction.publicKeys, publicKey)
   447  	transaction.transactionSigners = append(transaction.transactionSigners, nil)
   448  	transaction.transactionIDs.locked = true
   449  
   450  	for index := 0; index < transaction.signedTransactions._Length(); index++ {
   451  		var temp *services.SignedTransaction
   452  		switch t := transaction.signedTransactions._Get(index).(type) { //nolint
   453  		case *services.SignedTransaction:
   454  			temp = t
   455  		}
   456  		temp.SigMap.SigPair = append(
   457  			temp.SigMap.SigPair,
   458  			publicKey._ToSignaturePairProtobuf(signature),
   459  		)
   460  		transaction.signedTransactions._Set(index, temp)
   461  	}
   462  
   463  	return transaction
   464  }
