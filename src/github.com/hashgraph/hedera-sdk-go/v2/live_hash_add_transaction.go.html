    32  // LiveHashAddTransaction At consensus, attaches the given livehash to the given account.  The hash can be deleted by the
    33  // key controlling the account, or by any of the keys associated to the livehash.  Hence livehashes
    34  // provide a revocation service for their implied credentials; for example, when an authority grants
    35  // a credential to the account, the account owner will cosign with the authority (or authorities) to
    36  // attach a hash of the credential to the account---hence proving the grant. If the credential is
    37  // revoked, then any of the authorities may delete it (or the account owner). In this way, the
    38  // livehash mechanism acts as a revocation service.  An account cannot have two identical livehashes
    39  // associated. To modify the list of keys in a livehash, the livehash should first be deleted, then
    40  // recreated with a new list of keys.

    42  	Transaction
    43  	accountID *AccountID
    44  	hash      []byte
    45  	keys      *KeyList
    46  	duration  *time.Duration

   401  func (transaction *LiveHashAddTransaction) AddSignature(publicKey PublicKey, signature []byte) *LiveHashAddTransaction {
   402  	transaction._RequireOneNodeAccountID()
   403  
   404  	if transaction._KeyAlreadySigned(publicKey) {
   405  		return transaction
   406  	}
   407  
   408  	if transaction.signedTransactions._Length() == 0 {
   409  		return transaction
   410  	}
   411  
   412  	transaction.transactions = _NewLockableSlice()
   413  	transaction.publicKeys = append(transaction.publicKeys, publicKey)
   414  	transaction.transactionSigners = append(transaction.transactionSigners, nil)
   415  	transaction.transactionIDs.locked = true
   416  
   417  	for index := 0; index < transaction.signedTransactions._Length(); index++ {
   418  		var temp *services.SignedTransaction
   419  		switch t := transaction.signedTransactions._Get(index).(type) { //nolint
   420  		case *services.SignedTransaction:
   421  			temp = t
   422  		}
   423  		temp.SigMap.SigPair = append(
   424  			temp.SigMap.SigPair,
   425  			publicKey._ToSignaturePairProtobuf(signature),
   426  		)
   427  		transaction.signedTransactions._Set(index, temp)
   428  	}
   429  
   430  	return transaction
   431  }
