    35  func TestIntegrationTokenUnfreezeTransactionCanExecute(t *testing.T) {
    36  	env := NewIntegrationTestEnv(t)
    37  
    38  	newKey, err := PrivateKeyGenerateEd25519()
    39  	require.NoError(t, err)
    40  
    41  	newBalance := NewHbar(2)
    42  
    43  	assert.Equal(t, 2*HbarUnits.Hbar._NumberOfTinybar(), newBalance.tinybar)
    44  
    45  	resp, err := NewAccountCreateTransaction().
    46  		SetNodeAccountIDs(env.NodeAccountIDs).
    47  		SetKey(newKey.PublicKey()).
    48  		SetInitialBalance(newBalance).
    49  		Execute(env.Client)
    50  	require.NoError(t, err)
    51  
    52  	receipt, err := resp.SetValidateStatus(true).GetReceipt(env.Client)
    53  	require.NoError(t, err)
    54  
    55  	accountID := *receipt.AccountID
    56  
    57  	resp, err = NewTokenCreateTransaction().
    58  		SetNodeAccountIDs(env.NodeAccountIDs).
    59  		SetTokenName("ffff").
    60  		SetTokenSymbol("F").
    61  		SetDecimals(3).
    62  		SetInitialSupply(1000000).
    63  		SetTreasuryAccountID(env.Client.GetOperatorAccountID()).
    64  		SetAdminKey(env.Client.GetOperatorPublicKey()).
    65  		SetFreezeKey(env.Client.GetOperatorPublicKey()).
    66  		SetWipeKey(env.Client.GetOperatorPublicKey()).
    67  		SetKycKey(env.Client.GetOperatorPublicKey()).
    68  		SetSupplyKey(env.Client.GetOperatorPublicKey()).
    69  		SetFreezeDefault(false).
    70  		Execute(env.Client)
    71  	require.NoError(t, err)
    72  
    73  	receipt, err = resp.SetValidateStatus(true).GetReceipt(env.Client)
    74  	require.NoError(t, err)
    75  
    76  	tokenID := *receipt.TokenID
    77  
    78  	transaction, err := NewTokenAssociateTransaction().
    79  		SetNodeAccountIDs([]AccountID{resp.NodeID}).
    80  		SetAccountID(accountID).
    81  		SetTokenIDs(tokenID).
    82  		FreezeWith(env.Client)
    83  	require.NoError(t, err)
    84  
    85  	resp, err = transaction.
    86  		Sign(newKey).
    87  		Execute(env.Client)
    88  	require.NoError(t, err)
    89  
    90  	_, err = resp.SetValidateStatus(true).GetReceipt(env.Client)
    91  	require.NoError(t, err)
    92  
    93  	resp, err = NewTokenFreezeTransaction().
    94  		SetNodeAccountIDs([]AccountID{resp.NodeID}).
    95  		SetAccountID(accountID).
    96  		SetTokenID(tokenID).
    97  		Execute(env.Client)
    98  	require.NoError(t, err)
    99  
   100  	_, err = resp.SetValidateStatus(true).GetReceipt(env.Client)
   101  	require.NoError(t, err)
   102  
   103  	resp, err = NewTokenUnfreezeTransaction().
   104  		SetNodeAccountIDs([]AccountID{resp.NodeID}).
   105  		SetAccountID(accountID).
   106  		SetTokenID(tokenID).
   107  		Execute(env.Client)
   108  	require.NoError(t, err)
   109  
   110  	_, err = resp.SetValidateStatus(true).GetReceipt(env.Client)
   111  	require.NoError(t, err)
   112  
   113  	tx, err := NewAccountDeleteTransaction().
   114  		SetAccountID(accountID).
   115  		SetTransferAccountID(env.Client.GetOperatorAccountID()).
   116  		FreezeWith(env.Client)
   117  	require.NoError(t, err)
   118  
   119  	resp, err = tx.
   120  		Sign(newKey).
   121  		Execute(env.Client)
   122  	require.NoError(t, err)
   123  
   124  	_, err = resp.SetValidateStatus(true).GetReceipt(env.Client)
   125  	require.NoError(t, err)
   126  
   127  	err = CloseIntegrationTestEnv(env, &tokenID)
   128  	require.NoError(t, err)
   129  }
