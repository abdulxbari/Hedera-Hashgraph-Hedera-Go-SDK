   485  // Execute executes the Transaction with the provided client
   486  func (transaction *ContractCreateTransaction) Execute(
   487  	client *Client,
   488  ) (TransactionResponse, error) {
   489  	if client == nil {
   490  		return TransactionResponse{}, errNoClientProvided
   491  	}
   492  
   493  	if transaction.freezeError != nil {
   494  		return TransactionResponse{}, transaction.freezeError
   495  	}
   496  
   497  	if !transaction.IsFrozen() {
   498  		_, err := transaction.FreezeWith(client)
   499  		if err != nil {
   500  			return TransactionResponse{}, err
   501  		}
   502  	}
   503  
   504  	transactionID := transaction.transactionIDs._GetCurrent().(TransactionID)
   505  
   506  	if !client.GetOperatorAccountID()._IsZero() && client.GetOperatorAccountID()._Equals(*transactionID.AccountID) {
   507  		transaction.SignWith(
   508  			client.GetOperatorPublicKey(),
   509  			client.operator.signer,
   510  		)
   511  	}
   512  
   513  	resp, err := _Execute(
   514  		client,
   515  		&transaction.Transaction,
   516  		_TransactionShouldRetry,
   517  		_TransactionMakeRequest,
   518  		_TransactionAdvanceRequest,
   519  		_TransactionGetNodeAccountID,
   520  		_ContractCreateTransactionGetMethod,
   521  		_TransactionMapStatusError,
   522  		_TransactionMapResponse,
   523  		transaction._GetLogID(),
   524  		transaction.grpcDeadline,
   525  		transaction.maxBackoff,
   526  		transaction.minBackoff,
   527  		transaction.maxRetry,
   528  	)
   529  
   530  	if err != nil {
   531  		return TransactionResponse{
   532  			TransactionID:  transaction.GetTransactionID(),
   533  			NodeID:         resp.(TransactionResponse).NodeID,
   534  			ValidateStatus: true,
   535  		}, err
   536  	}
   537  
   538  	return TransactionResponse{
   539  		TransactionID:  transaction.GetTransactionID(),
   540  		NodeID:         resp.(TransactionResponse).NodeID,
   541  		Hash:           resp.(TransactionResponse).Hash,
   542  		ValidateStatus: true,
   543  	}, nil
   544  }

   550  func (transaction *ContractCreateTransaction) FreezeWith(client *Client) (*ContractCreateTransaction, error) {
